87int TvCtrlPointDeleteNode(TvDeviceNode* node) {
88 int rc, service, var;
89
90 if (NULL == node) {
91 SampleUtil_Print("ERROR: TvCtrlPointDeleteNode: Node is empty\n");
92 return TV_ERROR;
93 }
94
95 for (service = 0; service < TV_SERVICE_SERVCOUNT; service++) {
96 /*
97 If we have a valid control SID, then unsubscribe
98 */
99 if (strcmp(node->device.TvService[service].SID, "") != 0) {
100 rc = UpnpUnSubscribe(ctrlpt_handle,
101 node->device.TvService[service].SID);
102 if (UPNP_E_SUCCESS == rc) {
103 SampleUtil_Print("Unsubscribed from Tv %s "
104 "EventURL with SID=%s\n",
105 TvServiceName[service],
106 node->device.TvService[service].SID);
107 } else {
108 SampleUtil_Print("Error unsubscribing to Tv %s "
109 "EventURL -- %d\n",
110 TvServiceName[service], rc);
111 }
112 }
113
114 for (var = 0; var < TvVarCount[service]; var++) {
115 if (node->device.TvService[service].VariableStrVal[var]) {
116 free(node->device.TvService[service].VariableStrVal[var]);
117 }
118 }
119 }
120
121 /*Notify New Device Added */
122 SampleUtil_StateUpdate(NULL, NULL, node->device.UDN, DEVICE_REMOVED);
123 free(node);
124 node = NULL;
125
126 return TV_SUCCESS;
127}

129int TvCtrlPointRemoveDevice(const char* UDN) {
130 struct TvDeviceNode* curdevnode;
131 struct TvDeviceNode* prevdevnode;
132
133 pthread_mutex_lock(&DeviceListMutex);
134
135 curdevnode = GlobalDeviceList;
136 if (!curdevnode) {
137 SampleUtil_Print("WARNING: TvCtrlPointRemoveDevice: Device "
138 "list empty\n");
139 } else {
140 if (0 == strcmp(curdevnode->device.UDN, UDN)) {
141 GlobalDeviceList = curdevnode->next;
142 TvCtrlPointDeleteNode(curdevnode);
143 } else {
144 prevdevnode = curdevnode;
145 curdevnode = curdevnode->next;
146 while (curdevnode) {
147 if (strcmp(curdevnode->device.UDN, UDN) == 0) {
148 prevdevnode->next = curdevnode->next;
149 TvCtrlPointDeleteNode(curdevnode);
150 break;
151 }
152 prevdevnode = curdevnode;
153 curdevnode = curdevnode->next;
154 }
155 }
156 }
157
158 pthread_mutex_unlock(&DeviceListMutex);
159
160 return TV_SUCCESS;
161}

261 if (TV_SUCCESS == rc) {
262 if (0 == param_count) {
263 actionNode =

265 } else {
266 for (param = 0; param < param_count; param++) {
267 if (UpnpAddToAction(&actionNode, actionname,
268 TvServiceType[service], param_name[param],
269 param_val[param]) != UPNP_E_SUCCESS) {
270 SampleUtil_Print("ERROR: TvCtrlPointSendAction: "
271 "Trying to add action param\n");
272 /*return -1; // TBD - BAD! leaves mutex

274 }
275 }
276 }
277
278 rc = UpnpSendActionAsync(ctrlpt_handle,
279 devnode->device.TvService[service].ControlURL,
280 TvServiceType[service], NULL, actionNode,
281 TvCtrlPointCallbackEventHandler, NULL);
282
283 if (rc != UPNP_E_SUCCESS) {
284 SampleUtil_Print("Error in UpnpSendActionAsync -- %d\n", rc);
285 rc = TV_ERROR;
286 }
287 }
288
289 pthread_mutex_unlock(&DeviceListMutex);
290
291 if (actionNode)
292 ixmlDocument_free(actionNode);

455void TvCtrlPointAddDevice(IXML_Document* DescDoc, const char* location,
456 int expires) {
457 char* deviceType{};
458 char* friendlyName{};
459 char* presURL{};
460 char* baseURL{};
461 char* relURL{};
462 char* UDN{};
463 char* serviceId[TV_SERVICE_SERVCOUNT]{};
464 char* eventURL[TV_SERVICE_SERVCOUNT]{};
465 char* controlURL[TV_SERVICE_SERVCOUNT]{};
466 Upnp_SID eventSID[TV_SERVICE_SERVCOUNT];
467 int TimeOut[TV_SERVICE_SERVCOUNT];
468 struct TvDeviceNode* deviceNode{};
469 struct TvDeviceNode* tmpdevnode{};
470 int ret = 1;
471 int found = 0;
472 int service;
473 int var;
474
475 TimeOut[0] = default_timeout;
476 TimeOut[1] = default_timeout;
477 pthread_mutex_lock(&DeviceListMutex);
478
479 /* Read key elements from description document */
480 UDN = SampleUtil_GetFirstDocumentItem(DescDoc, "UDN");
481 deviceType = SampleUtil_GetFirstDocumentItem(DescDoc, "deviceType");
482 friendlyName = SampleUtil_GetFirstDocumentItem(DescDoc, "friendlyName");
483 baseURL = SampleUtil_GetFirstDocumentItem(DescDoc, "URLBase");
484 relURL = SampleUtil_GetFirstDocumentItem(DescDoc, "presentationURL");
485
486 ret = UpnpResolveURL2((baseURL ? baseURL : location), relURL, &presURL);
487
488 if (UPNP_E_SUCCESS != ret)
489 SampleUtil_Print(
490 "%s(%d): Error generating presURL from %s + %s, %s(%d)\n", __FILE__,
491 __LINE__, baseURL, relURL, UpnpGetErrorMessage(ret), ret);
492
493 if (strcmp(deviceType, TvDeviceType) == 0) {
494 SampleUtil_Print("Found Tv device\n");
495
496 /* Check if this device is already in the list */
497 tmpdevnode = GlobalDeviceList;
498 while (tmpdevnode) {
499 if (strcmp(tmpdevnode->device.UDN, UDN) == 0) {
500 found = 1;
501 break;
502 }
503 tmpdevnode = tmpdevnode->next;
504 }
505
506 if (found) {
507 /* The device is already there, so just update */
508 /* the advertisement timeout field */
509 tmpdevnode->device.AdvrTimeOut = expires;
510 } else {
511 for (service = 0; service < TV_SERVICE_SERVCOUNT; service++) {
512 if (SampleUtil_FindAndParseService(
513 DescDoc, location, TvServiceType[service],
514 &serviceId[service], &eventURL[service],
515 &controlURL[service])) {
516 SampleUtil_Print("Subscribing to "
517 "EventURL %s...\n",
518 eventURL[service]);
519 ret = UpnpSubscribe(ctrlpt_handle, eventURL[service],
520 &TimeOut[service], eventSID[service]);
521 if (ret == UPNP_E_SUCCESS) {
522 SampleUtil_Print("Subscribed to "
523 "EventURL with "
524 "SID=%s\n",
525 eventSID[service]);
526 } else {
527 SampleUtil_Print("Error Subscribing to "
528 "EventURL -- %d\n",
529 ret);
530 strcpy(eventSID[service], "");
531 }
532 } else {
533 SampleUtil_Print("Error: Could not "
534 "find Service: %s\n",
535 TvServiceType[service]);
536 }
537 }
538 /* Create a new device node */
539 deviceNode =
540 (struct TvDeviceNode*)malloc(sizeof(struct TvDeviceNode));
541 strcpy(deviceNode->device.UDN, UDN);
542 strcpy(deviceNode->device.DescDocURL, location);
543 strcpy(deviceNode->device.FriendlyName, friendlyName);
544 // BUGFIX: Next could segfault
545 if (presURL)
546 strcpy(deviceNode->device.PresURL, presURL);
547 deviceNode->device.AdvrTimeOut = expires;
548 for (service = 0; service < TV_SERVICE_SERVCOUNT; service++) {
549 if (serviceId[service] == NULL) {
550 /* not found */
551 continue;
552 }
553 strcpy(deviceNode->device.TvService[service].ServiceId,
554 serviceId[service]);
555 strcpy(deviceNode->device.TvService[service].ServiceType,
556 TvServiceType[service]);
557 strcpy(deviceNode->device.TvService[service].ControlURL,
558 controlURL[service]);
559 strcpy(deviceNode->device.TvService[service].EventURL,
560 eventURL[service]);
561 strcpy(deviceNode->device.TvService[service].SID,
562 eventSID[service]);
563 for (var = 0; var < TvVarCount[service]; var++) {
564 deviceNode->device.TvService[service].VariableStrVal[var] =
565 (char*)malloc(TV_MAX_VAL_LEN);
566 strcpy(deviceNode->device.TvService[service]
567 .VariableStrVal[var],
568 "");
569 }
570 }
571 deviceNode->next = NULL;
572 /* Insert the new device node in the list */
573 if ((tmpdevnode = GlobalDeviceList) != nullptr) {
574 while (tmpdevnode) {
575 if (tmpdevnode->next) {
576 tmpdevnode = tmpdevnode->next;
577 } else {
578 tmpdevnode->next = deviceNode;
579 break;
580 }
581 }
582 } else {
583 GlobalDeviceList = deviceNode;
584 }
585 /*Notify New Device Added */
586 SampleUtil_StateUpdate(NULL, NULL, deviceNode->device.UDN,
587 DEVICE_ADDED);
588 }
589 }
590
591 pthread_mutex_unlock(&DeviceListMutex);
592
593 if (deviceType)
594 free(deviceType);
595 if (friendlyName)
596 free(friendlyName);
597 if (UDN)
598 free(UDN);
599 if (baseURL)
600 free(baseURL);
601 if (relURL)
602 free(relURL);
603 if (presURL)
604 free(presURL);
605 for (service = 0; service < TV_SERVICE_SERVCOUNT; service++) {
606 if (serviceId[service])
607 free(serviceId[service]);
608 if (controlURL[service])
609 free(controlURL[service]);
610 if (eventURL[service])
611 free(eventURL[service]);
612 }
613}

615void TvStateUpdate(char* UDN, int Service, IXML_Document* ChangedVariables,
616 char** State) {
617 IXML_NodeList* properties;
618 IXML_NodeList* variables;
619 IXML_Element* property;
620 IXML_Element* variable;
621 long unsigned int length;
622 long unsigned int length1;
623 long unsigned int i;
624 int j;
625 char* tmpstate = NULL;
626 (void)UDN;
627
628 SampleUtil_Print("Tv State Update (service %d):\n", Service);
629 /* Find all of the e:property tags in the document */
630 properties =
631 ixmlDocument_getElementsByTagName(ChangedVariables, "e:property");
632 if (properties) {
633 length = ixmlNodeList_length(properties);
634 for (i = 0; i < length; i++) {
635 /* Loop through each property change found */
636 property = (IXML_Element*)ixmlNodeList_item(properties, i);
637 /* For each variable name in the state table,
638 * check if this is a corresponding property change */
639 for (j = 0; j < TvVarCount[Service]; j++) {
640 variables = ixmlElement_getElementsByTagName(
641 property, TvVarName[Service][j]);
642 /* If a match is found, extract
643 * the value, and update the state table */
644 if (variables) {
645 length1 = ixmlNodeList_length(variables);
646 if (length1) {
647 variable =
648 (IXML_Element*)ixmlNodeList_item(variables, 0);
649 tmpstate = SampleUtil_GetElementValue(variable);
650 if (tmpstate) {
651 strcpy(State[j], tmpstate);
652 SampleUtil_Print(" Variable "
653 "Name: %s New "
654 "Value:'%s'\n",
655 TvVarName[Service][j], State[j]);
656 }
657 if (tmpstate)
658 free(tmpstate);
659 tmpstate = NULL;
660 }
661 ixmlNodeList_free(variables);
662 variables = NULL;
663 }
664 }
665 }
666 ixmlNodeList_free(properties);
667 }
668 return;
669}

887void TvCtrlPointVerifyTimeouts(int incr) {
888 struct TvDeviceNode* prevdevnode;
889 struct TvDeviceNode* curdevnode;
890 int ret;
891
892 pthread_mutex_lock(&DeviceListMutex);
893
894 prevdevnode = NULL;
895 curdevnode = GlobalDeviceList;
896 while (curdevnode) {
897 curdevnode->device.AdvrTimeOut -= incr;
898 /*SampleUtil_Print("Advertisement Timeout: %d\n",
899 * curdevnode->device.AdvrTimeOut); */
900 if (curdevnode->device.AdvrTimeOut <= 0) {
901 /* This advertisement has expired, so we should remove
902 * the device from the list */
903 if (GlobalDeviceList == curdevnode)
904 GlobalDeviceList = curdevnode->next;
905 else
906 prevdevnode->next = curdevnode->next;
907 TvCtrlPointDeleteNode(curdevnode);
908 if (prevdevnode)
909 curdevnode = prevdevnode->next;
910 else
911 curdevnode = GlobalDeviceList;
912 } else {
913 if (curdevnode->device.AdvrTimeOut < 2 * incr) {
914 /* This advertisement is about to expire, so
915 * send out a search request for this device
916 * UDN to try to renew */
917 ret = UpnpSearchAsync(ctrlpt_handle, incr,
918 curdevnode->device.UDN, NULL);
919 if (ret != UPNP_E_SUCCESS)
920 SampleUtil_Print("Error sending search request "
921 "for Device UDN: %s -- err = "
922 "%d\n",
923 curdevnode->device.UDN, ret);
924 }
925 prevdevnode = curdevnode;
926 curdevnode = curdevnode->next;
927 }
928 }
929
930 pthread_mutex_unlock(&DeviceListMutex);
931}
