120int UpnpGetIfInfo(
123 std::string_view a_iface = "") //
124{
125 TRACE("Executing compa::UpnpGetIfInfo()")
126 UPnPsdk::CNetadapter nadaptObj;
127 try {
128 nadaptObj.get_first(); // May throw exception
129
130 if (!nadaptObj.find_first(a_iface) &&
131 !nadaptObj.find_first(
132 static_cast<unsigned int>(std::stoi(std::string(a_iface))))) {
133 UPnPsdk_LOGERR("MSG1033") "Local network interface \""
134 << a_iface << "\" not found.\n";
135 return UPNP_E_INVALID_INTERFACE;
136 }
137
138 // Get gIF_NAME and gIF_INDEX
139 ::memset(gIF_NAME, 0, sizeof(gIF_NAME));
140 ::strncpy(gIF_NAME, nadaptObj.name().c_str(), sizeof(gIF_NAME) - 1);
141 gIF_INDEX = nadaptObj.index();
142
143 // With default lookup restrict finding to the current selected adapter.
144 if (a_iface.empty())
145 nadaptObj.find_first(gIF_INDEX);
146
147 ::memset(gIF_IPV6, 0, sizeof(gIF_IPV6));
148 ::memset(gIF_IPV6_ULA_GUA, 0, sizeof(gIF_IPV6_ULA_GUA));
149 ::memset(gIF_IPV4, 0, sizeof(gIF_IPV4));
150 ::memset(gIF_IPV4_NETMASK, 0, sizeof(gIF_IPV4_NETMASK));
151
152 UPnPsdk::SSockaddr saObj;
153 do {
154 nadaptObj.sockaddr(saObj);
155 switch (saObj.ss.ss_family) {
156 case AF_INET6:
157 // The loopback address belongs to link-local unicast
158 // addresses.
159 if (gIF_IPV6[0] == '\0' &&
160 (IN6_IS_ADDR_LINKLOCAL(&saObj.sin6.sin6_addr) ||
161 IN6_IS_ADDR_LOOPBACK(&saObj.sin6.sin6_addr))) {
162 // Strip leading bracket on copying.
163 ::strncpy(gIF_IPV6, saObj.netaddr().c_str() + 1,
164 sizeof(gIF_IPV6) - 1);
165 // Strip trailing scope id if any.
166 if (char* chptr{::strchr(gIF_IPV6, '%')})
167 *chptr = '\0';
168 // Strip trailing bracket if any.
169 else if (char* chptr{::strchr(gIF_IPV6, ']')})
170 *chptr = '\0';
171 gIF_IPV6_PREFIX_LENGTH = nadaptObj.bitmask();
172 } else if (gIF_IPV6_ULA_GUA[0] == '\0') {
173 // Strip leading bracket on copying.
174 ::strncpy(gIF_IPV6_ULA_GUA, saObj.netaddr().c_str() + 1,
175 sizeof(gIF_IPV6_ULA_GUA) - 1);
176 // Strip trailing scope id if any.
177 if (char* chptr{::strchr(gIF_IPV6, '%')})
178 *chptr = '\0';
179 // Strip trailing bracket if any.
180 else if (char* chptr{::strchr(gIF_IPV6_ULA_GUA, ']')})
181 *chptr = '\0';
182 gIF_IPV6_ULA_GUA_PREFIX_LENGTH = nadaptObj.bitmask();
183 }
184 break;
185
186 case AF_INET:
187 if (gIF_IPV4[0] == '\0') {
188 ::strncpy(gIF_IPV4, saObj.netaddr().c_str(),
189 sizeof(gIF_IPV4) - 1);
190 nadaptObj.socknetmask(saObj);
191 ::strncpy(gIF_IPV4_NETMASK, saObj.netaddr().c_str(),
192 sizeof(gIF_IPV4_NETMASK) - 1);
193 }
194 break;
195
196 default:
197 UPnPsdk_LOGCRIT("MSG1029") "Unsupported address family("
198 << saObj.ss.ss_family << "), only AF_INET6(" << AF_INET6
199 << ") or AF_INET(" << AF_INET << ") are valid.\n";
200 return UPNP_E_INVALID_INTERFACE;
201 } // switch
202
203 } while (nadaptObj.find_next() &&
204 (gIF_IPV6[0] == '\0' || gIF_IPV6_ULA_GUA[0] == '\0' ||
205 gIF_IPV4[0] == '\0'));
206
207 } catch (const std::exception& ex) {
208 UPnPsdk_LOGCATCH("MSG1006") "catched next line...\n" << ex.what();
209 return UPNP_E_INVALID_INTERFACE;
210 }
211
212 return UPNP_E_SUCCESS;
213}

483 UpnpSdkInit = 0;

501static int UpnpInitPreamble() {
502 TRACE("Executing UpnpInitPreamble()")
503 int retVal = UPNP_E_SUCCESS;
504
505 /* needed by SSDP or other parts. */
506 srand((unsigned int)time(NULL));
507
508 /* Initialize debug output. */
509 retVal = UpnpInitLog();
510 if (retVal != UPNP_E_SUCCESS) {
511 /* UpnpInitLog does not return a valid UPNP_E_*. */
512 return UPNP_E_INIT_FAILED;
513 }
514
515 /* Initialize SDK global mutexes. */
516 retVal = UpnpInitMutexes();
517 if (retVal != UPNP_E_SUCCESS) {
518 return retVal;
519 }
520
521#ifdef COMPA_HAVE_OPTION_SSDP
522 /* Create the NLS uuid. */
523 uuid_upnp nls_uuid;
524 uuid_create(&nls_uuid);
525 upnp_uuid_unpack(&nls_uuid, gUpnpSdkNLSuuid);
526#endif
527
528 /* Initializes the handle list. */
529 HandleLock();
530 for (int i{0}; i < NUM_HANDLE; ++i) {
531 compa::HandleTable[i] = nullptr;
532 }
533 HandleUnlock();
534
535 /* Initialize SDK global thread pools. */
536 retVal = UpnpInitThreadPools();
537 if (retVal != UPNP_E_SUCCESS) {
538 return retVal;
539 }
540
541#ifdef COMPA_HAVE_DEVICE_SOAP
542 SetSoapCallback(soap_device_callback);
543#endif
544
545#ifdef COMPA_HAVE_DEVICE_GENA
546 SetGenaCallback(genaCallback);
547#endif
548
549 /* Initialize the SDK timer thread. */
550 retVal = TimerThreadInit(&gTimerThread, &gSendThreadPool);
551 if (retVal != UPNP_E_SUCCESS) {
552 UpnpFinish();
553
554 return retVal;
555 }
556
557 return UPNP_E_SUCCESS;
558}

598int UpnpInit2(const char* IfName, unsigned short DestPort) {
599 UPnPsdk_LOGINFO("MSG1096") "Executing...\n";
600 int retVal;
601
602 // The mutex must be initialized.
603 if (pthread_mutex_lock(&compa::sdkInit_mutex) != 0) {
604 retVal = UPNP_E_INIT_FAILED;
605 goto exit_function;
606 }
607
608 /* Check if we're already initialized. */
609 if (UpnpSdkInit == 1) {
610 retVal = UPNP_E_INIT;
611 goto exit_function;
612 }
613
614 /* Set the UpnpSdkInit flag to 1 to indicate we're successfully
615 * initialized. */
616 UpnpSdkInit = 1;
617
618 /* Perform initialization preamble. */
619 retVal = UpnpInitPreamble();
620 if (retVal != UPNP_E_SUCCESS) {
621 goto exit_function;
622 }
623
624 /* Retrieve interface information (Addresses, index, etc). */
625 retVal = compa::UpnpGetIfInfo(IfName == nullptr ? "" : IfName);
626 if (retVal != UPNP_E_SUCCESS) {
627 goto exit_function;
628 }
629
630 /* Finish initializing the SDK. */
631 retVal = UpnpInitStartServers(DestPort);
632 if (retVal != UPNP_E_SUCCESS) {
633 goto exit_function;
634 }
635
636exit_function:
637 if (retVal != UPNP_E_SUCCESS && retVal != UPNP_E_INIT) {
638 UpnpFinish();
639 }
640 pthread_mutex_unlock(&compa::sdkInit_mutex);
641 return retVal;
642}

688int UpnpFinish() {
689 TRACE("Executing UpnpFinish()")
690 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__,
691 "Inside UpnpFinish: UpnpSdkInit is %d\n", UpnpSdkInit);
692#ifdef COMPA_HAVE_DEVICE_SSDP
693 UpnpDevice_Handle device_handle;
694#endif
695#ifdef COMPA_HAVE_CTRLPT_SSDP
696 UpnpClient_Handle client_handle;
697#endif
698 [[maybe_unused]] struct Handle_Info* temp{nullptr};
699#ifdef UPnPsdk_HAVE_OPENSSL
700 freeSslCtx();
701#endif
702 if (UpnpSdkInit != 1)
703 return UPNP_E_FINISH;
704 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__,
705 "UpnpFinish: UpnpSdkInit is ONE\n");
706 // May be usable for DEBUG:
707 // PrintThreadPoolStats(&gSendThreadPool, __FILE__, __LINE__,
708 // "Send Thread Pool");
709 // PrintThreadPoolStats(&gRecvThreadPool, __FILE__, __LINE__,
710 // "Recv Thread Pool");
711 // PrintThreadPoolStats(&gMiniServerThreadPool, __FILE__, __LINE__,
712 // "MiniServer Thread Pool");
713#ifdef COMPA_HAVE_DEVICE_SSDP
714 while (GetDeviceHandleInfo(0, AF_INET, &device_handle, &temp) ==
715 HND_DEVICE) {
716 UpnpUnRegisterRootDevice(device_handle);
717 }
718 while (GetDeviceHandleInfo(0, AF_INET6, &device_handle, &temp) ==
719 HND_DEVICE) {
720 UpnpUnRegisterRootDevice(device_handle);
721 }
722#endif
723#ifdef COMPA_HAVE_CTRLPT_SSDP
724 while (HND_CLIENT == GetClientHandleInfo(&client_handle, &temp)) {
725 UpnpUnRegisterClient(client_handle);
726 }
727#endif
728 TimerThreadShutdown(&gTimerThread);
729#ifdef COMPA_HAVE_MINISERVER
730 StopMiniServer();
731#endif
732#ifdef COMPA_HAVE_WEBSERVER
733 web_server_destroy();
734#endif
735 ThreadPoolShutdown(&gMiniServerThreadPool);
736 PrintThreadPoolStats(&gMiniServerThreadPool, __FILE__, __LINE__,
737 "MiniServer Thread Pool");
738 ThreadPoolShutdown(&gRecvThreadPool);
739 PrintThreadPoolStats(&gSendThreadPool, __FILE__, __LINE__,
740 "Send Thread Pool");
741 ThreadPoolShutdown(&gSendThreadPool);
742 PrintThreadPoolStats(&gRecvThreadPool, __FILE__, __LINE__,
743 "Recv Thread Pool");
744#ifdef COMPA_HAVE_CTRLPT_GENA
745 clientSubscribeMutexDestroy();
746#endif
747 pthread_rwlock_destroy(&GlobalHndRWLock);
748#ifdef COMPA_HAVE_OPTION_SSDP
749 uuidMutexDestroy(); // May fail but not checked due to compatibility.
750#endif
751 UpnpRemoveAllVirtualDirs(); /* remove all virtual dirs */
752 UpnpSdkInit = 0;
753 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__,
754 "Exiting UpnpFinish: UpnpSdkInit is :%d:\n", UpnpSdkInit);
755 UpnpCloseLog();
756
757 return UPNP_E_SUCCESS;
758}

3428 if ((newDirName == NULL) || (strlen(newDirName) == (size_t)0)) {
3429 return UPNP_E_INVALID_PARAM;

3434 return UPNP_E_INVALID_PARAM;
3435 dirName[0] = '/';
3436 strncpy(dirName + 1, newDirName, sizeof(dirName) - 2);
3437 } else {

3439 return UPNP_E_INVALID_PARAM;

3449 pCurVirtualDir->cookie = cookie;

3452
