23bool INetadapter::find_first(const std::string& a_name_or_addr) {
24 TRACE2(this, " Executing INetadapter::find_first(" + a_name_or_addr + ")")
25 // First look for a local network adapter name
26 this->reset();
27 do {
28 if (this->name() == a_name_or_addr)
29 return true;
30 } while (this->get_next());
31
32 // No name found, look for the ip address of a local network adapter.
33 // Try to translate input argument to a socket address.
34 CAddrinfo ainfoObj(a_name_or_addr, AI_NUMERICHOST, 0);
35 if (!ainfoObj.get_first())
36 return false;
37
38 // Valid ip address string given as input argument. Get its socket address.
39 SSockaddr sa_inputObj{};
40 ainfoObj.sockaddr(sa_inputObj);
41
42 // Parse network adapter list for the given input argument.
43 SSockaddr sa_nadObj{};
44 this->reset();
45 do {
46 this->sockaddr(sa_nadObj);
47 if (sa_nadObj == sa_inputObj.ss)
48 return true;
49 } while (this->get_next());
50
51 return false;
52}
