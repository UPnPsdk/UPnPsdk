49inline void CAddrinfo::set_ai_flags(const int a_family, const int a_socktype,
50 const int a_flags,
51 const int a_protocol) noexcept {
52 // I cannot use the initialization list of the constructor because the
53 // member order in the structure addrinfo is different on Linux, MacOS and
54 // win32. I have to use the member names to initialize them, what's not
55 // possible for structures in the constructors initialization list.
56 m_hints.ai_flags = a_flags;
57 m_hints.ai_family = a_family;
58 m_hints.ai_socktype = a_socktype;
59 m_hints.ai_protocol = a_protocol;
60}

89// Setter to load an addrinfo from the operating system to the object
90// ------------------------------------------------------------------
91// Get address information with cached hints.
92void CAddrinfo::load() {
93 TRACE2(this, " Executing CAddrinfo::load()")
94
95 std::string node, service;
96 if (m_service.empty()) {
97 split_addr_port(m_node, node, service);
98 } else {
99 split_addr_port(m_node + ":" + m_service, node, service);
100 }
101
102 // syscall ::getaddrinfo() with prepared arguments
103 // -----------------------------------------------
104 ::addrinfo* new_res{nullptr}; // Result from ::getaddrinfo()
105 const int ret = umock::netdb_h.getaddrinfo(
106 (node.empty() ? nullptr : node.c_str()),
107 (service.empty() ? nullptr : service.c_str()), &m_hints, &new_res);
108#ifndef __APPLE__
109 // std::format() since c++20 isn't supported on AppleClang 15, even on c++23
110 TRACE2(this, " syscall ::getaddrinfo(" + node + ", " + service +
111 ") with new_res = " +
112 std::format("{:#x}", reinterpret_cast<uintptr_t>(new_res)))
113#endif
114 if (g_dbug) {
115 // Very helpful for debugging to see what is given to ::getaddrinfo()
116 char addrStr[INET6_ADDRSTRLEN]{};
117 char servStr[NI_MAXSERV]{};
118 if (ret == 0)
119 ::getnameinfo(new_res->ai_addr,
120 static_cast<socklen_t>(new_res->ai_addrlen), addrStr,
121 sizeof(addrStr), servStr, sizeof(servStr),
122 NI_NUMERICHOST | NI_NUMERICSERV);
123 // clang-format off
124 UPnPsdk_LOGINFO << "MSG1111: syscall ::getaddrinfo("
125 << (node.empty() ? "nullptr, " : "\"" + node + "\", ")
126 << (service.empty() ? "nullptr, " : "\"" + service + "\", ")
127 << &m_hints << ", " << &new_res
128 << ") node=\"" << m_node << "\", "
129 << (m_hints.ai_flags & AI_NUMERICHOST ? "AI_NUMERICHOST, " : "")
130 << (m_hints.ai_flags & AI_NUMERICSERV ? "AI_NUMERICSERV, " : "")
131 << (m_hints.ai_flags & AI_PASSIVE ? "AI_PASSIVE, " : "")
132 << (m_hints.ai_family == AF_INET6 ? "AF_INET6" :
133 (m_hints.ai_family == AF_INET ? "AF_INET" :
134 (m_hints.ai_family == AF_UNSPEC ? "AF_UNSPEC" :
135 "m_hints.ai_family=" + std::to_string(m_hints.ai_family))))
136 << ", "
137 << (m_hints.ai_socktype == SOCK_STREAM ? "SOCK_STREAM" :
138 (m_hints.ai_socktype == SOCK_DGRAM ? "SOCK_DGRAM" :
139 (m_hints.ai_socktype == SOCK_RAW ? "SOCK_RAW" :
140 "socktype=" + std::to_string(m_hints.ai_socktype))))
141 << (ret != 0
142 ? ". Get GAI_ERROR(" + std::to_string(ret) + ")"
143 : ". Get first \"" + std::string(addrStr) + "\" port "
144 + std::string(servStr)
145 + (new_res->ai_next == nullptr ? ", no more entries." : ", more entries..."))
146 << '\n';
147 }
148
149 if (ret == EAI_SERVICE /* Servname not supported for ai_socktype */
150 || ret == EAI_NONAME /* Node or service not known */
151 || ret == EAI_MEMORY /* Out of memory */
152 || ret == EAI_AGAIN /* The name server returned a temporary failure indication, try again later */
154#ifndef _MSC_VER
155 || ret == EAI_ADDRFAMILY /* Address family for NAME not supported */
156#endif
157 || ret == EAI_NODATA) /* No address associated with hostname */ {
158 // Error numbers definded in netdb.h.
159 // Maybe an alphanumeric node name that cannot be resolved (e.g. by
160 // DNS)? Anyway, the user has to decide what to do. Because this
161 // depends on extern available DNS server the error can occur
162 // unexpectedly at any time. We have no influence on it but I will give
163 // an extended error message.
164 throw std::runtime_error(UPnPsdk_LOGEXCEPT + "MSG1112: errid(" +
165 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\", " +
166 ((m_hints.ai_family == AF_UNSPEC) ? "IPv?_" :
167 ((m_hints.ai_family == AF_INET6) ? "IPv6_" : "IPv4_")) +
168 ((m_hints.ai_flags & AI_NUMERICHOST) ? "numeric_host=\"" : "alphanum_name=\"") +
169 m_node + "\", service=\"" +
170 m_service + "\"" +
171 ((m_hints.ai_flags & AI_PASSIVE) ? ", passive_listen" : "") +
172 ((m_hints.ai_flags & AI_NUMERICHOST) ? "" : ", (maybe DNS query temporary failed?)"));
173 }
174 // clang-format on
175
176 for (::addrinfo* res{new_res}; res != nullptr; res = res->ai_next) {
177 // Different on platforms: Ubuntu & MacOS return protocol number, win32
178 // returns 0. I just return what was used to call ::getaddrinfo().
179 res->ai_protocol = m_hints.ai_protocol;
180 //
181 // Different on platforms: Ubuntu returns set flags, MacOS & win32
182 // return 0. I just return what was used to call ::getaddrinfo().
183 res->ai_flags = m_hints.ai_flags;
184 }
185
186 // If load() is called the second time then m_res still points to the
187 // previous allocated memory. To avoid a memory leak it must be freed
188 // before pointing to the new allocated memory.
189 this->free_addrinfo();
190 // finaly point to the new address information from the operating
191 // system.
192 m_res = new_res;
193 m_res_current = new_res;
194}

197// Getter for the next available address information
198// -------------------------------------------------
199bool CAddrinfo::get_next() noexcept {
200 if (m_res_current->ai_next == nullptr) {
201 // It doesn't matter if already pointing to m_hints. m_hints->ai_next is
202 // also nullptr.
203 m_res_current = &m_hints;
204 return false;
205 }
206 m_res_current = m_res_current->ai_next;
207 return true;
208}

211// Get netaddress with port from current selcted address information
212// -----------------------------------------------------------------
213std::string CAddrinfo::netaddrp() noexcept {
214 if (m_res == &m_hints)
215 return "";
216
217 char addrStr[INET6_ADDRSTRLEN];
218 char servStr[NI_MAXSERV];
219 int ret = ::getnameinfo(m_res_current->ai_addr,
220 static_cast<socklen_t>(m_res_current->ai_addrlen),
221 addrStr, static_cast<socklen_t>(sizeof(addrStr)),
222 servStr, static_cast<socklen_t>(sizeof(servStr)),
223 NI_NUMERICHOST | NI_NUMERICSERV);
224 if (ret != 0) {
225 UPnPsdk_LOGERR "MSG1130: Failed to get name information: "
226 << gai_strerror(ret) << ". Continue with empty netaddress \"\".\n";
227 return "";
228 }
229
230 switch (m_res_current->ai_family) {
231 case AF_INET6:
232 return "[" + std::string(addrStr) + "]:" + std::string(servStr);
233 case AF_INET:
234 return std::string(addrStr) + ":" + std::string(servStr);
235 case AF_UNSPEC:
236 return ":" + std::string(servStr);
237 }
238
239 UPnPsdk_LOGERR "MSG1033: Unsupported address family "
240 << m_res_current->ai_family
241 << ". Continue with empty netaddress \"\".\n";
242 return "";
243}
