102void remove_dot_segments(std::string& a_path) {
103 // The letters (A., B., ...) are the steps as given by the algorithm in
104 // RFC3986_5.2.4.
105 TRACE("Executing UPnPsdk::remove_dot_segments(\"" + a_path + "\")")
106
107 std::string_view path_sv{a_path};
108 std::string output;
109
110 while (!path_sv.empty()) {
111 // A.
112 if (path_sv.substr(0, 3) == "../") {
113 path_sv.remove_prefix(3);
114 } else if (path_sv.substr(0, 2) == "./") {
115 path_sv.remove_prefix(2);
116 }
117 // B.
118 else if (path_sv.substr(0, 3) == "/./") {
119 path_sv.remove_prefix(2);
120 } else if (path_sv == "/.") {
121 path_sv.remove_suffix(1);
122 }
123 // C.
124 else if (path_sv.substr(0, 4) == "/../") {
125 path_sv.remove_prefix(3);
126 if (!output.empty()) {
127 size_t pos;
128 output.erase((pos = output.find_last_of('/')) ==
129 std::string::npos
130 ? 0
131 : pos); // Remove last segment
132 }
133 } else if (path_sv == "/..") {
134 path_sv.remove_suffix(2);
135 if (!output.empty()) {
136 size_t pos;
137 output.erase((pos = output.find_last_of('/')) ==
138 std::string::npos
139 ? 0
140 : pos); // Remove last segment
141 }
142 }
143 // D.
144 else if (path_sv == "." || path_sv == "..") {
145 path_sv = "";
146 }
147 // E.
148 else {
149 size_t start = path_sv.front() == '/' ? 1 : 0;
150 size_t end = path_sv.find_first_of('/', start);
151 if (end == std::string_view::npos)
152 end = path_sv.size();
153 output += path_sv.substr(0, end);
154 path_sv.remove_prefix(end);
155 }
156 }
157
158 a_path = output;
159}

223std::string_view get_scheme(std::string_view a_uri_sv) {
224 // A scheme, if any, must begin with a letter, must have alphanum
225 // characters, or '-', or '+', or '.', and ends with ':'.
226 // The function returns:
227 // an empty string_view if scheme is undefined,
228 // only separator ':' if the scheme is empty,
229 // a valid scheme string_view that always ends with separator ':'.
230 TRACE("Executing get_scheme(a_uri_sv)")
231
232 size_t pos;
233 if ((pos = a_uri_sv.find_first_of(':')) == std::string_view::npos)
234 // No separator found means the scheme is undefined. The URI-reference
235 // may be a relative reference (RFC3986_4.1).
236 return ""; // Guaranteed to be valid after return with size 0.
237
238 if (pos == 0)
239 // ':' at the first position means the scheme is empty. This is invalid
240 // but will be checked later. I cannot simply return ':' because the
241 // string ":" is only valid within this function.
242 return std::string_view(a_uri_sv.data(), 1);
243
244 // Strip the view to have only the scheme with ':'. If pos > size(), the
245 // behavior is undefined. But that is guarded above.
246 a_uri_sv.remove_suffix(a_uri_sv.size() - pos - 1);
247
248 // Check if the scheme has valid character (RFC3986_3.1.).
249 if (!std::isalpha(static_cast<unsigned char>(a_uri_sv.front()))) {
250 // First character is not alpha. This is not a scheme.
251 return ""; // Guaranteed to be valid after return with size 0.
252 } else {
253 unsigned char ch;
254 for (auto it{a_uri_sv.begin()}; it < a_uri_sv.end() - 1; it++) {
255 ch = static_cast<unsigned char>(*it);
256 if (!(std::isalnum(ch) || (ch == '-') || (ch == '+') ||
257 (ch == '.'))) {
258 // Invalid character for scheme. This is not a scheme.
259 return ""; // Guaranteed to be valid after return with size 0.
260 }
261 }
262 }
263
264 return a_uri_sv;
265}

311std::string_view get_authority(std::string_view a_uri_sv) {
312 // The authority component is preceded by a double slash ("//") and is
313 // terminated by the next slash ('/'), question mark ('?'), or number sign
314 // ('#') character, or by the end of the URI (RFC3986 3.2.). A sender MUST
315 // NOT generate an "http" URI with an empty host identifier. A recipient
316 // that processes such a URI reference MUST reject it as invalid (RFC7230
317 // 2.7.1.). This means an authority component is mandatory for "http[s]".
318 // The function returns:
319 // an empty string_view if authority is undefined,
320 // "//" if the authority is empty,
321 // a valid authority string_view that always starts with "//".
322 TRACE("Executing get_authority(a_uri_sv)")
323
324 auto& npos = std::string_view::npos;
325 size_t pos;
326 if ((pos = a_uri_sv.find("//")) != npos) {
327 // Extract the authority component with its separator "//".
328 a_uri_sv.remove_prefix(pos);
329 // Find end of the authority and remove the rest to the end.
330 if ((pos = a_uri_sv.find_first_of("/?#", 2)) != npos)
331 a_uri_sv.remove_suffix(a_uri_sv.size() - pos);
332
333 // Here we have the authority string-view with separator "//" and end
334 // separator (one of '/', '?', '#', ''). Having authority only "//"
335 // means authority is empty.
336 return a_uri_sv;
337 }
338 // No "//" separator found. An empty authority means it is undefined.
339 return ""; // Guaranteed to be valid after return with size 0.
340}

389CUserinfo::CUserinfo(std::string_view a_uri_sv) {
390 // The user information, if present, starts with "//" and is followed by a
391 // commercial at-sign ("@") that delimits it from the host (RFC3986
392 // 3.2.1.). The userinfo may be undefined (no "@" found within the
393 // authority). I also take an empty userinfo into account (first character
394 // of the authority is "@"). Applications should not render as clear text
395 // any password data after the first colon (:) found within a userinfo
396 // subcomponent unless the data after the colon is the empty string
397 // (indicating no password).
398 TRACE2(this, " Construct CUserinfo(a_uri_sv)")
399
400 std::string_view authority_sv = get_authority(a_uri_sv);
401 if (authority_sv.empty())
402 return; // Undefined authority.
403 authority_sv.remove_prefix(2); // remove authority separator.
404
405 auto& npos = std::string_view::npos;
406 size_t pos;
407 if ((pos = authority_sv.find_first_of("/?#")) != npos)
408 authority_sv.remove_suffix(authority_sv.size() - pos);
409
410 // Check if there is a separator, or if it is on the first position (no
411 // content preceeding).
412 if ((pos = authority_sv.find_first_of('@')) == npos)
413 return; // No userinfo sub-component available.
414
415 if (pos == 0 || authority_sv[0] == ':') {
416 // Separator '@' or ':' (for password) at first position means userinfo
417 // is empty.
418 m_component.clear();
419 m_state = STATE::empty;
420 return;
421 }
422
423 // Extract userinfo
424 authority_sv.remove_suffix(authority_sv.size() - pos);
425
426 // Check special case with clear text password.
427 if ((pos = authority_sv.find_first_of(':')) != npos) {
428 // Here we have found a username with clear text password appended.
429 // Strip deprecated clear text password incl. ':' (RFC3986 3.3.1.).
430 authority_sv.remove_suffix(authority_sv.size() - pos);
431 }
432 m_component = authority_sv;
433 m_state = STATE::avail;
434 return;
435}

446CHost::CHost(std::string_view a_uri_sv) {
447 // The Host information starts with "//" when a userinfo is removed. It
448 // ends with ':' if a port is available, or with the end of the authority,
449 // that is '/', or '?', or '#', or end of uri.
450 TRACE2(this, " Construct CHost(a_uri_sv)")
451
452 std::string_view authority_sv = get_authority(a_uri_sv);
453 if (authority_sv.empty())
454 return; // Undefined authority.
455 authority_sv.remove_prefix(2); // remove authority separator.
456
457 auto& npos = std::string_view::npos;
458 size_t pos;
459 if ((pos = authority_sv.find_first_of("/?#")) != npos)
460 authority_sv.remove_suffix(authority_sv.size() - pos);
461
462 // Strip userinfo from authority string if present.
463 if ((pos = authority_sv.find_first_of('@')) != npos)
464 authority_sv.remove_prefix(pos + 1);
465 // Strip port from authority string if present. I have to look for last
466 // occurrance of the port separator. If there is last ']' instead of ':'
467 // then it is an IPv6 address with colons for the host without port.
468 if ((pos = authority_sv.find_last_of("]:")) != npos &&
469 authority_sv[pos] == ':')
470 authority_sv.remove_suffix(authority_sv.size() - pos);
471
472 // Here we have the extracted host string.
473 if (authority_sv.empty()) {
474 m_component.clear();
475 m_state = STATE::empty;
476 return;
477 }
478
479 m_component = authority_sv;
480 // Normalize host to lower case character (RFC3986_3.2.2.).
481 for (auto it{m_component.begin()}; it < m_component.end(); it++)
482 *it = static_cast<char>(std::tolower(static_cast<unsigned char>(*it)));
483
484 // Check if the host_name is valid.
485 // Check IPv6 address.
486 if (m_component.front() == '[') {
487 SSockaddr saObj;
488 try {
489 saObj = m_component;
490 } catch (const std::exception&) {
491 goto exception;
492 }
493 }
494 // Check IPv4 address and DNS name.
495 // The syntax rule for host is ambiguous because it does not completely
496 // distinguish between an IPv4 address and a reg-name. In order to
497 // disambiguate the syntax, we apply the "first-match-wins" algorithm:
498 // If host matches the rule for IPv4 address, then it should be
499 // considered an IPv4 address literal and not a reg-name.
500 // (RFC3986_3.2.2.)
501 else if (!is_ipv4_addr(m_component) && !is_dns_name(m_component)) {
502 goto exception;
503 }
504
505 m_state = STATE::avail;
506 return;
507
508exception:
509 throw std::invalid_argument(
510 UPnPsdk_LOGEXCEPT(
511 "MSG1160") "URI with invalid host address or host name \"" +
512 m_component + "\".\n");
513}

524CPort::CPort(std::string_view a_uri_sv) {
525 // The Port information starts with ':' and ends with the end of the
526 // authority, that is '/', or '?', or '#', or end of uri. Because there may
527 // be also colons as separator for a password, or within an IPv6 address, I
528 // must look at the last occurrence of a colon within the authority
529 // information that is the separator for the port.
530 TRACE2(this, " Construct CPort(a_uri_sv)")
531
532 std::string_view authority_sv = get_authority(a_uri_sv);
533 if (authority_sv.empty())
534 return; // Undefined authority means also no port.
535 authority_sv.remove_prefix(2); // remove authority separator "//".
536
537 auto& npos = std::string_view::npos;
538 size_t pos;
539 // Remove path, query, fragment if any.
540 if ((pos = authority_sv.find_first_of("/?#")) != npos)
541 authority_sv.remove_suffix(authority_sv.size() - pos);
542
543 // Extract port from authority string if present:
544 // first strip userinfo from authority string if present.
545 if ((pos = authority_sv.find_first_of('@')) != npos)
546 authority_sv.remove_prefix(pos + 1);
547 // I have to look for last occurrance of the port separator. If we find
548 // first ']' then it is an IPv6 address with colons for the host without
549 // port.
550 if ((pos = authority_sv.find_last_of("]:")) == npos ||
551 authority_sv[pos] != ':')
552 return; // No port found. It is undefined.
553
554 // Remove host if any.
555 authority_sv.remove_prefix(pos + 1);
556
557 // Here we have the port string that is defined because we found a ':', but
558 // it may be empty.
559 if (authority_sv.empty()) {
560 m_component.clear();
561 m_state = STATE::empty;
562 return;
563 } else if (authority_sv == "80") {
564 // Default Port 80 is normalized for an undefined port.
565 m_state = STATE::undef;
566 return;
567 }
568
569 // Check if the port string is valid.
570 if (to_port(authority_sv) != 0)
571 throw std::invalid_argument(
572 UPnPsdk_LOGEXCEPT("MSG1164") "Invalid port number. Failed URI=\"" +
573 std::string(a_uri_sv) + "\".\n");
574
575 m_component = authority_sv;
576 m_state = STATE::avail;
577}
