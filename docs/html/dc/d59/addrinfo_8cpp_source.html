30// Constructor for getting an address information from only a netaddress.
31// ----------------------------------------------------------------------
32CAddrinfo::CAddrinfo(std::string_view a_node, const int a_family,
33 const int a_socktype, const int a_flags,
34 const int a_protocol)
35 : m_node(a_node) {
36 // Just do a simple check for a possible port and split it from address
37 // string. Detailed tests will be done with this->init().
38 TRACE2(this, " Construct CAddrinfo() without service")
39
40 size_t pos;
41 // The smalest valid netaddress is "[::]".
42 if (m_node.size() < 4 || (m_node.front() == '[' && m_node.back() == ']')) {
43 // IPv6 without port, use given address string
44 goto set_flags;
45 }
46 pos = m_node.rfind("]:"); // noexcept
47 if (pos != m_node.npos) {
48 // IPv6 with port, split the address string. substr() throws exception
49 // std::out_of_range if pos > size(). We have at least 4 character. pos
50 // is 0-based, size() is 1-based so with "]:" is size()==2, pos==0,
51 // pos+2==size() does not throw.
52 m_service = m_node.substr(pos + 2);
53 m_node = m_node.substr(0, pos + 1);
54 goto set_flags;
55 }
56 pos = m_node.find_last_of(']'); // noexcept
57 if (pos != m_node.npos) {
58 // Maybe IPv6 with any remaining character, split address string
59 m_service = m_node.substr(pos + 1);
60 m_node = m_node.substr(0, pos + 1);
61 goto set_flags;
62 }
63 pos = m_node.find_last_of(':'); // noexcept
64 if (pos != m_node.npos) {
65 // IPv4 or URL name with port, split address string
66 m_service = m_node.substr(pos + 1);
67 m_node = m_node.substr(0, pos);
68 }
69
70set_flags:
71 this->set_ai_flags(a_family, a_socktype, a_flags, a_protocol);
72}

76inline void CAddrinfo::set_ai_flags(const int a_family, const int a_socktype,
77 const int a_flags,
78 const int a_protocol) noexcept {
79 // I cannot use the initialization list of the constructor because the
80 // member order in the structure addrinfo is different on Linux, MacOS and
81 // win32. I have to use the member names to initialize them, what's not
82 // possible for structures in the constructors init list.
83 m_hints.ai_flags = a_flags;
84 m_hints.ai_family = a_family;
85 m_hints.ai_socktype = a_socktype;
86 m_hints.ai_protocol = a_protocol;

115// Setter to initialize an addrinfo and set it to the object
116// ---------------------------------------------------------
117// Get address information with cached hints.
118void CAddrinfo::init() {
119 TRACE2(this, " Executing CAddrinfo::init()")
120
121 // Local working copies: modified node, service, and hints to use for
122 // syscall ::getaddrinfo().
123 std::string node;
124 std::string service;
125 ::addrinfo hints{m_hints}; // user given opgtions have priority
126
127 // Correct weak hints:
128 // Always call upnplib::is_netaddr() with AF_UNSPEC (default argument) to
129 // check all types. I always set AI_NUMERICHOST if it match, no matter what
130 // 'a_family' was requested by the caller. But if AF_UNSPEC was requested,
131 // the found address family will also be set.
132 const sa_family_t addr_family =
133 is_netaddr(m_node, m_hints.ai_family); // noexcept
134 if (addr_family != AF_UNSPEC) { // Here we have a numeric netaddress
135 hints.ai_flags |= AI_NUMERICHOST;
136 if (m_hints.ai_family == AF_UNSPEC) { // Correct ai_family, we know it
137 hints.ai_family = addr_family;
138 }
139 }
140 // Check node name
141 if (addr_family == AF_INET6) {
142 // Here we have only ipv6 node strings representing a numeric ip
143 // address (netaddress) without port that is at least "[::]". Remove
144 // surounding brackets for ::getaddrinfo()
145 node = m_node.substr(1, m_node.length() - 2);
146 } else if (addr_family == AF_INET) {
147 // Here we have a valid ipv4 neetaddress. It can be unmodified given to
148 // ::getaddrinfo(). No need to set AI_NUMERICHOST, it's already done
149 // with getting addr_family above.
150 node = m_node;
151 } else if (is_netaddr('[' + m_node + ']', AF_INET6) == AF_INET6) {
152 // ipv6 Netaddresses without brackets would be accepted by
153 // ::getaddrinfo(). So I make them invalid.
154 hints.ai_flags |= AI_NUMERICHOST;
155 node = m_node + "(no_brackets)";
156 } else if (m_node.find_first_of("[]:") != m_node.npos) { // noexcept
157 // An address string without port (m_node is without port) containing
158 // one of theese characters cannot be a valid alphanumeric URL. It can
159 // only be a numeric address (or be invalid).
160 hints.ai_flags |= AI_NUMERICHOST;
161 node = m_node;
162 } else {
163 // Here we have a non numeric node name (no netaddress). Is it a valid
164 // (maybe alphanumeric) node name? ::getaddrinfo() shall decide it.
165 node = m_node;
166 }
167 const char* c_node = node.empty() ? nullptr : node.c_str();
168 std::string node_out = node.empty() ? "nullptr" : "\"" + node + "\"";
169
170 // Check service/port
171 // I have to do this because ::getaddrinfo() does not detect wrong port
172 // numbers >65535. It silently overruns to port number 0, 1, 2...
173 switch (is_numport(m_service)) {
174 case 0:
175 // Valid numeric port number
176 service = m_service;
177 hints.ai_flags |= AI_NUMERICSERV;
178 break;
179 case 1:
180 // Invalid numeric port number > 65535
181 service = m_service + "(invalid)";
182 hints.ai_flags |= AI_NUMERICSERV;
183 break;
184 default:
185 // Any alphanumeric port name
186 service = m_service;
187 }
188
189 // Result from ::getaddrinfo()
190 ::addrinfo* new_res{nullptr};
191
192 // syscall ::getaddrinfo() with prepared arguments
193 // -----------------------------------------------
194 int ret =
195 umock::netdb_h.getaddrinfo(c_node, service.c_str(), &hints, &new_res);
196 TRACE2("syscall ::getaddrinfo() with new_res = ", new_res)
197
198 // Very helpful for debugging to see what is given to ::getaddrinfo()
199 // clang-format off
200 UPNPLIB_LOGINFO << "MSG1114: syscall ::getaddrinfo(" << node_out
201 << ", " << "\"" << service << "\", "
202 << &hints << ", " << &new_res
203 << ") node=\"" << m_node << "\", "
204 << (hints.ai_flags & AI_NUMERICHOST ? "AI_NUMERICHOST, " : "")
205 << (hints.ai_flags & AI_NUMERICSERV ? "AI_NUMERICSERV, " : "")
206 << (hints.ai_flags & AI_PASSIVE ? "AI_PASSIVE, " : "")
207 << (hints.ai_family == AF_INET6 ? "AF_INET6" :
208 (hints.ai_family == AF_INET ? "AF_INET" :
209 (hints.ai_family == AF_UNSPEC ? "AF_UNSPEC" :
210 "hints.ai_family=" + std::to_string(hints.ai_family))))
211 << (ret != 0
212 ? ". Get GAI_ERROR(" + std::to_string(ret) + ")"
213 : ". Get \"" + to_netaddrp(reinterpret_cast
214 <const sockaddr_storage*>(new_res->ai_addr)) + "\"") << "\n";
215
216 if (ret == EAI_SERVICE /* Servname not supported for ai_socktype */
217 || ret == EAI_NONAME /* Node or service not known */
218 || ret == EAI_AGAIN /* The name server returned a temporary failure indication, try again later */
219 || ret == EAI_NODATA) /* No address associated with hostname */ {
220 // Error numbers definded in netdb.h.
221 // Maybe an alphanumeric node name that cannot be resolved (e.g. by
222 // DNS)? Anyway, the user has to decide what to do. Because this
223 // depends on extern available DNS server the error can occur
224 // unexpectedly at any time. We have no influence on it but I will give
225 // an extended error message.
226 throw std::runtime_error(UPNPLIB_LOGEXCEPT + "MSG1111: errid(" +
227 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\", " +
228 ((hints.ai_family == AF_UNSPEC) ? "IPv?_" :
229 ((hints.ai_family == AF_INET6) ? "IPv6_" : "IPv4_")) +
230 ((hints.ai_flags & AI_NUMERICHOST) ? "numeric_host=\"" : "alphanum_name=\"") +
231 node + "\", service=\"" +
232 service + "\"" +
233 ((hints.ai_flags & AI_PASSIVE) ? ", passive_listen" : "") +
234 ((hints.ai_flags & AI_NUMERICHOST) ? "" : ", (maybe DNS query temporary failed?)"));
235 }
236 // clang-format on
237
238 if (ret != 0) {
239 throw std::invalid_argument(
240 UPNPLIB_LOGEXCEPT +
241 "MSG1037: Failed to get address information: errid(" +
242 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\"\n");
243 }
244 // Different on platforms: man getsockaddr says "Specifying 0 in
245 // hints.ai_socktype indicates that socket addresses of any type can be
246 // returned". Linux returns SOCK_STREAM, MacOS returns SOCK_DGRAM and win32
247 // returns 0. To be portable we always return 0 in this case, the same as
248 // requested by the user.
249 if (m_hints.ai_socktype == 0)
250 new_res->ai_socktype = 0;
251 // Different on platforms: Ubuntu & MacOS return protocol number, win32
252 // returns 0. We just return what was requested by the user.
253 new_res->ai_protocol = m_hints.ai_protocol;
254 // Different on platforms: Ubuntu returns set flags, MacOS & win32 return 0.
255 // We just return what was used to query ::getaddrinfo():
256 new_res->ai_flags = hints.ai_flags;
257 // Man getaddrinfo says: "If service is NULL, then the port number of the
258 // returned socket addresses will be left uninitialized." The service is
259 // never set to NULL so we always have a defined service/portnumber.
260 // if (service.empty())
261 // // port for AF_INET6 is also valid for AF_INET
262 // reinterpret_cast<sockaddr_in6*>(new_res->ai_addr)->sin6_port = 0;
263
264 // If init() is called the second time then m_res still points to the
265 // previous allocated memory. To avoid a memory leak it must be freed
266 // before pointing to the new allocated memory.
267 this->free_addrinfo();
268 // finaly point to the new address information from the operating
269 // system.
270 m_res = new_res;
271}

273// Getter for the assosiated ip address with port
274// ----------------------------------------------
275// e.g. "[2001:db8::2]:50001" or "192.168.254.253:50001".
276Netaddr CAddrinfo::netaddr() const noexcept {
277 // TRACE not usable with chained output.
278 // TRACE2(this, " Executing SSockaddr::get_netaddrp()")
279 Netaddr netaddr;
280 if (m_res == &m_hints)
281 return netaddr; // no information available
282
283 netaddr.m_netaddrp = to_netaddrp( // noexcept
284 reinterpret_cast<sockaddr_storage*>(m_res->ai_addr));
285
286 return netaddr;
287}
