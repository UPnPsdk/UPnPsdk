27void CNetadapter_platform::get_first() {
28 // For the structure look at
29 // REF:_https://docs.microsoft.com/en-us/windows/win32/api/iptypes/ns-iptypes-ip_adapter_addresses_lh
30 TRACE2(this, " Executing CNetadapter_platform::get_first()")
31
32 ULONG adapts_size{};
33 // Get Adapters addresses required size. Check with adapts_size = 0
34 // will fail with ERROR_BUFFER_OVERFLOW but returns required size.
35 ULONG ret = umock::iphlpapi_h.GetAdaptersAddresses(
36 AF_UNSPEC, GAA_FLAG_SKIP_ANYCAST | GAA_FLAG_SKIP_DNS_SERVER, nullptr,
37 m_adapt_first, &adapts_size);
38 if (ret != ERROR_BUFFER_OVERFLOW) {
39 throw std::runtime_error(
40 UPnPsdk_LOGEXCEPT +
41 "MSG1120: failed to get buffer size for list of adapters (errid=" +
42 std::to_string(ret) + ").");
43 }
44
45 this->free_adaptaddrs();
46
47 // Allocate enough memory that size was detected above.
48 m_adapt_first = static_cast<PIP_ADAPTER_ADDRESSES>(malloc(adapts_size));
49 if (m_adapt_first == nullptr) {
50 throw std::runtime_error(
51 UPnPsdk_LOGEXCEPT +
52 "MSG1121: failed to allocate memory for list of adapters.");
53 }
54 // Do the call that will actually return the info.
55 ret = umock::iphlpapi_h.GetAdaptersAddresses(
56 AF_UNSPEC, GAA_FLAG_SKIP_ANYCAST | GAA_FLAG_SKIP_DNS_SERVER, nullptr,
57 m_adapt_first, &adapts_size);
58 if (ret != ERROR_SUCCESS) {
59 throw std::runtime_error(
60 UPnPsdk_LOGEXCEPT +
61 "MSG1122: failed to find list of adapters (errid=" +
62 std::to_string(ret) + ").");
63 }
64
65 m_adapt_current = m_adapt_first;
66 m_unicastaddr_current = m_adapt_current->FirstUnicastAddress;
67}

69bool CNetadapter_platform::get_next() {
70 // A network adapter has several IP addresses. First the address list must
71 // be parsed and when at its end the next adapter must be selected. When
72 // entering this function there is no synchronous information as given in a
73 // loop (or two nested). I have to excamine the current state for the next
74 // action. Writing down a state table on paper has helped me.
75 TRACE2(this, " Executing CNetadapter_platform::get_next()")
76 if (m_adapt_current == nullptr) {
77 m_unicastaddr_current = nullptr;
78 return false;
79 }
80 if (m_adapt_current != nullptr && m_unicastaddr_current == nullptr) {
81 m_adapt_current = m_adapt_current->Next;
82 if (m_adapt_current == nullptr) {
83 return false;
84 } else {
85 m_unicastaddr_current = m_adapt_current->FirstUnicastAddress;
86 return true;
87 }
88 }
89 if (m_adapt_current != nullptr && m_unicastaddr_current != nullptr) {
90 m_unicastaddr_current = m_unicastaddr_current->Next;
91 if (m_unicastaddr_current == nullptr) {
92 m_adapt_current = m_adapt_current->Next;
93 if (m_adapt_current == nullptr) {
94 return false;
95 } else {
96 m_unicastaddr_current = m_adapt_current->FirstUnicastAddress;
97 return true;
98 }
99 } else {
100 return true;
101 }
102 }
103 throw std::invalid_argument(
104 UPnPsdk_LOGEXCEPT +
105 "MSG1123: Failed to get next network adapter entry. This should never "
106 "come up and must be fixed. m_adapt_current=" +
107 std::to_string(reinterpret_cast<uintptr_t>(m_adapt_current)) +
108 ", m_unicastaddr_current=" +
109 std::to_string(reinterpret_cast<uintptr_t>(m_unicastaddr_current)) +
110 "\n");
111}

113bool CNetadapter_platform::find_first(const std::string& a_name_or_addr) {
114 TRACE2(this, " Executing CNetadapter_platform::find_first(" +
115 a_name_or_addr + ")")
116 // First look for a local network adapter name
117 this->reset();
118 do {
119 if (this->name() == a_name_or_addr)
120 return true;
121 } while (this->get_next());
122
123 // No name found, look for the ip address of a local network adapter.
124 // Try to translate input argument to a socket address.
125 CAddrinfo ainfoObj(a_name_or_addr, AI_NUMERICHOST, 0);
126 if (!ainfoObj.get_first())
127 return false;
128
129 // Valid ip address string given as input argument. Get its socket address.
130 SSockaddr sa_inputObj{};
131 ainfoObj.sockaddr(sa_inputObj);
132
133 // Parse network adapter list for the given input argument.
134 SSockaddr sa_nadObj{};
135 this->reset();
136 do {
137 this->sockaddr(sa_nadObj);
138 if (sa_nadObj == sa_inputObj)
139 return true;
140 } while (this->get_next());
141
142 return false;
143}

159std::string CNetadapter_platform::name() const {
160 TRACE2(this, " Executing CNetadapter_platform::name()")
161 if (m_adapt_current == nullptr)
162 return "";
163
164 /* Partial fix for Windows: Friendly name is wchar string, but currently
165 * compatible names are char string. For now try to convert it, which will
166 * work with many (but not all) adapters. A full fix would require a lot of
167 * big changes (using UTF-8 as planned). */
168
169 // If I need this constant more than once, I'll make it global, Here it is
170 // used for the friendly name of a network adapter and is compatible to old
171 // gIF_NAME size. ('grep' is your friend)
172 constexpr size_t LINE_SIZE{180};
173
174 char if_name[LINE_SIZE]{};
175 size_t i;
176 wcstombs_s(&i, if_name, sizeof(if_name), m_adapt_current->FriendlyName,
177 sizeof(if_name) - 1); // -1 so the appended '\0' doesn't fall
178 // outside the allocated buffer
179 return if_name;
180}

214unsigned int CNetadapter_platform::index() const {
215 TRACE2(this, " Executing CNetadapter_platform::index()")
216 if (m_adapt_current == nullptr)
217 return 0;
218 // No matter if the adapter supports only IPv4 or only IPv6 or both, the
219 // adapter index should be always the same.
220 if (m_adapt_current->IfIndex != 0) // IPv4 interface
221 return m_adapt_current->IfIndex;
222 if (m_adapt_current->Ipv6IfIndex != 0) // IPv6 interface
223 return m_adapt_current->Ipv6IfIndex;
224 return 0; // neither IPv4 nor IPv6
225}
