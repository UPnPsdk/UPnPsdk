71/* This is called from UpnpInit2(). So the user must call UpnpSetLogFileNames()
72 * before. This can be called again, for example to rotate the log
73 * file, and we try to avoid multiple calls to the mutex init, with a
74 * risk of race, probably not a problem, and not worth fixing. */
75int UpnpInitLog() {
76 TRACE("Executing UpnpInitLog()")
77 if (!initwascalled) {
78 umock::pthread_h.pthread_mutex_init(&GlobalDebugMutex, NULL);
79 initwascalled = 1;
80 }
81 /* If the user did not ask for logging do nothing */
82 if (setlogwascalled == 0) {
83 return UPNP_E_SUCCESS;
84 }
85
86 if (filed != nullptr && filed != stderr) {
87 umock::stdio_h.fclose(filed);
88 filed = nullptr;
89 }
90
91 if (fileName) {
92 filed = umock::stdio_h.fopen(fileName, "a");
93 if (filed == nullptr) {
94 fprintf(stderr,
95 "UPnPlib Error: failed to open filename \"%s\": %s. "
96 "Fallback to stderr.\n",
97 fileName, std::strerror(errno));
98 }
99 }
100
101 // If not set, always set to stderr, also as fallback for a wrong fileName.
102 if (filed == nullptr)
103 filed = stderr;
104
105 return UPNP_E_SUCCESS;
106}

114void UpnpCloseLog() {
115 TRACE("Executing UpnpCloseLog()")
116
117 /* Calling lock() assumes that someone called UpnpInitLog(), but
118 * this is reasonable as it is called from UpnpInit2(). We risk a
119 * crash if we do this without a lock.*/
120 if (initwascalled)
121 umock::pthread_h.pthread_mutex_lock(&GlobalDebugMutex);
122
123 if (filed != nullptr && filed != stderr) {
124 umock::stdio_h.fclose(filed);
125 }
126 filed = nullptr;
127 setlogwascalled = 0;
128 int init_called = initwascalled;
129 initwascalled = 0;
130 if (init_called) {
131 umock::pthread_h.pthread_mutex_unlock(&GlobalDebugMutex);
132 umock::pthread_h.pthread_mutex_destroy(&GlobalDebugMutex);
133 }
134}

242void UpnpPrintf(Upnp_LogLevel DLevel, Dbg_Module Module,
243 const char* DbgFileName, int DbgLineNo, const char* FmtStr,
244 ...) {
245 /*fprintf(stderr, "UpnpPrintf: filed %p level %d glev %d mod %d DEBUG_ALL
246 %d\n", filed, DLevel, g_log_level, Module, DEBUG_ALL);*/
247 va_list ArgList;
248
249 if (!initwascalled) {
250 return;
251 }
252
253 if (!DebugAtThisLevel(DLevel, Module))
254 return;
255 umock::pthread_h.pthread_mutex_lock(&GlobalDebugMutex);
256 if (filed == nullptr) {
257 umock::pthread_h.pthread_mutex_unlock(&GlobalDebugMutex);
258 return;
259 }
260
261 va_start(ArgList, FmtStr);
262 if (DbgFileName) {
263 // flush stdout to have a sequencial output with stderr on screen.
264 fflush(stdout); // Don't mock this because we use it for debuging.
265 UpnpDisplayFileAndLine(filed, DbgFileName + UPNPLIB_PROJECT_PATH_LENGTH,
266 DbgLineNo, DLevel, Module);
267 vfprintf(filed, FmtStr, ArgList);
268 if (filed != nullptr && filed != stderr)
269 fflush(filed); // Don't mock this because we use it for debuging.
270 }
271 va_end(ArgList);
272 umock::pthread_h.pthread_mutex_unlock(&GlobalDebugMutex);
273}
