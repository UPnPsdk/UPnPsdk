62// upnplib wrapper for the ::getsockname() system function
63// -------------------------------------------------------
64static int getsockname(SOCKET a_sockfd, sockaddr* a_addr,
65 socklen_t* a_addrlen) {
66 TRACE("Executing getsockname()");
67
68 int ret = umock::sys_socket_h.getsockname(a_sockfd, a_addr, a_addrlen);
69 if (ret == 0) {
70 return 0;
71 }
72#ifndef _MSC_VER
73 return ret;
74
75#else
76 if (umock::winsock2_h.WSAGetLastError() != WSAEINVAL) // Error 10022 not set
77 return ret;
78
79 // WSAEINVAL indicates that the socket is unbound. We will return an
80 // empty sockaddr with address family set. This is what we get on Unix
81 // platforms. On Microsoft Windows we cannot use ::getsockname() like
82 // on unix platforms because it returns an error indicating an unbound
83 // socket and an untouched socket address storage. Here we have to use
84 // an alternative ::getsockopt() that provides additional info with a
85 // non-standard option SO_PROTOCOL_INFO.
86 ::WSAPROTOCOL_INFO protocol_info{};
87 int len{sizeof(protocol_info)}; // May be modified?
88 if (umock::sys_socket_h.getsockopt(a_sockfd, SOL_SOCKET, SO_PROTOCOL_INFO,
89 &protocol_info, &len) != 0)
90 return WSAENOBUFS; // Insufficient resources were available in the
91 // system to perform the operation.
92
93 memset(a_addr, 0, *a_addrlen);
94 // Microsoft itself defines sockaddr.sa_family as ushort, so the typecast
95 // from int should not do any harm.
96 a_addr->sa_family =
97 static_cast<unsigned short>(protocol_info.iAddressFamily);
98 return 0;
99#endif
100}

206bool CSocket_basic::is_bound() {
207 // We get the socket address from the file descriptor and check if its
208 // address is unspecified or if the address and port are all zero. This
209 // replaces the method with direct compare of the socket addresses with a
210 // 16 byte AF_INET6 compare which last version can be found at commit
211 // a5ec86a93608234016630123c776c09f8ff276fb.
212 TRACE2(this, " Executing CSocket::is_bound()")
213
214 // binding is protected.
215 std::scoped_lock lock(m_bound_mutex);
216
217 const std::string& netaddr = this->get_addrp_str();
218 return (netaddr.empty() || netaddr.compare("[::]:0") == 0 ||
219 netaddr.compare("0.0.0.0:0") == 0)
220 ? false
221 : true;
222}

245// Setter for new socket file descriptor
246void CSocket::set(sa_family_t a_family, int a_socktype) {
247 TRACE2(this, " Construct CSocket(af, socktype)")
248
249 if (a_family != AF_INET6 && a_family != AF_INET)
250 throw std::invalid_argument(
251 UPNPLIB_LOGEXCEPT +
252 "MSG1015: Failed to create socket: invalid address family " +
253 std::to_string(a_family));
254 if (a_socktype != SOCK_STREAM && a_socktype != SOCK_DGRAM)
255 throw std::invalid_argument(
256 UPNPLIB_LOGEXCEPT +
257 "MSG1016: Failed to create socket: invalid socket type " +
258 std::to_string(a_socktype));
259
260 // Get new socket file descriptor.
261 SOCKET sfd = ::socket(a_family, a_socktype, 0);
262 if (sfd == INVALID_SOCKET)
263 throw_error("MSG1017: Failed to create socket:");
264
265 int so_option{0};
266 constexpr socklen_t optlen{sizeof(so_option)};
267
268 // Reset SO_REUSEADDR on all platforms if it should be set by default. This
269 // is unclear on WIN32. See note below.
270 // Type cast (char*)&so_option is needed for Microsoft Windows.
271 if (::setsockopt(sfd, SOL_SOCKET, SO_REUSEADDR,
272 reinterpret_cast<char*>(&so_option), optlen) != 0) {
273 throw_error("MSG1018: Failed to set socket option SO_REUSEADDR:");
274 }
275
276#ifdef _MSC_VER
277 // Set socket option SO_EXCLUSIVEADDRUSE on Microsoft Windows.
278 // THIS IS AN IMPORTANT SECURITY ISSUE! Lock at
279 // REF: [Using SO_REUSEADDR and SO_EXCLUSIVEADDRUSE]
280 // (https://learn.microsoft.com/en-us/windows/win32/winsock/using-so-reuseaddr-and-so-exclusiveaddruse#application-strategies)
281 so_option = 1; // Set SO_EXCLUSIVEADDRUSE
282 // Type cast (char*)&so_option is needed for Microsoft Windows.
283 if (std::setsockopt(sfd, SOL_SOCKET, SO_EXCLUSIVEADDRUSE,
284 reinterpret_cast<char*>(&so_option), optlen) != 0) {
285 throw_error(
286 "MSG1019: Failed to set socket option SO_EXCLUSIVEADDRUSE:");
287 }
288#endif
289
290 // Try to set socket option IPV6_V6ONLY to false if possible, means
291 // allowing IPv4 and IPv6. Although it makes no sense to enable
292 // IPV6_V6ONLY on an interface that can only use IPv4 we must reset it on
293 // Microsoft Windows because it is set there.
294#ifndef _MSC_VER
295 if (a_family == AF_INET6) {
296#endif
297 so_option = 0;
298 // Type cast (char*)&so_option is needed for Microsoft Windows.
299 if (::setsockopt(sfd, IPPROTO_IPV6, IPV6_V6ONLY,
300 reinterpret_cast<char*>(&so_option), optlen) != 0) {
301 throw_error("MSG1020: Failed to set socket option IPV6_V6ONLY:");
302 }
303#ifndef _MSC_VER
304 }
305#endif
306
307 // Store socket file descriptor
308 m_sfd = sfd;
309}

358// Bind socket to local address
359// REF:_[Bind:_Address_Already_in_Use]_(https://hea-www.harvard.edu/~fine/Tech/addrinuse.html)
360void CSocket::bind(const std::string& a_node, const std::string& a_port,
361 const int a_flags) {
362 TRACE2(this, " Executing CSocket::bind()")
363
364 // Protect binding.
365 std::scoped_lock lock(m_bound_mutex);
366
367 const sa_family_t addr_family = this->get_family();
368
369 // With address family AF_INET6 and not passive mode we always set
370 // IPV6_V6ONLY to true. This is the behavior on Unix platforms when binding
371 // the address and cannot be modified afterwards. MacOS does not modify the
372 // flag with binding. To be portable with same behavior on all platforms we
373 // set the flag before binding.
374 if (addr_family == AF_INET6 && (a_flags & AI_PASSIVE) == 0) {
375 // Don't use 'this->set_v6only(true)' because binding is protected with
376 // a mutex and we will get a deadlock due to using
377 // 'this->is_bound()' in 'this->set_v6only(true)'.
378 constexpr int so_option{1}; // true
379 // Type cast (char*)&so_option is needed for Microsoft Windows.
380 if (::setsockopt(m_sfd, IPPROTO_IPV6, IPV6_V6ONLY,
381 reinterpret_cast<const char*>(&so_option),
382 sizeof(so_option)) != 0)
383 throw_error("MSG1007: Failed to set socket option IPV6_V6ONLY:");
384 }
385
386 // Here we bind the socket to an address
387 const CAddrinfo ai(a_node, a_port, addr_family, this->get_socktype(),
388 AI_NUMERICHOST | AI_NUMERICSERV | a_flags);
389 // Type cast socklen_t is needed for Microsoft Windows.
390 if (::bind(m_sfd, ai->ai_addr, static_cast<socklen_t>(ai->ai_addrlen)) != 0)
391 throw_error("MSG1008: Failed to bind socket to an address:");
392}
