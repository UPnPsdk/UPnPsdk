56bool is_dns_name(const std::string& label
57) {
58 // Regular expression to validate DNS label.
59 // Negative Lookbehind: (?<!...) is not supported by C++ STL. I workaround
60 // it. std::regex
61 // pattern("^((?!-)[A-Za-z0-9-]{1,63}(?<!-)\\.)+[A-Za-z]{2,6}$");
62 TRACE("Executing is_dns_name()")
63 std::regex pattern("^((?!-)[A-Za-z0-9-]{1,63}\\.)+[A-Za-z]{2,6}$");
64 return (std::regex_match(label, pattern) && !label.contains("-.") &&
65 !label.contains("--") && !label.ends_with('-')) ||
66 label == "localhost";
67}

72void remove_dot_segments(std::string& a_path) {
73 // The letters (A., B., ...) are the steps as given by the algorithm in
74 // RFC3986_5.2.4.
75 TRACE("Executing UPnPsdk::remove_dot_segments(\"" + a_path + "\")")
76
77 std::string_view path_sv{a_path};
78 std::string output;
79
80 while (!path_sv.empty()) {
81 // A.
82 if (path_sv.substr(0, 3) == "../") {
83 path_sv.remove_prefix(3);
84 } else if (path_sv.substr(0, 2) == "./") {
85 path_sv.remove_prefix(2);
86 }
87 // B.
88 else if (path_sv.substr(0, 3) == "/./") {
89 path_sv.remove_prefix(2);
90 } else if (path_sv == "/.") {
91 path_sv.remove_suffix(1);
92 }
93 // C.
94 else if (path_sv.substr(0, 4) == "/../") {
95 path_sv.remove_prefix(3);
96 if (!output.empty()) {
97 size_t pos;
98 output.erase((pos = output.find_last_of('/')) ==
99 std::string::npos
100 ? 0
101 : pos); // Remove last segment
102 }
103 } else if (path_sv == "/..") {
104 path_sv.remove_suffix(2);
105 if (!output.empty()) {
106 size_t pos;
107 output.erase((pos = output.find_last_of('/')) ==
108 std::string::npos
109 ? 0
110 : pos); // Remove last segment
111 }
112 }
113 // D.
114 else if (path_sv == "." || path_sv == "..") {
115 path_sv = "";
116 }
117 // E.
118 else {
119 size_t start = path_sv.front() == '/' ? 1 : 0;
120 size_t end = path_sv.find_first_of('/', start);
121 if (end == std::string_view::npos)
122 end = path_sv.size();
123 output += path_sv.substr(0, end);
124 path_sv.remove_prefix(end);
125 }
126 }
127
128 a_path = output;
129}

161std::string_view
162get_scheme(std::string_view a_uri_sv
163) {
164 // A scheme, if any, must begin with a letter, must have alphanum
165 // characters, or '-', or '+', or '.', and ends with ':'.
166 TRACE("Executing get_scheme(a_uri_sv)")
167
168 size_t pos;
169 if ((pos = a_uri_sv.find_first_of(':')) == std::string_view::npos)
170 // No separator found means the scheme is undefined. The URI-reference
171 // may be a relative reference (RFC3986_4.1).
172 return ""; // Valid after return with size 0 (but dangling pointer).
173
174 if (pos == 0)
175 // ':' at the first position means the scheme is empty. This is invalid
176 // but will be checked later. I cannot simply return ':' as string_view
177 // because the string ":" is only valid within this function.
178 return std::string_view(a_uri_sv.data(), 1);
179
180 // Strip the view to have only the scheme with ':'. If pos > size(), the
181 // behavior is undefined. But that is guarded above.
182 a_uri_sv.remove_suffix(a_uri_sv.size() - pos - 1);
183
184 // Check if the scheme has valid character (RFC3986_3.1.).
185 if (!std::isalpha(static_cast<unsigned char>(a_uri_sv.front()))) {
186 // First character is not alpha. This is not a scheme.
187 return ""; // Valid after return with size 0 (but dangling pointer).
188 } else {
189 unsigned char ch;
190 for (auto it{a_uri_sv.begin()}; it < a_uri_sv.end() - 1; it++) {
191 ch = static_cast<unsigned char>(*it);
192 if (!(std::isalnum(ch) || (ch == '-') || (ch == '+') ||
193 (ch == '.'))) {
194 // Invalid character for scheme. This is not a scheme.
195 return ""; // Valid after return with size 0 (but dangling ptr).
196 }
197 }
198 }
199
200 return a_uri_sv;
201}

207CScheme::CScheme(std::string_view a_uri_sv) {
208 TRACE2(this, " Construct CScheme(a_uri_sv)")
209
210 std::string_view scheme_sv = get_scheme(a_uri_sv);

213 // Undefined scheme.
214 return;
215 if (scheme_sv == ":") {
216 // Empty scheme.
217 m_state = STATE::empty;
218 return;
219 }
220
221 // Normalize scheme to lower case character (RFC3986_6.2.2.1).
222 scheme_sv.remove_suffix(1); // Remove trailing ':'

225 *it = static_cast<char>(std::tolower(static_cast<unsigned char>(*it)));
226
227 m_state = STATE::avail;

233namespace {
250std::string_view
251get_authority(std::string_view a_uriref_sv
252) {
253 // The authority component is preceded by a double slash ("//") and is
254 // terminated by the next slash ('/'), question mark ('?'), or number sign
255 // ('#') character, or by the end of the URI (RFC3986 3.2.).
256 TRACE("Executing get_authority(a_uriref_sv)")
257
258 auto& npos = std::string_view::npos;
259 size_t pos;
260 if ((pos = a_uriref_sv.find("//")) != npos) {
261 // Extract the authority component with its separator "//".
262 a_uriref_sv.remove_prefix(pos);
263 // Find end of the authority and remove the rest to the end.
264 if ((pos = a_uriref_sv.find_first_of("/?#", 2)) != npos)
265 a_uriref_sv.remove_suffix(a_uriref_sv.size() - pos);
266
267 // Here we have the authority string-view with separator "//" and end
268 // separator (one of '/', '?', '#', ''). Having authority only "//"
269 // means authority is empty.
270 return a_uriref_sv;
271 }
272 // No "//" separator found. An empty authority means it is undefined.
273 return ""; // Valid after return with size 0 (but dangling pointer).
274}

1142 TRACE2(this, " Construct CPrepUriStr(a_uriref_str)")
1143
1144 auto it{a_uriref_str.begin()};
1145 auto it_end{a_uriref_str.end()};
1146 // clang-format off
1147 // First check begin and end of the URI for invalid encoded character.
1148 if (!a_uriref_str.empty() &&
1149 ( /*begin*/ (a_uriref_str.size() < 3 && *it == '%') ||
1150 /*end*/ (a_uriref_str.size() >= 3 && (*(it_end - 2) == '%' || *(it_end - 1) == '%'))))

1165 *it = static_cast<char>(
1166 std::toupper(static_cast<unsigned char>(*it)));
1167 }
1168 }
1169 return;
1170
1171exception:
