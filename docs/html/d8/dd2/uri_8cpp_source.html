150size_t parse_uric(
152 const char* in,
154 size_t max,
156 token* out) {
157 size_t i{};
158
159 while (i < max &&
160 (is_unreserved((unsigned char)in[i]) ||
161 is_reserved((unsigned char)in[i]) ||
162 ((i + (size_t)2 < max) &&
163 is_escaped(reinterpret_cast<const unsigned char*>(&in[i]))))) {
164 i++;
165 }
166
167 out->size = i;
168 out->buff = in;
169 return i;
170}

205int parse_hostport(
207 const char* in,
209 unsigned short int defaultPort,
212 hostport_type* out) {
213 char workbuf[256];
214 char* c;
215 struct sockaddr_in* sai4 = (struct sockaddr_in*)&out->IPaddress;
216 struct sockaddr_in6* sai6 = (struct sockaddr_in6*)&out->IPaddress;
217 char* srvname = NULL;
218 char* srvport = NULL;
219 char* last_dot = NULL;
220 unsigned short int port;
221 int af = AF_UNSPEC;
222 size_t hostport_size;
223 int has_port = 0;
224 int ret;
225
226 memset(out, 0, sizeof(hostport_type));
227 memset(workbuf, 0, sizeof(workbuf));
228 /* Work on a copy of the input string. */
229 strncpy(workbuf, in, sizeof(workbuf) - 1);
230 c = workbuf;
231 if (*c == '[') {
232 /* IPv6 addresses are enclosed in square brackets. */
233 srvname = ++c;
234 while (*c != '\0' && *c != ']')
235 c++;
236 if (*c == '\0')
237 /* did not find closing bracket. */
238 return UPNP_E_INVALID_URL;
239 /* NULL terminate the srvname and then increment c. */
240 *c++ = '\0'; /* overwrite the ']' */
241 if (*c == ':') {
242 has_port = 1;
243 c++;
244 }
245 af = AF_INET6;
246 } else {
247 /* IPv4 address -OR- host name. */
248 srvname = c;
249 while (*c != ':' && *c != '/' &&
250 (isalnum(*c) || *c == '.' || *c == '-')) {
251 if (*c == '.')
252 last_dot = c;
253 c++;
254 }
255 has_port = (*c == ':') ? 1 : 0;
256 /* NULL terminate the srvname */
257 *c = '\0';
258 if (has_port == 1)
259 c++;
260 if (last_dot != NULL && isdigit(*(last_dot + 1)))
261 /* Must be an IPv4 address. */
262 af = AF_INET;
263 else {
264 /* Must be a host name. */
265 struct addrinfo hints, *res, *res0;
266
267 memset(&hints, 0, sizeof(hints));
268 hints.ai_family = AF_UNSPEC;
269 hints.ai_socktype = SOCK_STREAM;
270
271 ret = umock::netdb_h.getaddrinfo(srvname, NULL, &hints, &res0);
272 if (ret == 0) {
273 for (res = res0; res; res = res->ai_next) {
274 switch (res->ai_family) {
275 case AF_INET:
276 case AF_INET6:
277 /* Found a valid IPv4 or IPv6
278 * address. */
279 memcpy(&out->IPaddress, res->ai_addr, res->ai_addrlen);
280 goto found;
281 }
282 }
283 found:
284 umock::netdb_h.freeaddrinfo(res0);
285 if (res == NULL)
286 /* Didn't find an AF_INET or AF_INET6
287 * address. */
288 return UPNP_E_INVALID_URL;
289 } else
290 /* getaddrinfo failed. */
291 return UPNP_E_INVALID_URL;
292 }
293 }
294 /* Check if a port is specified. */
295 if (has_port == 1) {
296 /* Port is specified. */
297 srvport = c;
298 while (*c != '\0' && isdigit(*c))
299 c++;
300 port = (unsigned short int)atoi(srvport);
301 if (port == 0)
302 /* Bad port number. */
303 return UPNP_E_INVALID_URL;
304 } else
305 /* Port was not specified, use default port. */
306 port = defaultPort;
307 /* The length of the host and port string can be calculated by */
308 /* subtracting pointers. */
309 hostport_size = (size_t)c - (size_t)workbuf;
310 /* Fill in the 'out' information. */
311 switch (af) {
312 case AF_INET:
313 sai4->sin_family = (sa_family_t)af;
314 sai4->sin_port = htons(port);
315 ret = inet_pton(AF_INET, srvname, &sai4->sin_addr);
316 break;
317 case AF_INET6:
318 sai6->sin6_family = (sa_family_t)af;
319 sai6->sin6_port = htons(port);
320 sai6->sin6_scope_id = gIF_INDEX;
321 ret = inet_pton(AF_INET6, srvname, &sai6->sin6_addr);
322 break;
323 default:
324 /* IP address was set by the hostname (getaddrinfo). */
325 /* Override port: */
326 if (out->IPaddress.ss_family == (sa_family_t)AF_INET)
327 sai4->sin_port = htons(port);
328 else
329 sai6->sin6_port = htons(port);
330 ret = 1;
331 }
332 /* Check if address was converted successfully. */
333 if (ret <= 0)
334 return UPNP_E_INVALID_URL;
335 out->text.size = hostport_size;
336 out->text.buff = in;
337
338 return (int)hostport_size;
339}

359size_t parse_scheme(
361 const char* in,
363 size_t max,
365 token* out) {
366 size_t i = (size_t)0;
367
368 out->size = (size_t)0;
369 out->buff = NULL;
370
371 if ((max == (size_t)0) || (!isalpha(in[0])))
372 return (size_t)0;
373
374 i++;
375 while ((i < max) && (in[i] != ':')) {
376 if (!(isalnum(in[i]) || (in[i] == '+') || (in[i] == '-') ||
377 (in[i] == '.')))
378 return (size_t)0;
379 i++;
380 }
381 if (i < max) {
382 out->size = i;
383 out->buff = &in[0];
384 return i;
385 }
386
387 return (size_t)0;
388}

419int replace_escaped(
421 char* in,
423 size_t index,
426 size_t* max) {
427 int tempInt = 0;
428 char tempChar = 0;
429 size_t i = (size_t)0;
430 size_t j = (size_t)0;
431
432 if (in[index] == '%' && isxdigit(in[index + (size_t)1]) &&
433 isxdigit(in[index + (size_t)2])) {
434 /* Note the "%2x", makes sure that we convert a maximum of two
435 * characters. */
436#ifdef _WIN32
437 if (sscanf_s(&in[index + (size_t)1],
438#else
439 if (sscanf(&in[index + (size_t)1],
440#endif
441 "%2x", (unsigned int*)&tempInt) != 1) {
442 return 0;
443 }
444 tempChar = (char)tempInt;
445 for (i = index + (size_t)3, j = index; j < *max; i++, j++) {
446 in[j] = tempChar;
447 if (i < *max) {
448 tempChar = in[i];
449 } else {
450 tempChar = 0;
451 }
452 }
453 *max -= (size_t)2;
454 return 1;
455 } else {
456 return 0;
457 }
458}

464int create_url_list(memptr* a_url_list, URL_list* a_out) {
465 std::cerr << "DEBUG! a_url_list=\""
466 << std::string_view(a_url_list->buf, a_url_list->length)
467 << "\", size=" << a_url_list->length << "\n";
468 if (!a_out)
469 return 0;
470 if (!a_url_list || a_url_list->length == 0) {
471 memset(a_out, 0, sizeof(*a_out));
472 return 0;
473 }
474
475 // Verify correct delimiter '<', '><', and '>'.
476 std::string_view urls_str(a_url_list->buf, a_url_list->length);
477 // Without any delimiter it is assumed to be just a single url...
478 if (urls_str.find_first_of("<>") != std::string_view::npos) {
479 // ...otherwise delimiter must be checked.
480 if (urls_str.front() == '<')
481 urls_str.remove_prefix(1);
482 else
483 return UPNP_E_INVALID_URL;
484 if (urls_str.back() == '\0')
485 urls_str.remove_suffix(1);
486 if (urls_str.back() == '>')
487 urls_str.remove_suffix(1);
488 else
489 return UPNP_E_INVALID_URL;
490 for (size_t i{0}; i < urls_str.size(); i++) {
491 if (urls_str[i] == '>') {
492 if (urls_str[i + 1] == '<')
493 i++;
494 else
495 return UPNP_E_INVALID_URL;
496 } else if (urls_str[i] == '<') {
497 return UPNP_E_INVALID_URL;
498 }
499 } // for
500 }
501
502 // Allocate memory and copy the serialized urls to it. It is important to do
503 // this beforehand because parsing the base_urls for components create
504 // pointer to it.
505 char* base_urls = static_cast<char*>(malloc(a_url_list->length + 1));
506 if (!base_urls)
507 return UPNP_E_OUTOF_MEMORY;
508 memcpy(base_urls, a_url_list->buf, a_url_list->length);
509 size_t base_urls_length = a_url_list->length;
510 // terminate list ("<url><url>\0") C string.
511 base_urls[base_urls_length] = '\0';
512
513 // Create views into the base_urls to its components (scheme, hostport,
514 // path, etc.) and cache them in a vector for later use.
515 std::vector<uri_type> url_views;
516
517 for (size_t i{0}; i < base_urls_length; i++) {
518 // Find start of a url...
519 if ((base_urls[i] == '<') && (i + 1 < base_urls_length)) {
520 // ...got it. Parse url with error handling to get its components.
521 uri_type splitted_url;
522
523 // parse_uri
524 int return_code = parse_uri(
525 &base_urls[i + 1], base_urls_length - i - 1, &splitted_url);
526
527 std::cerr << "DEBUG! parse_uri() return_code=" << return_code
528 << "\n";
529 if (return_code == HTTP_SUCCESS &&
530 splitted_url.hostport.text.size != 0) {
531 std::cerr << "DEBUG! Tracepoint1\n";
532 // No errors detected, cache the parsed result. Only URLs with
533 // network addresses are considered.
534 url_views.push_back(splitted_url);
535
536 } else if (return_code == UPNP_E_OUTOF_MEMORY) {
537 std::cerr << "DEBUG! Tracepoint2\n";
538 free(base_urls);
539 return UPNP_E_OUTOF_MEMORY;
540 }
541 std::cerr << "DEBUG! Tracepoint3\n";
542 }
543 }
544
545 // The vector tells us how many urls it has.
546 const size_t urls_count{url_views.size()};
547
548 if (urls_count > 0) {
549 // Allocate memory for the parsed urls to concatenate.
550 uri_type* parsed_urls =
551 static_cast<uri_type*>(malloc(sizeof(uri_type) * urls_count));
552 if (!parsed_urls) {
553 free(base_urls);
554 return UPNP_E_OUTOF_MEMORY;
555 }
556
557 // Concatenate parsed urls into a trivial C array.
558 for (size_t i{0}; i < urls_count; i++)
559 parsed_urls[i] = url_views[i];
560
561 // All values are available now. Fill the destination structure.
562 memset(a_out, 0, sizeof(*a_out));
563 a_out->parsedURLs = parsed_urls;
564 a_out->URLs = base_urls;
565 a_out->size = urls_count;
566
567 } else {
568 free(base_urls);
569 base_urls = nullptr;
570 memset(a_out, 0, sizeof(*a_out));
571 }
572
573 if (urls_count > INT_MAX) // Paranoia checking due to cast.
574 return UPNP_E_OUTOF_MEMORY;
575 return static_cast<int>(urls_count);
576}

579int copy_URL_list(URL_list* in, URL_list* out) {
580 out->URLs = nullptr;
581 out->parsedURLs = nullptr;
582 out->size = 0u;
583
584 size_t len = strlen(in->URLs) + 1;
585 out->URLs = static_cast<char*>(malloc(len));
586 if (!out->URLs)
587 return UPNP_E_OUTOF_MEMORY;
588 // out->parsedURLs = static_cast<uri_type*>(calloc(in->size,
589 // sizeof(uri_type)));
590 out->parsedURLs =
591 static_cast<uri_type*>(malloc(sizeof(uri_type) * in->size));
592 if (!out->parsedURLs) {
593 free(out->URLs);
594 return UPNP_E_OUTOF_MEMORY;
595 }
596
597 memcpy(out->URLs, in->URLs, len);
598 for (size_t i{0}; i < in->size; i++) {
599 // Having pointer within the coppied structure a deep copy is needed.
600 out->parsedURLs[i].type = in->parsedURLs[i].type;
601 copy_token(&in->parsedURLs[i].scheme, in->URLs,
602 &out->parsedURLs[i].scheme, out->URLs);
603 out->parsedURLs[i].path_type = in->parsedURLs[i].path_type;
604 copy_token(&in->parsedURLs[i].pathquery, in->URLs,
605 &out->parsedURLs[i].pathquery, out->URLs);
606 copy_token(&in->parsedURLs[i].fragment, in->URLs,
607 &out->parsedURLs[i].fragment, out->URLs);
608 copy_token(&in->parsedURLs[i].hostport.text, in->URLs,
609 &out->parsedURLs[i].hostport.text, out->URLs);
610 memcpy(&out->parsedURLs[i].hostport.IPaddress,
611 &in->parsedURLs[i].hostport.IPaddress,
612 sizeof(struct sockaddr_storage));
613 }
614 out->size = in->size;
615
616 return HTTP_SUCCESS;
617}

650int token_string_casecmp(token* in1, const char* in2) {
651 size_t in2_length = strlen(in2);
652 if (in1->buff == nullptr && in2_length == 0)
653 return 0;
654 else if (in1->buff == nullptr && in2_length != 0)
655 return -1;
656 else if (in1->size < in2_length)
657 return -1;
658 else if (in1->size > in2_length)
659 return 1;
660
661 return strncasecmp(in1->buff, in2, in1->size);
662}

687int remove_dots(char* buf, size_t size) {
688 char* in = buf;
689 char* out = buf;
690 char* max = buf + size;
691
692 while (!is_end_path(in[0])) {
693 assert(buf <= out);
694 assert(out <= in);
695 assert(in < max);
696
697 /* case 2.A: */
698 if (strncmp(in, "./", 2) == 0) {
699 in += 2;
700 } else if (strncmp(in, "../", 3) == 0) {
701 in += 3;
702 /* case 2.B: */
703 } else if (strncmp(in, "/./", 3) == 0) {
704 in += 2;
705 } else if (strncmp(in, "/.", 2) == 0 && is_end_path(in[2])) {
706 in += 1;
707 in[0] = '/';
708 /* case 2.C: */
709 } else if (strncmp(in, "/../", 4) == 0 ||
710 (strncmp(in, "/..", 3) == 0 && is_end_path(in[3]))) {
711 /* Make the next character in the input buffer a '/': */
712 if (is_end_path(in[3])) { /* terminating "/.." case */
713 in += 2;
714 in[0] = '/';
715 } else { /* "/../" prefix case */
716 in += 3;
717 }
718 /* Trim the last component from the output buffer, or
719 * empty it. */
720 while (buf < out)
721 if (*--out == '/')
722 break;
723#ifdef DEBUG
724 if (out < in)
725 out[0] = '\0';
726#endif
727 /* case 2.D: */
728 } else if (strncmp(in, ".", 1) == 0 && is_end_path(in[1])) {
729 in += 1;
730 } else if (strncmp(in, "..", 2) == 0 && is_end_path(in[2])) {
731 in += 2;
732 /* case 2.E */
733 } else {
734 /* move initial '/' character (if any) */
735 if (in[0] == '/')
736 *out++ = *in++;
737 /* move first segment up to, but not including, the next
738 * '/' character */
739 while (in < max && in[0] != '/' && !is_end_path(in[0]))
740 *out++ = *in++;
741#ifdef DEBUG
742 if (out < in)
743 out[0] = '\0';
744#endif
745 }
746 }
747 while (in < max)
748 *out++ = *in++;
749 if (out < max)
750 out[0] = '\0';
751 return UPNP_E_SUCCESS;
752}

754char* resolve_rel_url(char* base_url, char* rel_url) {
755 uri_type base;
756 uri_type rel;
757 int rv;
758 size_t len_rel;
759 size_t len_base;
760 size_t len;
761 char* out;
762 char* out_finger;
763 char* path;
764 size_t i;
765 size_t prefix;
766
767 if (!base_url && !rel_url)
768 return nullptr;
769 if (!base_url && rel_url)
770 return strdup(rel_url);
771 if (base_url && !rel_url)
772 return strdup(base_url);
773
774 len_rel = strlen(rel_url);
775 if (parse_uri(rel_url, len_rel, &rel) != HTTP_SUCCESS)
776 return NULL;
777 if (rel.type == Absolute)
778 return strdup(rel_url);
779
780 len_base = strlen(base_url);
781 if ((parse_uri(base_url, len_base, &base) != HTTP_SUCCESS) ||
782 (base.type != Absolute))
783 return NULL;
784 if (len_rel == (size_t)0)
785 return strdup(base_url);
786
787 len = len_base + len_rel + (size_t)2;
788 out = (char*)malloc(len);
789 if (out == NULL)
790 return NULL;
791 memset(out, 0, len);
792 out_finger = out;
793
794 /* scheme */
795 rv = snprintf(out_finger, len, "%.*s:", (int)base.scheme.size,
796 base.scheme.buff);
797 if (rv < 0 || rv >= (int)len)
798 goto error;
799 out_finger += rv;
800 len -= (size_t)rv;
801
802 /* authority */
803 if (rel.hostport.text.size > (size_t)0) {
804 rv = snprintf(out_finger, len, "%s", rel_url);
805 if (rv < 0 || rv >= (int)len)
806 goto error;
807 return out;
808 }
809 if (base.hostport.text.size > (size_t)0) {
810 rv = snprintf(out_finger, len, "//%.*s", (int)base.hostport.text.size,
811 base.hostport.text.buff);
812 if (rv < 0 || rv >= (int)len)
813 goto error;
814 out_finger += rv;
815 len -= (size_t)rv;
816 }
817
818 /* path */
819 path = out_finger;
820 if (rel.path_type == ABS_PATH) {
821 rv = snprintf(out_finger, len, "%s", rel_url);
822 } else if (base.pathquery.size == (size_t)0) {
823 rv = snprintf(out_finger, len, "/%s", rel_url);
824 } else {
825 if (rel.pathquery.size == (size_t)0) {
826 rv = snprintf(out_finger, len, "%.*s", (int)base.pathquery.size,
827 base.pathquery.buff);
828 } else {
829 if (len < base.pathquery.size)
830 goto error;
831 i = 0;
832 prefix = 1;
833 while (i < base.pathquery.size) {
834 out_finger[i] = base.pathquery.buff[i];
835 switch (base.pathquery.buff[i++]) {
836 case '/':
837 prefix = i;
838 /* fall-through */
839 default:
840 continue;

877error:
878 free(out);

907 } else
908 return begin_path; // error code from parse_hostport()
909 } else {

911 begin_path = static_cast<int>(begin_hostport);
