34#include "tv_device.hpp"
35#include <signal.h>
36
37int main(int argc, char* argv[]) {
38 pthread_t cmdloop_thread;
39#ifndef _WIN32
40 int sig;
41 sigset_t sigs_to_catch;
42#endif
43
44 int rc = device_main(argc, argv);
45 if (rc != UPNP_E_SUCCESS) {
46 return rc;
47 }
48
49 /* start a command loop thread */
50 if (pthread_create(&cmdloop_thread, NULL, TvDeviceCommandLoop, NULL) != 0) {
51 return UPNP_E_INTERNAL_ERROR;
52 }
53#ifdef _MSC_VER
54 pthread_join(cmdloop_thread, NULL);
55#else
56 /* Catch Ctrl-C and properly shutdown */
57 sigemptyset(&sigs_to_catch);
58 /* Comment the line below if you are running gdb, otherwise you will be
59 * unable to interrupt with CTRL-C. */
60 sigaddset(&sigs_to_catch, SIGINT);
61 sigwait(&sigs_to_catch, &sig);
62 SampleUtil_Print("Shutting down on signal %d...\n", sig);
63#endif
64
65 return TvDeviceStop();
66}
