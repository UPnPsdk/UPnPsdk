43int main(int argc, char* argv[]) {
44 TRACE("Executing main()");
45 pthread_t cmdloop_thread;
46#ifndef _WIN32
47 int sig;
48 sigset_t sigs_to_catch;
49#endif
50
51 // Start TV device
52 int rc = device_main(argc, argv);
53 if (rc != UPNP_E_SUCCESS) {
54 SampleUtil_Print("Error starting UPnP TV Device\n");
55 return rc;
56 }
57
58 /* start a command loop thread */
59 if (pthread_create(&cmdloop_thread, NULL, TvDeviceCommandLoop, NULL) != 0) {
60 return UPNP_E_INTERNAL_ERROR;
61 }
62#ifdef _MSC_VER
63 pthread_join(cmdloop_thread, NULL);
64#else
65 /* Catch Ctrl-C and properly shutdown */
66 sigemptyset(&sigs_to_catch);
67 /* Comment the line below if you are running gdb, otherwise you will be
68 * unable to interrupt with CTRL-C. */
69 sigaddset(&sigs_to_catch, SIGINT);
70 sigwait(&sigs_to_catch, &sig);
71 SampleUtil_Print("Shutting down on signal %d...\n", sig);
72#endif
73
74 return TvDeviceStop();
75}
