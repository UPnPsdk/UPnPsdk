15uint8_t netmask_to_bitmask(const ::sockaddr_storage* a_netmask) {
16 TRACE("Executing netmask_to_bitmask()")
17 if (a_netmask == nullptr)
18 throw std::runtime_error(
19 UPnPsdk_LOGEXCEPT +
20 "MSG1059: No network socket address information given.\n");
21
22 uint8_t bitmask{};
23 uint8_t nullbits{};
24 switch (a_netmask->ss_family) {
25 case AF_INET6: {
26 // Count set bits.
27 for (size_t i{}; i < sizeof(in6_addr); i++) {
28 uint8_t s6addr = reinterpret_cast<const ::sockaddr_in6*>(a_netmask)
29 ->sin6_addr.s6_addr[i];
30 if (s6addr == 255) {
31 bitmask += 8;
32 } else {
33 while (s6addr) {
34 bitmask++;
35 s6addr <<= 1;
36 }
37 break; // for() loop
38 }
39 }
40 // Check if all remaining bits are zero.
41 for (int i{sizeof(in6_addr) - 1}; i >= 0; i--) {
42 uint8_t s6addr = reinterpret_cast<const ::sockaddr_in6*>(a_netmask)
43 ->sin6_addr.s6_addr[i];
44 if (s6addr == 0) {
45 nullbits += 8;
46 } else {
47 while ((s6addr & 1) == 0) {
48 nullbits++;
49 s6addr >>= 1;
50 }
51 break; // for() loop
52 }
53 }
54 // Check valid netmask.
55 if (bitmask + nullbits != 128) {
56 char ip_str[INET6_ADDRSTRLEN]{};
57 ::inet_ntop(
58 AF_INET6,
59 &reinterpret_cast<const ::sockaddr_in6*>(a_netmask)->sin6_addr,
60 ip_str, sizeof(ip_str));
61 throw std::runtime_error(
62 UPnPsdk_LOGEXCEPT +
63 "MSG1067: Invalid ip-address prefix bitmask \"[" +
64 std::string(ip_str) + "]\".\n");
65 }
66 } break; // switch()
67
68 case AF_INET: {
69 in_addr_t saddr = ntohl(
70 reinterpret_cast<const ::sockaddr_in*>(a_netmask)->sin_addr.s_addr);
71 in_addr_t saddr0 = saddr;
72 // Count set bits.
73 while (saddr) {
74 bitmask++;
75 saddr <<= 1;
76 }
77 // Check if all remaining bits are zero.
78 for (int i{}; (saddr0 & 1) == 0 && i < 32; i++) {
79 nullbits++;
80 saddr0 >>= 1;
81 }
82 // Check valid netmask.
83 if (bitmask + nullbits != 32) {
84 char ip_str[INET_ADDRSTRLEN]{};
85 ::inet_ntop(
86 AF_INET,
87 &reinterpret_cast<const ::sockaddr_in*>(a_netmask)->sin_addr,
88 ip_str, sizeof(ip_str));
89 throw std::runtime_error(UPnPsdk_LOGEXCEPT +
90 "MSG1069: Invalid ip-address netmask \"" +
91 std::string(ip_str) + "\".\n");
92 }
93 } break;
94
95 case AF_UNSPEC:
96 return 0;
97
98 default:
99 throw std::runtime_error(
100 UPnPsdk_LOGEXCEPT + "MSG1028: Unsupported address family(" +
101 std::to_string(a_netmask->ss_family) + "), only AF_INET6(" +
102 std::to_string(AF_INET6) + "), AF_INET(" + std::to_string(AF_INET) +
103 "), or AF_UNSPEC(" + std::to_string(AF_UNSPEC) + ") are valid.\n");
104 } // switch()
105
106 return bitmask;
107}

110void bitmask_to_netmask(const ::sockaddr_storage* a_saddr,
111 const uint8_t a_prefixlength, SSockaddr& a_saddrObj) {
112 TRACE("Executing bitmask_to_netmask()")
113 if (a_saddr == nullptr)
114 throw std::runtime_error(
115 UPnPsdk_LOGEXCEPT +
116 "MSG1070: No associated socket address for the netmask given.\n");
117
118 switch (a_saddr->ss_family) {
119 case AF_INET6: {
120 // I have to manage 16 bytes from the binary IPv6 address to set its
121 // bits. 15 bytes are all ones or all zero bits. Only one byte may have
122 // partly ones and zero bits. First I calculate how many leading bytes
123 // have full one bits. Following is the partial set byte. The rest are
124 // bytes with full zero bits set.
125 in6_addr netmask6{};
126
127 // All prefix lengths > 128 will throw this exception.
128 if (a_prefixlength > 128)
129 throw std::runtime_error(
130 UPnPsdk_LOGEXCEPT +
131 "MSG1124: Invalid IPv6 address prefix bitmask(" +
132 std::to_string(a_prefixlength) + ") exceeds 128.\n");
133
134 // Calculate number of leading bytes with full one bits.
135 int ones_bytes{a_prefixlength / 8};
136
137 // Fill leading bytes with all bit ones.
138 int i{};
139 for (; i < ones_bytes; i++)
140 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
141
142 // Handle the one partly bit-set byte.
143 if (i < 16 /*bytes*/) {
144 // Preset byte with all bit ones.
145 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
146 // Shift remaining zero bits from right into byte with using the
147 // reminder from the byte devision.
148 netmask6.s6_addr[i] <<= (8 - (a_prefixlength % 8));
149 i++;
150
151 // Fill remaining bytes with zero bits.
152 for (; i < 16 /*bytes*/; i++)
153 netmask6.s6_addr[i] = 0;
154 }
155
156 // Return the result.
157 a_saddrObj = "";
158 a_saddrObj.sin6.sin6_family = AF_INET6;
159 memcpy(&a_saddrObj.sin6.sin6_addr, &netmask6,
160 sizeof(a_saddrObj.sin6.sin6_addr));
161 } break;
162
163 case AF_INET: {
164 // All prefix lengths > 32 will throw this exception.
165 if (a_prefixlength > 32) {
166 throw std::runtime_error(
167 UPnPsdk_LOGEXCEPT +
168 "MSG1125: Invalid IPv4 address prefix bitmask(" +
169 std::to_string(a_prefixlength) + ") exceeds 32.\n");
170 }
171 // Prepare socket address and shift zero bits from right into the mask.
172 ::sockaddr_in saddr{};
173 saddr.sin_family = AF_INET;
174 if (a_prefixlength) {
175 in_addr_t bitshift{~0u}; // all one bits
176 bitshift <<= (32 - a_prefixlength); // shift zero bits from right
177 saddr.sin_addr.s_addr = htonl(bitshift);
178 }
179 // Return the result.
180 a_saddrObj.sin = saddr;
181 } break;
182
183 case AF_UNSPEC: {
184 a_saddrObj = "";
185 } break;
186
187 default: {
188 throw std::runtime_error(
189 UPnPsdk_LOGEXCEPT + "MSG1126: Unsupported address family(" +
190 std::to_string(a_saddr->ss_family) + "), only AF_INET6(" +
191 std::to_string(AF_INET6) + "), AF_INET(" + std::to_string(AF_INET) +
192 "), or AF_UNSPEC(" + std::to_string(AF_UNSPEC) + ") are valid.\n");
193 }
194 } // switch
195}

208bool CNetadapter::find_first(std::string_view a_name_or_addr) {
209 TRACE2(this, " Executing CNetadapter::find_first()")
210
211 // By default look for a valid ip address.
212 // ---------------------------------------
213 // The operating system presents one as best choise.
214 if (a_name_or_addr.empty()) {
215 SSockaddr sa_nadObj;
216 this->reset(); // noexcept
217 do {
218 this->sockaddr(sa_nadObj);
219 if (!sa_nadObj.is_loopback())
220 return true;
221 } while (this->get_next());
222
223 // No usable address found, nothing more to do.
224 return false;
225 }
226
227 // Look for a loopback interface.
228 // ------------------------------
229 if (a_name_or_addr == "loopback") {
230 SSockaddr sa_nadObj;
231 this->reset();
232 do {
233 this->sockaddr(sa_nadObj);
234 if (sa_nadObj.is_loopback()) {
235 m_find_index = this->index();
236 return true;
237 }
238 } while (this->get_next());
239
240 // No loopback interface found, nothing more to do.
241 return false;
242 }
243
244 // Look for a local network adapter name.
245 // --------------------------------------
246 this->reset();
247 do {
248 if (this->name() == a_name_or_addr) {
249 m_find_index = this->index();
250 return true;
251 }
252 } while (this->get_next());
253
254 // No name found, look for the ip address of a local network adapter.
255 // ------------------------------------------------------------------
256 // Try to translate input argument to a socket address.
257 CAddrinfo ainfoObj(a_name_or_addr, AI_NUMERICHOST, 0);
258 if (!ainfoObj.get_first())
259 return false;
260
261 // Valid ip address string given as input argument. Get its socket address.
262 SSockaddr sa_inputObj{};
263 ainfoObj.sockaddr(sa_inputObj);
264
265 // Parse network adapter list for the given input ip address.
266 SSockaddr sa_nadObj;
267 this->reset();
268 do {
269 this->sockaddr(sa_nadObj);
270 if (sa_nadObj == sa_inputObj)
271 return true;
272 } while (this->get_next());
273
274 return false;
275}
