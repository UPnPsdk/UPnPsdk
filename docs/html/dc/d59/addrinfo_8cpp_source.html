19// Free function to check for a netaddress without port
20// ----------------------------------------------------
38// I use the system function ::getaddrinfo() to check if the node string is
39// acceptable. Using ::getaddrinfo() is needed to cover all special address
40// features like scope id for link local addresses, Internationalized Domain
41// Names, and so on.
42sa_family_t is_netaddr(
44 const std::string& a_node,
47 const int a_addr_family = AF_UNSPEC) noexcept {
48 // clang-format off
49 TRACE("Executing is_netaddr(\"" + a_node + "\", " +
50 (a_addr_family == AF_INET6 ? "AF_INET6" :
51 (a_addr_family == AF_INET ? "AF_INET" :
52 (a_addr_family == AF_UNSPEC ? "AF_UNSPEC" :
53 std::to_string(a_addr_family)))) + ")")
54 // clang-format on
55
56 // The shortest numeric netaddress string is "[::]".
57 if (a_node.size() < 4) { // noexcept
58 return AF_UNSPEC;
59 }
60
61 // Provide resources for ::getaddrinfo()
62 // AI_NUMERICHOST ensures that only numeric addresses accepted.
63 std::string node;
64 ::addrinfo hints{};
65 hints.ai_flags = AI_NUMERICHOST;
66 hints.ai_family = a_addr_family;
67 ::addrinfo* res{nullptr};
68
69 // Check for ipv6 addresses and remove surounding brackets for
70 // ::getaddrinfo().
71 // front() and back() have undefined behavior with an empty string. Here
72 // its size() is at least 4. Substr() throws exception out of range if pos
73 // > size(). All this means that we cannot get an exception here.
74 if (a_node.front() == '[' && a_node.back() == ']' &&
75 (a_addr_family == AF_UNSPEC || a_addr_family == AF_INET6)) {
76 node = a_node.substr(1, a_node.length() - 2);
77 hints.ai_family = AF_INET6;
78
79 } else if (a_node.find_first_of(":") != std::string::npos) {
80 // Ipv6 addresses are already checked and here are only ipv4 addresses
81 // and URL names possible. Both are not valid if they contain a colon.
82 // find_first_of() does not throw an exception.
83 return AF_UNSPEC;
84
85 } else {
86 node = a_node;
87 }
88
89 // Call ::getaddrinfo() to check the remaining node string.
90 int rc = umock::netdb_h.getaddrinfo(node.c_str(), nullptr, &hints, &res);
91 TRACE2("syscall ::getaddrinfo() with res = ", res)
92 if (rc != 0) {
93 TRACE2("syscall ::freeaddrinfo() with res = ", res)
94 umock::netdb_h.freeaddrinfo(res);
95 UPnPsdk_LOGINFO "MSG1116: syscall ::getaddrinfo(\""
96 << node.c_str() << "\", nullptr, " << &hints << ", " << &res
97 << "), (" << rc << ") " << gai_strerror(rc) << '\n';
98 return AF_UNSPEC;
99 }
100
101 int af_family = res->ai_family;
102 TRACE2("syscall ::freeaddrinfo() with res = ", res)
103 umock::netdb_h.freeaddrinfo(res);
104 // Guard different types on different platforms (win32); need to cast to
105 // af_family (unsigned short).
106 if (af_family < 0 || af_family > 65535) {
107 return AF_UNSPEC;
108 }
109 return static_cast<sa_family_t>(af_family);
110}

131// Constructor for getting an address information from only a netaddress.
132// ----------------------------------------------------------------------
133CAddrinfo::CAddrinfo(std::string_view a_node, const int a_family,
134 const int a_socktype, const int a_flags,
135 const int a_protocol)
136 : m_node(a_node) {
137 // Just do a simple check for a possible port and split it from address
138 // string. Detailed tests will be done with this->load().
139 TRACE2(this, " Construct CAddrinfo() without service")
140
141 size_t pos;
142 // The smalest valid netaddress is "[::]".
143 if (m_node.size() < 4 || (m_node.front() == '[' && m_node.back() == ']')) {
144 // IPv6 without port, use given address string
145 goto set_flags;
146 }
147 pos = m_node.rfind("]:"); // noexcept
148 if (pos != m_node.npos) {
149 // IPv6 with port, split the address string. substr() throws exception
150 // std::out_of_range if pos > size(). We have at least 4 character. pos
151 // is 0-based, size() is 1-based so with "]:" is size()==2, pos==0,
152 // pos+2==size() does not throw.
153 m_service = m_node.substr(pos + 2);
154 m_node = m_node.substr(0, pos + 1);
155 goto set_flags;
156 }
157 pos = m_node.find_last_of(']'); // noexcept
158 if (pos != m_node.npos) {
159 // Maybe IPv6 with any remaining character, split address string
160 m_service = m_node.substr(pos + 1);
161 m_node = m_node.substr(0, pos + 1);
162 goto set_flags;
163 }
164 pos = m_node.find_last_of(':'); // noexcept
165 if (pos != m_node.npos) {
166 // IPv4 or URL name with port, split address string
167 m_service = m_node.substr(pos + 1);
168 m_node = m_node.substr(0, pos);
169 }
170
171set_flags:
172 this->set_ai_flags(a_family, a_socktype, a_flags, a_protocol);
173}

177inline void CAddrinfo::set_ai_flags(const int a_family, const int a_socktype,
178 const int a_flags,
179 const int a_protocol) noexcept {
180 // I cannot use the initialization list of the constructor because the
181 // member order in the structure addrinfo is different on Linux, MacOS and
182 // win32. I have to use the member names to initialize them, what's not
183 // possible for structures in the constructors initialization list.
184 m_hints.ai_flags = a_flags;
185 m_hints.ai_family = a_family;
186 m_hints.ai_socktype = a_socktype;
187 m_hints.ai_protocol = a_protocol;
188}

217// Setter to load an addrinfo from the operating system to the object
218// ------------------------------------------------------------------
219// Get address information with cached hints.
220void CAddrinfo::load() {
221 TRACE2(this, " Executing CAddrinfo::load()")
222
223 // Local working copies: modified node, service, and hints to use for
224 // syscall ::getaddrinfo().
225 std::string node;
226 std::string service;
227 ::addrinfo hints{m_hints}; // user given opgtions have priority
228
229 // Correct weak hints
230 // ------------------
231 // Always call UPnPsdk::is_netaddr() with AF_UNSPEC (default argument) to
232 // check all types. I always set AI_NUMERICHOST if it match, no matter what
233 // 'a_family' was requested by the caller. But if AF_UNSPEC was requested,
234 // the found address family will also be set.
235 const sa_family_t addr_family =
236 is_netaddr(m_node, m_hints.ai_family); // noexcept
237 if (addr_family != AF_UNSPEC) { // Here we have a numeric netaddress
238 hints.ai_flags |= AI_NUMERICHOST;
239 if (m_hints.ai_family == AF_UNSPEC) { // Correct ai_family, we know it
240 hints.ai_family = addr_family;
241 }
242 }
243 // Check node name
244 // ---------------
245 if (addr_family == AF_INET6) {
246 // Here we have only ipv6 node strings representing a numeric ip
247 // address (netaddress) without port that is at least "[::]". Remove
248 // surounding brackets for ::getaddrinfo()
249 node = m_node.substr(1, m_node.length() - 2);
250 } else if (addr_family == AF_INET) {
251 // Here we have a valid ipv4 neetaddress. It can be unmodified given to
252 // ::getaddrinfo(). No need to set AI_NUMERICHOST, it's already done
253 // with getting addr_family above.
254 node = m_node;
255 } else if (is_netaddr('[' + m_node + ']', AF_INET6) == AF_INET6) {
256 // ipv6 addresses without brackets would be accepted by ::getaddrinfo()
257 // but they are not valid netaddresses. So I make them invalid.
258 hints.ai_flags |= AI_NUMERICHOST;
259 node = m_node + "(no_brackets)";
260 } else if (m_node.find_first_of("[]:") != m_node.npos) { // noexcept
261 // An address string without port (m_node is without port) containing
262 // one of theese characters cannot be a valid alphanumeric URL. It can
263 // only be a numeric address (or be invalid).
264 hints.ai_flags |= AI_NUMERICHOST;
265 node = m_node;
266 } else {
267 // Here we have a non numeric node name (no netaddress). Is it a valid
268 // (maybe alphanumeric) node name? ::getaddrinfo() shall decide it.
269 node = m_node;
270 }
271 const char* c_node = node.empty() ? nullptr : node.c_str();
272 std::string node_out = node.empty() ? "nullptr" : "\"" + node + "\"";
273
274 // Check service/port
275 // ------------------
276 // I have to do this because ::getaddrinfo() does not detect wrong port
277 // numbers >65535. It silently overruns to port number 0, 1, 2...
278 try {
279 to_port(m_service);
280 // Valid numeric port number
281 service = m_service;
282 hints.ai_flags |= AI_NUMERICSERV;
283
284 } catch (const std::out_of_range&) {
285 // Valid numeric port number > 65535
286 service = m_service + "(invalid)";
287 hints.ai_flags |= AI_NUMERICSERV;
288
289 } catch (const std::invalid_argument&) {
290 // Any alphanumeric port name
291 service = m_service;
292 }
293 if (service.empty()) {
294 service = "0";
295 hints.ai_flags |= AI_NUMERICSERV;
296 }
297
298 // syscall ::getaddrinfo() with prepared arguments
299 // -----------------------------------------------
300 ::addrinfo* new_res{nullptr}; // Result from ::getaddrinfo()
301 const int ret =
302 umock::netdb_h.getaddrinfo(c_node, service.c_str(), &hints, &new_res);
303 TRACE2("syscall ::getaddrinfo() with new_res = ", new_res)
304
305 if (g_dbug) {
306 // Very helpful for debugging to see what is given to ::getaddrinfo()
307 char addrStr[INET6_ADDRSTRLEN]{};
308 char servStr[NI_MAXSERV]{};
309 if (ret == 0)
310 ::getnameinfo(new_res->ai_addr,
311 static_cast<socklen_t>(new_res->ai_addrlen), addrStr,
312 sizeof(addrStr), servStr, sizeof(servStr),
313 NI_NUMERICHOST | NI_NUMERICSERV);
314 // clang-format off
315 UPnPsdk_LOGINFO << "MSG1111: syscall ::getaddrinfo(" << node_out
316 << ", " << "\"" << service << "\", "
317 << &hints << ", " << &new_res
318 << ") node=\"" << m_node << "\", "
319 << (hints.ai_flags & AI_NUMERICHOST ? "AI_NUMERICHOST, " : "")
320 << (hints.ai_flags & AI_NUMERICSERV ? "AI_NUMERICSERV, " : "")
321 << (hints.ai_flags & AI_PASSIVE ? "AI_PASSIVE, " : "")
322 << (hints.ai_family == AF_INET6 ? "AF_INET6" :
323 (hints.ai_family == AF_INET ? "AF_INET" :
324 (hints.ai_family == AF_UNSPEC ? "AF_UNSPEC" :
325 "hints.ai_family=" + std::to_string(hints.ai_family))))
326 << (ret != 0
327 ? ". Get GAI_ERROR(" + std::to_string(ret) + ")"
328 : ". Get first \"" + std::string(addrStr) + "\", port "
329 + std::string(servStr)) << " (maybe more)\n";
330 }
331 if (ret == EAI_SERVICE /* Servname not supported for ai_socktype */
332 || ret == EAI_NONAME /* Node or service not known */
333 || ret == EAI_AGAIN /* The name server returned a temporary failure indication, try again later */
335#ifndef _MSC_VER
336 || ret == EAI_ADDRFAMILY /* Address family for NAME not supported */
337#endif
338 || ret == EAI_NODATA) /* No address associated with hostname */ {
339 // Error numbers definded in netdb.h.
340 // Maybe an alphanumeric node name that cannot be resolved (e.g. by
341 // DNS)? Anyway, the user has to decide what to do. Because this
342 // depends on extern available DNS server the error can occur
343 // unexpectedly at any time. We have no influence on it but I will give
344 // an extended error message.
345 throw std::runtime_error(UPnPsdk_LOGEXCEPT + "MSG1112: errid(" +
346 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\", " +
347 ((hints.ai_family == AF_UNSPEC) ? "IPv?_" :
348 ((hints.ai_family == AF_INET6) ? "IPv6_" : "IPv4_")) +
349 ((hints.ai_flags & AI_NUMERICHOST) ? "numeric_host=\"" : "alphanum_name=\"") +
350 node + "\", service=\"" +
351 service + "\"" +
352 ((hints.ai_flags & AI_PASSIVE) ? ", passive_listen" : "") +
353 ((hints.ai_flags & AI_NUMERICHOST) ? "" : ", (maybe DNS query temporary failed?)"));
354 }
355 // clang-format on
356
357 if (ret != 0) {
358 throw std::invalid_argument(
359 UPnPsdk_LOGEXCEPT +
360 "MSG1037: Failed to get address information: errid(" +
361 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\"\n");
362 }
363
364 for (::addrinfo* res{new_res}; res != nullptr; res = res->ai_next) {
365 // First ai_socktype is different on platforms: man getsockaddr says
366 // "Specifying 0 in hints.ai_socktype indicates that socket addresses
367 // of any type can be returned". Linux returns SOCK_STREAM first, MacOS
368 // returns SOCK_DGRAM first and win32 returns 0.
369 // if (hints.ai_socktype == 0)
370 // res->ai_socktype = 0;
371 //
372 // Different on platforms: Ubuntu & MacOS return protocol number, win32
373 // returns 0. I just return what was used to call ::getaddrinfo().
374 res->ai_protocol = hints.ai_protocol;
375 //
376 // Different on platforms: Ubuntu returns set flags, MacOS & win32
377 // return 0. I just return what was used to call ::getaddrinfo().
378 res->ai_flags = hints.ai_flags;
379 }
380 // Man getaddrinfo says: "If service is NULL, then the port number of the
381 // returned socket addresses will be left uninitialized." The service is
382 // never set to NULL so we always have a defined service/portnumber.
383 // if (service.empty())
384 // // port for AF_INET6 is also valid for AF_INET
385 // reinterpret_cast<sockaddr_in6*>(new_res->ai_addr)->sin6_port = 0;
386
387 // If load() is called the second time then m_res still points to the
388 // previous allocated memory. To avoid a memory leak it must be freed
389 // before pointing to the new allocated memory.
390 this->free_addrinfo();
391 // finaly point to the new address information from the operating
392 // system.
393 m_res = new_res;
394 m_res_current = new_res;
395}

397// Getter for the next available address information
398// -------------------------------------------------
399bool CAddrinfo::get_next() noexcept {
400 if (m_res_current->ai_next == nullptr) {
401 // It doesn't matter if already pointing to m_hints. m_hints->ai_next is
402 // also nullptr.
403 m_res_current = &m_hints;
404 return false;
405 }
406 m_res_current = m_res_current->ai_next;
407 return true;
408}

410// Get netaddress with port from current selcted address information
411// -----------------------------------------------------------------
412std::string CAddrinfo::netaddrp() noexcept {
413 if (m_res == &m_hints)
414 return "";
415
416 char addrStr[INET6_ADDRSTRLEN];
417 char servStr[NI_MAXSERV];
418 int ret = ::getnameinfo(m_res_current->ai_addr,
419 static_cast<socklen_t>(m_res_current->ai_addrlen),
420 addrStr, static_cast<socklen_t>(sizeof(addrStr)),
421 servStr, static_cast<socklen_t>(sizeof(servStr)),
422 NI_NUMERICHOST | NI_NUMERICSERV);
423 if (ret != 0) {
424 UPnPsdk_LOGERR "MSG1032: Failed to get name information: "
425 << gai_strerror(ret) << ". Continue with empty netaddress \"\".\n";
426 return "";
427 }
428
429 switch (m_res_current->ai_family) {
430 case AF_INET6:
431 return "[" + std::string(addrStr) + "]:" + std::string(servStr);
432 case AF_INET:
433 return std::string(addrStr) + ":" + std::string(servStr);
434 case AF_UNSPEC:
435 return ":" + std::string(servStr);
436 }
437
438 UPnPsdk_LOGERR "MSG1033: Unsupported address family "
439 << m_res_current->ai_family
440 << ". Continue with empty netaddress \"\".\n";
441 return "";
442}
