91int NewRequestHandler(
93 struct sockaddr* DestAddr,
95 int NumPacket,
97 char** RqPacket) {
98 char errorBuffer[ERROR_BUFFER_LEN];
99 SOCKET ReplySock;
100 socklen_t socklen = sizeof(struct sockaddr_storage);
101 int Index;
102 struct in_addr replyAddr;
103 /* a/c to UPNP Spec */
104 int ttl = 4;
105#ifdef UPNP_ENABLE_IPV6
106 int hops = 1;
107#endif
108 char buf_ntop[INET6_ADDRSTRLEN];
109 int ret = UPNP_E_SUCCESS;
110
111 if (strlen(gIF_IPV4) > (size_t)0 &&
112 !inet_pton(AF_INET, gIF_IPV4, &replyAddr)) {
113 return UPNP_E_INVALID_PARAM;
114 }
115
116 ReplySock =
117 umock::sys_socket_h.socket((int)DestAddr->sa_family, SOCK_DGRAM, 0);
118 if (ReplySock == INVALID_SOCKET) {
119 strerror_r(errno, errorBuffer, ERROR_BUFFER_LEN);
120 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
121 "SSDP_LIB: New Request Handler:"
122 "Error in socket(): %s\n",
123 errorBuffer);
124
125 return UPNP_E_OUTOF_SOCKET;
126 }
127
128 switch (DestAddr->sa_family) {
129 case AF_INET:
130 inet_ntop(AF_INET, &((struct sockaddr_in*)DestAddr)->sin_addr, buf_ntop,
131 sizeof(buf_ntop));
132 umock::sys_socket_h.setsockopt(ReplySock, IPPROTO_IP, IP_MULTICAST_IF,
133 (char*)&replyAddr, sizeof(replyAddr));
134 umock::sys_socket_h.setsockopt(ReplySock, IPPROTO_IP, IP_MULTICAST_TTL,
135 (char*)&ttl, sizeof(int));
136 socklen = sizeof(struct sockaddr_in);
137 break;
138#ifdef UPNP_ENABLE_IPV6
139 case AF_INET6:
140 inet_ntop(AF_INET6, &((struct sockaddr_in6*)DestAddr)->sin6_addr,
141 buf_ntop, sizeof(buf_ntop));
142 umock::sys_socket_h.setsockopt(ReplySock, IPPROTO_IPV6,
143 IPV6_MULTICAST_IF, (char*)&gIF_INDEX,
144 sizeof(gIF_INDEX));
145 umock::sys_socket_h.setsockopt(ReplySock, IPPROTO_IPV6,
146 IPV6_MULTICAST_HOPS, (char*)&hops,
147 sizeof(hops));
148 break;
149#endif
150 default:
151 UpnpPrintf(UPNP_CRITICAL, SSDP, __FILE__, __LINE__,
152 "Invalid destination address specified.");
153 ret = UPNP_E_NETWORK_ERROR;
154 goto end_NewRequestHandler;
155 }
156
157 for (Index = 0; Index < NumPacket; Index++) {
158 ssize_t rc;
159 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
160 ">>> SSDP SEND to %s >>>\n%s\n", buf_ntop,
161 *(RqPacket + Index));
162 rc = sendto(ReplySock, *(RqPacket + Index),
163 (SIZEP_T)strlen(*(RqPacket + Index)), 0, DestAddr,
164 (SIZEP_T)socklen);
165 if (rc == -1) {
166 strerror_r(errno, errorBuffer, ERROR_BUFFER_LEN);
167 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
168 "SSDP_LIB: New Request Handler:"
169 "Error in socket(): %s\n",
170 errorBuffer);
171 ret = UPNP_E_SOCKET_WRITE;
172 goto end_NewRequestHandler;
173 }
174 }
175
176end_NewRequestHandler:
177 UpnpCloseSocket(ReplySock);
178
179 return ret;
180}

187inline int extractIPv6address( //
188 char* url,
189 char* address
190) {
191 int i = 0;
192 int j = 0;
193 int ret = 0;
194
195 while (url[i] != '[' && url[i] != '\0') {
196 i++;
197 }
198 if (url[i] == '\0') {
199 goto exit_function;
200 }
201
202 /* bracket has been found, we deal with an IPv6 address */
203 i++;
204 while (url[i] != '\0' && url[i] != ']') {
205 address[j] = url[i];
206 i++;
207 j++;
208 }
209 if (url[i] == '\0') {
210 goto exit_function;
211 }
212
213 if (url[i] == ']') {
214 address[j] = '\0';
215 ret = 1;
216 }
217
218exit_function:
219 return ret;
220}

483void ssdp_handle_device_request(http_message_t* hmsg,
484 struct sockaddr_storage* dest_addr) {
485 constexpr int MX_FUDGE_FACTOR{10};
486 int handle, start;
487 struct Handle_Info* dev_info = NULL;
488 memptr hdr_value;
489 int mx;
490 char save_char;
491 SsdpEvent event;
492 int ret_code;
493 SsdpSearchReply* threadArg = NULL;
494 ThreadPoolJob job;
495 int replyTime;
496 int maxAge;
497
498 memset(&job, 0, sizeof(job));
499
500 /* check man hdr. */
501 if (httpmsg_find_hdr(hmsg, HDR_MAN, &hdr_value) == NULL ||
502 memptr_cmp(&hdr_value, "\"ssdp:discover\"") != 0)
503 /* bad or missing hdr. */
504 return;
505 /* MX header. */
506 if (httpmsg_find_hdr(hmsg, HDR_MX, &hdr_value) == NULL ||
507 (mx = raw_to_int(&hdr_value, 10)) < 0)
508 return;
509 /* ST header. */
510 if (httpmsg_find_hdr(hmsg, HDR_ST, &hdr_value) == NULL)
511 return;
512 save_char = hdr_value.buf[hdr_value.length];
513 hdr_value.buf[hdr_value.length] = '\0';
514 ret_code = ssdp_request_type(hdr_value.buf, &event);
515 /* restore. */
516 hdr_value.buf[hdr_value.length] = save_char;
517 if (ret_code == -1)
518 /* bad ST header. */
519 return;
520
521 start = 0;
522 for (;;) {
523 HandleLock();
524 /* device info. */
525 switch (GetDeviceHandleInfo(start, (int)dest_addr->ss_family, &handle,
526 &dev_info)) {
527 case HND_DEVICE:
528 break;
529 default:
530 HandleUnlock();
531 /* no info found. */
532 return;
533 }
534 maxAge = dev_info->MaxAge;
535 HandleUnlock();
536
537 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "MAX-AGE = %d\n",
538 maxAge);
539 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "MX = %d\n",
540 event.Mx);
541 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "DeviceType = %s\n",
542 event.DeviceType);
543 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "DeviceUuid = %s\n",
544 event.UDN);
545 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "ServiceType = %s\n",
546 event.ServiceType);
547 threadArg = (SsdpSearchReply*)malloc(sizeof(SsdpSearchReply));
548 if (threadArg == NULL)
549 return;
550 threadArg->handle = handle;
551 memcpy(&threadArg->dest_addr, dest_addr, sizeof(threadArg->dest_addr));
552 threadArg->event = event;
553 threadArg->MaxAge = maxAge;
554
555 TPJobInit(&job, advertiseAndReplyThread, threadArg);
556 TPJobSetFreeFunction(&job, (free_routine)free);
557
558 /* Subtract a percentage from the mx to allow for network and processing
559 * delays (i.e. if search is for 30 seconds, respond
560 * within 0 - 27 seconds). */
561 if (mx >= 2)
562 mx -= std::max(1, mx / MX_FUDGE_FACTOR);
563 if (mx < 1)
564 mx = 1;
565 replyTime = rand() % mx;
566 TimerThreadSchedule(&gTimerThread, replyTime, REL_SEC, &job, SHORT_TERM,
567 NULL);
568 start = handle;
569 }
570}

572int DeviceAdvertisement(char* DevType, int RootDev, char* Udn, char* Location,
573 int Duration, int AddressFamily, int PowerState,
574 int SleepPeriod, int RegistrationState) {
575 struct sockaddr_storage __ss;
576 struct sockaddr_in* DestAddr4 = (struct sockaddr_in*)&__ss;
577 struct sockaddr_in6* DestAddr6 = (struct sockaddr_in6*)&__ss;
578 /* char Mil_Nt[LINE_SIZE] */
579 char Mil_Usn[LINE_SIZE];
580 char* msgs[3];
581 int ret_code = UPNP_E_OUTOF_MEMORY;
582 int rc = 0;
583
584 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
585 "In function DeviceAdvertisement\n");
586 memset(&__ss, 0, sizeof(__ss));
587 switch (AddressFamily) {
588 case AF_INET:
589 DestAddr4->sin_family = (sa_family_t)AF_INET;
590 inet_pton(AF_INET, SSDP_IP, &DestAddr4->sin_addr);
591 DestAddr4->sin_port = htons(SSDP_PORT);
592 break;
593 case AF_INET6:
594 DestAddr6->sin6_family = (sa_family_t)AF_INET6;
595 inet_pton(AF_INET6,
596 (isUrlV6UlaGua(Location)) ? SSDP_IPV6_SITELOCAL
597 : SSDP_IPV6_LINKLOCAL,
598 &DestAddr6->sin6_addr);
599 DestAddr6->sin6_port = htons(SSDP_PORT);
600 DestAddr6->sin6_scope_id = gIF_INDEX;
601 break;
602 default:
603 UpnpPrintf(UPNP_CRITICAL, SSDP, __FILE__, __LINE__,
604 "Invalid device address family.\n");
605 }
606 msgs[0] = NULL;
607 msgs[1] = NULL;
608 msgs[2] = NULL;
609 // If device is a root device, here we need to send 3 advertisement or reply
610 if (RootDev) {
611 rc = snprintf(Mil_Usn, sizeof(Mil_Usn), "%s::upnp:rootdevice", Udn);
612 if (rc < 0 || (unsigned int)rc >= sizeof(Mil_Usn))
613 goto error_handler;
614 CreateServicePacket(MSGTYPE_ADVERTISEMENT, "upnp:rootdevice", Mil_Usn,
615 Location, Duration, &msgs[0], AddressFamily,
616 PowerState, SleepPeriod, RegistrationState);
617 }
618 /* both root and sub-devices need to send these two messages */
619 CreateServicePacket(MSGTYPE_ADVERTISEMENT, Udn, Udn, Location, Duration,
620 &msgs[1], AddressFamily, PowerState, SleepPeriod,
621 RegistrationState);
622 rc = snprintf(Mil_Usn, sizeof(Mil_Usn), "%s::%s", Udn, DevType);
623 if (rc < 0 || (unsigned int)rc >= sizeof(Mil_Usn))
624 goto error_handler;
625 CreateServicePacket(MSGTYPE_ADVERTISEMENT, DevType, Mil_Usn, Location,
626 Duration, &msgs[2], AddressFamily, PowerState,
627 SleepPeriod, RegistrationState);
628 /* check error */
629 if ((RootDev && msgs[0] == NULL) || msgs[1] == NULL || msgs[2] == NULL) {
630 goto error_handler;
631 }
632 /* send packets */
633 if (RootDev) {
634 /* send 3 msg types */
635 ret_code = NewRequestHandler((struct sockaddr*)&__ss, 3, &msgs[0]);
636 } else { /* sub-device */
637
638 /* send 2 msg types */
639 ret_code = NewRequestHandler((struct sockaddr*)&__ss, 2, &msgs[1]);
640 }
641
642error_handler:
643 /* free msgs */
644 free(msgs[0]);
645 free(msgs[1]);
646 free(msgs[2]);
647
648 return ret_code;
649}

652 char* Location, int Duration, int ByType, int PowerState,

654 int ret_code = UPNP_E_OUTOF_MEMORY;

659 int rc = 0;
660
661 msgs[0] = NULL;
662 msgs[1] = NULL;
663 if (RootDev) {
664 /* one msg for root device */
665 num_msgs = 1;
666
667 rc = snprintf(Mil_Usn, sizeof(Mil_Usn), "%s::upnp:rootdevice", Udn);
668 if (rc < 0 || (unsigned int)rc >= sizeof(Mil_Usn))
669 goto error_handler;
670 CreateServicePacket(MSGTYPE_REPLY, "upnp:rootdevice", Mil_Usn, Location,
671 Duration, &msgs[0], (int)DestAddr->sa_family,
672 PowerState, SleepPeriod, RegistrationState);
673 } else {
674 /* two msgs for embedded devices */
675 num_msgs = 1;
676
677 /*NK: FIX for extra response when someone searches by udn */
678 if (!ByType) {
679 CreateServicePacket(MSGTYPE_REPLY, Udn, Udn, Location, Duration,
680 &msgs[0], (int)DestAddr->sa_family, PowerState,
681 SleepPeriod, RegistrationState);
682 } else {
683 rc = snprintf(Mil_Usn, sizeof(Mil_Usn), "%s::%s", Udn, DevType);
684 if (rc < 0 || (unsigned int)rc >= sizeof(Mil_Usn))
685 goto error_handler;
686 CreateServicePacket(MSGTYPE_REPLY, DevType, Mil_Usn, Location,
687 Duration, &msgs[0], (int)DestAddr->sa_family,
688 PowerState, SleepPeriod, RegistrationState);
689 }
690 }
691 /* check error */
692 for (i = 0; i < num_msgs; i++) {
693 if (msgs[i] == NULL) {
694 goto error_handler;
695 }
696 }
697 /* send msgs */
698 ret_code = NewRequestHandler(DestAddr, num_msgs, msgs);
699
700error_handler:
701 for (i = 0; i < num_msgs; i++) {
702 if (msgs[i] != NULL)
703 free(msgs[i]);
704 }
705
706 return ret_code;
707}
