14uint8_t netmask_to_bitmask(const ::sockaddr_storage* a_netmask) {
15 TRACE("Executing netmask_to_bitmask()")
16 if (a_netmask == nullptr)
17 throw std::runtime_error(
18 UPnPsdk_LOGEXCEPT +
19 "MSG1059: No network socket address information given.\n");
20
21 uint8_t bitmask{};
22 uint8_t nullbits{};
23 switch (a_netmask->ss_family) {
24 case AF_INET6: {
25 // Count set bits.
26 for (size_t i{}; i < sizeof(in6_addr); i++) {
27 uint8_t s6addr = reinterpret_cast<const ::sockaddr_in6*>(a_netmask)
28 ->sin6_addr.s6_addr[i];
29 if (s6addr == 255) {
30 bitmask += 8;
31 } else {
32 while (s6addr) {
33 bitmask++;
34 s6addr <<= 1;
35 }
36 break; // for() loop
37 }
38 }
39 // Check if all remaining bits are zero.
40 for (int i{sizeof(in6_addr) - 1}; i >= 0; i--) {
41 uint8_t s6addr = reinterpret_cast<const ::sockaddr_in6*>(a_netmask)
42 ->sin6_addr.s6_addr[i];
43 if (s6addr == 0) {
44 nullbits += 8;
45 } else {
46 while ((s6addr & 1) == 0) {
47 nullbits++;
48 s6addr >>= 1;
49 }
50 break; // for() loop
51 }
52 }
53 // Check valid netmask.
54 if (bitmask + nullbits != 128) {
55 char ip_str[INET6_ADDRSTRLEN]{};
56 ::inet_ntop(
57 AF_INET6,
58 &reinterpret_cast<const ::sockaddr_in6*>(a_netmask)->sin6_addr,
59 ip_str, sizeof(ip_str));
60 throw std::runtime_error(
61 UPnPsdk_LOGEXCEPT +
62 "MSG1067: Invalid ip-address prefix bitmask \"[" +
63 std::string(ip_str) + "]\".\n");
64 }
65 } break; // switch()
66
67 case AF_INET: {
68 in_addr_t saddr = ntohl(
69 reinterpret_cast<const ::sockaddr_in*>(a_netmask)->sin_addr.s_addr);
70 in_addr_t saddr0 = saddr;
71 // Count set bits.
72 while (saddr) {
73 bitmask++;
74 saddr <<= 1;
75 }
76 // Check if all remaining bits are zero.
77 for (int i{}; (saddr0 & 1) == 0 && i < 32; i++) {
78 nullbits++;
79 saddr0 >>= 1;
80 }
81 // Check valid netmask.
82 if (bitmask + nullbits != 32) {
83 char ip_str[INET_ADDRSTRLEN]{};
84 ::inet_ntop(
85 AF_INET,
86 &reinterpret_cast<const ::sockaddr_in*>(a_netmask)->sin_addr,
87 ip_str, sizeof(ip_str));
88 throw std::runtime_error(UPnPsdk_LOGEXCEPT +
89 "MSG1069: Invalid ip-address netmask \"" +
90 std::string(ip_str) + "\".\n");
91 }
92 } break;
93
94 case AF_UNSPEC:
95 return 0;
96
97 default:
98 throw std::runtime_error(
99 UPnPsdk_LOGEXCEPT + "MSG1028: Unsupported address family(" +
100 std::to_string(a_netmask->ss_family) + "), only AF_INET6(" +
101 std::to_string(AF_INET6) + "), AF_INET(" + std::to_string(AF_INET) +
102 "), or AF_UNSPEC(" + std::to_string(AF_UNSPEC) + ") are valid.\n");
103 } // switch()
104
105 return bitmask;
106}

109void bitmask_to_netmask(const ::sockaddr_storage* a_saddr,
110 const uint8_t a_prefixlength, SSockaddr& a_saddrObj) {
111 TRACE("Executing bitmask_to_netmask()")
112 if (a_saddr == nullptr)
113 throw std::runtime_error(
114 UPnPsdk_LOGEXCEPT +
115 "MSG1070: No associated socket address for the netmask given.\n");
116
117 switch (a_saddr->ss_family) {
118 case AF_INET6: {
119 // I have to manage 16 bytes from the binary IPv6 address to set its
120 // bits. 15 bytes are all ones or all zero bits. Only one byte may have
121 // partly ones and zero bits. First I calculate how many leading bytes
122 // have full one bits. Following is the partial set byte. The rest are
123 // bytes with full zero bits set.
124 in6_addr netmask6{};
125
126 // All prefix lengths > 128 will throw this exception.
127 if (a_prefixlength > 128)
128 throw std::runtime_error(
129 UPnPsdk_LOGEXCEPT +
130 "MSG1124: Invalid IPv6 address prefix bitmask(" +
131 std::to_string(a_prefixlength) + ") exceeds 128.\n");
132
133 // Calculate number of leading bytes with full one bits.
134 int ones_bytes{a_prefixlength / 8};
135
136 // Fill leading bytes with all bit ones.
137 int i{};
138 for (; i < ones_bytes; i++)
139 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
140
141 // Handle the one partly bit-set byte.
142 if (i < 16 /*bytes*/) {
143 // Preset byte with all bit ones.
144 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
145 // Shift remaining zero bits from right into byte with using the
146 // reminder from the byte devision.
147 netmask6.s6_addr[i] <<= (8 - (a_prefixlength % 8));
148 i++;
149
150 // Fill remaining bytes with zero bits.
151 for (; i < 16 /*bytes*/; i++)
152 netmask6.s6_addr[i] = 0;
153 }
154
155 // Return the result.
156 a_saddrObj = "";
157 a_saddrObj.sin6.sin6_family = AF_INET6;
158 memcpy(&a_saddrObj.sin6.sin6_addr, &netmask6,
159 sizeof(a_saddrObj.sin6.sin6_addr));
160 } break;
161
162 case AF_INET: {
163 // All prefix lengths > 32 will throw this exception.
164 if (a_prefixlength > 32) {
165 throw std::runtime_error(
166 UPnPsdk_LOGEXCEPT +
167 "MSG1125: Invalid IPv4 address prefix bitmask(" +
168 std::to_string(a_prefixlength) + ") exceeds 32.\n");
169 }
170 // Prepare socket address and shift zero bits from right into the mask.
171 ::sockaddr_in saddr{};
172 saddr.sin_family = AF_INET;
173 if (a_prefixlength) {
174 in_addr_t bitshift{~0u}; // all one bits
175 bitshift <<= (32 - a_prefixlength); // shift zero bits from right
176 saddr.sin_addr.s_addr = htonl(bitshift);
177 }
178 // Return the result.
179 a_saddrObj.sin = saddr;
180 } break;
181
182 case AF_UNSPEC: {
183 a_saddrObj = "";
184 } break;
185
186 default: {
187 throw std::runtime_error(
188 UPnPsdk_LOGEXCEPT + "MSG1126: Unsupported address family(" +
189 std::to_string(a_saddr->ss_family) + "), only AF_INET6(" +
190 std::to_string(AF_INET6) + "), AF_INET(" + std::to_string(AF_INET) +
191 "), or AF_UNSPEC(" + std::to_string(AF_UNSPEC) + ") are valid.\n");
192 }
193 } // switch
194}
