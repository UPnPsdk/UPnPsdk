30void bitnum_to_netmask(
32 const sa_family_t a_family,
35 uint8_t a_prefixlength,
38 SSockaddr& a_saddrObj) {
39 switch (a_family) {
40 case AF_INET6: {
41 // I have to manage 16 bytes from the binary IPv6 address to set its
42 // bits. 15 bytes are all ones or all zero bits. Only one byte may have
43 // partly ones and zero bits. First I calculate how many leading bytes
44 // have full one bits. Following is the partial set byte. The rest are
45 // bytes with full zero bits set.
46 in6_addr netmask6{};
47
48 // Also all prefix length > 128 will return a full netmask. Note that we
49 // have an unsigned value.
50 if (a_prefixlength > 128) {
51 UPnPsdk_LOGCRIT "MSG1124: Invalid IPv6 address prefix bit number("
52 << std::to_string(a_prefixlength)
53 << "). Continue with truncated to 128.\n";
54 a_prefixlength = 128;
55 }
56
57 // Calculate number of leading bytes with full one bits.
58 int ones_bytes{a_prefixlength / 8};
59
60 // Fill leading bytes with all bit ones.
61 int i{0};
62 for (; i < ones_bytes; i++)
63 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
64
65 // Handle the one partly bit-set byte.
66 if (i < 16 /*bytes*/) {
67 // Preset byte with all bit ones.
68 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
69 // Shift remaining zero bits from right into byte with using the
70 // reminder from the byte devision.
71 netmask6.s6_addr[i] = netmask6.s6_addr[i]
72 << (8 - (a_prefixlength % 8));
73 i++;
74
75 // Fill remaining bytes with zero bits.
76 for (; i < 16 /*bytes*/; i++)
77 netmask6.s6_addr[i] = 0;
78 }
79
80 // Return the result.
81 a_saddrObj = "";
82 a_saddrObj.sin6.sin6_family = AF_INET6;
83 memcpy(&a_saddrObj.sin6.sin6_addr, &netmask6,
84 sizeof(a_saddrObj.sin6.sin6_addr));
85 } break;
86
87 case AF_INET: {
88 if (a_prefixlength > 32) {
89 UPnPsdk_LOGCRIT "MSG1125: Invalid IPv4 address prefix bit number("
90 << std::to_string(a_prefixlength)
91 << "). Continue with truncated to 32.\n";
92 a_prefixlength = 32;
93 }
94 in_addr netmask4;
95 netmask4.s_addr = ~0u; // all one bits
96 netmask4.s_addr = htonl(
97 netmask4.s_addr
98 << (32 - a_prefixlength)); // shift zero bits from right into mask
99
100 // Return the result.
101 a_saddrObj = "";
102 a_saddrObj.sin.sin_family = AF_INET;
103 memcpy(&a_saddrObj.sin.sin_addr, &netmask4,
104 sizeof(a_saddrObj.sin.sin_addr));
105 } break;
106
107 default: {
108 UPnPsdk_LOGCRIT "MSG1126: Invalid address family("
109 << a_family << "), only AF_INET6(" << AF_INET6 << ") or AF_INET("
110 << AF_INET << ") are valid. Continue with AF_UNSPEC(" << AF_UNSPEC
111 << ").\n";
112 a_saddrObj = "";
113 a_saddrObj.ss.ss_family = AF_UNSPEC;
114 }
115 } // switch
116}

130void CNetadapter_platform::get_first() {
131 // For the structure look at
132 // REF:_https://docs.microsoft.com/en-us/windows/win32/api/iptypes/ns-iptypes-ip_adapter_addresses_lh
133 TRACE2(this, " Executing CNetadapter_platform::get_first()")
134
135 ULONG adapts_size{};
136 // Get Adapters addresses required size. Check with adapts_size = 0
137 // will fail with ERROR_BUFFER_OVERFLOW but returns required size.
138 ULONG ret = umock::iphlpapi_h.GetAdaptersAddresses(
139 AF_UNSPEC, GAA_FLAG_SKIP_ANYCAST | GAA_FLAG_SKIP_DNS_SERVER, nullptr,
140 m_adapt_first, &adapts_size);
141 if (ret != ERROR_BUFFER_OVERFLOW) {
142 throw std::runtime_error(
143 UPnPsdk_LOGEXCEPT +
144 "MSG1120: failed to get buffer size for list of adapters (errid=" +
145 std::to_string(ret) + ").");
146 }
147
148 this->free_adaptaddrs();
149
150 // Allocate enough memory that size was detected above.
151 m_adapt_first = static_cast<PIP_ADAPTER_ADDRESSES>(malloc(adapts_size));
152 if (m_adapt_first == nullptr) {
153 throw std::runtime_error(
154 UPnPsdk_LOGEXCEPT +
155 "MSG1121: failed to allocate memory for list of adapters.");
156 }
157 // Do the call that will actually return the info.
158 ret = umock::iphlpapi_h.GetAdaptersAddresses(
159 AF_UNSPEC, GAA_FLAG_SKIP_ANYCAST | GAA_FLAG_SKIP_DNS_SERVER, nullptr,
160 m_adapt_first, &adapts_size);
161 if (ret != ERROR_SUCCESS) {
162 throw std::runtime_error(
163 UPnPsdk_LOGEXCEPT +
164 "MSG1122: failed to find list of adapters (errid=" +
165 std::to_string(ret) + ").");
166 }
167
168 m_adapt_current = m_adapt_first;
169 m_unicastaddr_current = m_adapt_current->FirstUnicastAddress;
170}

172bool CNetadapter_platform::get_next() {
173 // A network adapter has several IP addresses. First the address list must
174 // be parsed and when at its end the next adapter must be selected. When
175 // entering this function there is no synchronous information as given in a
176 // loop (or two nested). I have to excamine the current state for the next
177 // action. Writing down a state table on paper has helped me.
178 TRACE2(this, " Executing CNetadapter_platform::get_next()")
179 if (m_adapt_current == nullptr) {
180 m_unicastaddr_current = nullptr;
181 return false;
182 }
183 if (m_adapt_current != nullptr && m_unicastaddr_current == nullptr) {
184 m_adapt_current = m_adapt_current->Next;
185 if (m_adapt_current == nullptr) {
186 return false;
187 } else {
188 m_unicastaddr_current = m_adapt_current->FirstUnicastAddress;
189 return true;
190 }
191 }
192 if (m_adapt_current != nullptr && m_unicastaddr_current != nullptr) {
193 m_unicastaddr_current = m_unicastaddr_current->Next;
194 if (m_unicastaddr_current == nullptr) {
195 m_adapt_current = m_adapt_current->Next;
196 if (m_adapt_current == nullptr) {
197 return false;
198 } else {
199 m_unicastaddr_current = m_adapt_current->FirstUnicastAddress;
200 return true;
201 }
202 } else {
203 return true;
204 }
205 }
206 throw std::invalid_argument(
207 UPnPsdk_LOGEXCEPT +
208 "MSG1123: Failed to get next network adapter entry. This should never "
209 "come up and must be fixed. m_adapt_current=" +
210 std::to_string(reinterpret_cast<uintptr_t>(m_adapt_current)) +
211 ", m_unicastaddr_current=" +
212 std::to_string(reinterpret_cast<uintptr_t>(m_unicastaddr_current)) +
213 "\n");
214}

216bool CNetadapter_platform::find_first(const std::string& a_name_or_addr) {
217 TRACE2(this, " Executing CNetadapter_platform::find_first(" +
218 a_name_or_addr + ")")
219 // First look for a local network adapter name
220 this->reset();
221 do {
222 if (this->name() == a_name_or_addr)
223 return true;
224 } while (this->get_next());
225
226 // No name found, look for the ip address of a local network adapter.
227 // Try to translate input argument to a socket address.
228 CAddrinfo ainfoObj(a_name_or_addr, AI_NUMERICHOST, 0);
229 if (!ainfoObj.get_first())
230 return false;
231
232 // Valid ip address string given as input argument. Get its socket address.
233 SSockaddr sa_inputObj{};
234 ainfoObj.sockaddr(sa_inputObj);
235
236 // Parse network adapter list for the given input argument.
237 SSockaddr sa_nadObj{};
238 this->reset();
239 do {
240 this->sockaddr(sa_nadObj);
241 if (sa_nadObj == sa_inputObj)
242 return true;
243 } while (this->get_next());
244
245 return false;
246}

262std::string CNetadapter_platform::name() const {
263 TRACE2(this, " Executing CNetadapter_platform::name()")
264 if (m_adapt_current == nullptr)
265 return "";
266
267 /* Partial fix for Windows: Friendly name is wchar string, but currently
268 * compatible names are char string. For now try to convert it, which will
269 * work with many (but not all) adapters. A full fix would require a lot of
270 * big changes (using UTF-8 as planned). */
271
272 // If I need this constant more than once, I'll make it global, Here it is
273 // used for the friendly name of a network adapter and is compatible to old
274 // gIF_NAME size. ('grep' is your friend)
275 constexpr size_t LINE_SIZE{180};
276
277 char if_name[LINE_SIZE]{};
278 size_t i;
279 wcstombs_s(&i, if_name, sizeof(if_name), m_adapt_current->FriendlyName,
280 sizeof(if_name) - 1); // -1 so the appended '\0' doesn't fall
281 // outside the allocated buffer
282 return if_name;
283}

308unsigned int CNetadapter_platform::index() const {
309 TRACE2(this, " Executing CNetadapter_platform::index()")
310 if (m_adapt_current == nullptr)
311 return 0;
312 // No matter if the adapter supports only IPv4 or only IPv6 or both, the
313 // adapter index should be always the same.
314 if (m_adapt_current->IfIndex != 0) // IPv4 interface
315 return m_adapt_current->IfIndex;
316 if (m_adapt_current->Ipv6IfIndex != 0) // IPv6 interface
317 return m_adapt_current->Ipv6IfIndex;
318 return 0; // neither IPv4 nor IPv6
319}
