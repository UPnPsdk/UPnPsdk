24// Constructor for getting an address information with port number string.
25CAddrinfo::CAddrinfo(std::string_view a_node, std::string_view a_service,
26 const int a_family, const int a_socktype,
27 const int a_flags, const int a_protocol)
28 : m_node(a_node), m_service(a_service) {
29 TRACE2(this, " Construct CAddrinfo(..) with arguments")
30
31 // I cannot use the initialization list of the constructor because the
32 // member order in the structure addrinfo is different on Linux, MacOS and
33 // win32. I have to use the member names to initialize them, what's not
34 // possible for structures in the constructors init list.
35 m_hints.ai_flags = a_flags;
36 m_hints.ai_family = a_family;
37 m_hints.ai_socktype = a_socktype;
38 m_hints.ai_protocol = a_protocol;
39
40 // Correct weak hints:
41 // Always call is_netaddr() with AF_UNSPEC (default argument) to check
42 // all types. We always set AI_NUMERICHOST if it match, no matter what
43 // 'a_family' was reqested by the caller. But if AF_UNSPEC was requested,
44 // the found address family will also be set.
45 const sa_family_t addr_family = is_netaddr(m_node);
46 std::cerr << "DEBUG! addr_family=" << addr_family << "\n";
47 if (addr_family != AF_UNSPEC) { // Here we have a numeric netaddress
48 m_hints.ai_flags |= AI_NUMERICHOST;
49 if (m_hints.ai_family == AF_UNSPEC) { // Correct ai_family, we know it
50 m_hints.ai_family = addr_family;
51 }
52 }
53}

81 TRACE2("syscall ::freeaddrinfo() with m_res = ", m_res)

93// Setter to initialize an addrinfo and set it to the object
94// ---------------------------------------------------------
95// Get address information with cached hints.
96void CAddrinfo::init() {
97 TRACE2(this, " Executing CAddrinfo::init()")
98
99 // Temporary working copies: modified node and hints to use for syscall
100 // ::getaddrinfo().
101 std::string node;
102 addrinfo hints = m_hints;
103
104 if (is_netaddr(m_node, AF_INET6) == AF_INET6) {
105 // Here we have only ipv6 node strings representing a numeric ip
106 // address (netaddress) that is at least "[::]". Remove surounding
107 // brackets for ::getaddrinfo()
108 node = m_node.substr(1, m_node.length() - 2);
109 } else if (m_node == "::1") {
110 // Special case that isn't a valid netaddress but would be accepted by
111 // ::getaddrinfo(). So I make it also invalid for it.
112 node = "::1(no_brackets)";
113 } else {
114 // Here we have an ipv4 netaddress or a non numeric node name (no
115 // netaddress). Is it a valid (maybe alphanumeric) node name?
116 // ::getaddrinfo() shall decide it.
117 node = m_node;
118 }
119 const char* c_node = node.empty() ? nullptr : node.c_str();
120 std::string node_out = node.empty() ? "nullptr" : "\"" + node + "\"";
121
122 addrinfo* new_res{nullptr};
123
124 // syscall ::getaddrinfo() with prepared arguments
125 // -----------------------------------------------
126 int ret =
127 umock::netdb_h.getaddrinfo(c_node, m_service.c_str(), &hints, &new_res);
128
129 // Very helpful for debugging to see what is given to ::getaddrinfo()
130 // clang-format off
131 UPNPLIB_LOGINFO << "MSG1114: syscall ::getaddrinfo(" << node_out
132 << ", " << "\"" << m_service << "\", "
133 << &hints << ", " << &new_res
134 << ") node=\"" << m_node << "\", "
135 << (hints.ai_flags & AI_NUMERICHOST ? "AI_NUMERICHOST, " : "")
136 << (hints.ai_flags & AI_NUMERICSERV ? "AI_NUMERICSERV, " : "")
137 << (hints.ai_flags & AI_PASSIVE ? "AI_PASSIVE, " : "")
138 << (m_hints.ai_family == AF_INET6 ? "AF_INET6" :
139 (m_hints.ai_family == AF_INET ? "AF_INET" :
140 "hints.ai_family=" + m_hints.ai_family))
141 << (ret != 0
142 ? ". Get ERROR"
143 : ". Get \"" + to_netaddrp(reinterpret_cast
144 <const sockaddr_storage*>(new_res->ai_addr)) + "\"") << "\n";
145
146 if (ret == EAI_NONAME || ret == EAI_AGAIN) {
147 // Node or service not known (EAI_NONAME) or the name server returned a
148 // temporary failure indication. Try again later (EAI_AGAIN). Maybe an
149 // alphanumeric node name that cannot be resolved (e.g. by DNS)?
150 // Anyway, the user has to decide what to do. Because this depends on
151 // extern available DNS server the error can occur unexpectedly at any
152 // time. We have no influence on it.
153 throw std::runtime_error(UPNPLIB_LOGEXCEPT + "MSG1111: errid(" +
154 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\", " +
155 ((hints.ai_family == AF_UNSPEC) ? "IPv?_" :
156 ((hints.ai_family == AF_INET6) ? "IPv6_" : "IPv4_")) +
157 ((hints.ai_flags & AI_NUMERICHOST) ? "numeric_host=\"" : "alphanum_name=\"") +
158 node + "\", service=\"" +
159 m_service + "\"" +
160 ((hints.ai_flags & AI_PASSIVE) ? ", passive_listen" : "") +
161 ((hints.ai_flags & AI_NUMERICHOST) ? "" : ", (maybe DNS query temporary failed?)"));
162 }
163 // clang-format on
164
165 if (ret != 0) {
166 throw std::invalid_argument(
167 UPNPLIB_LOGEXCEPT +
168 "MSG1037: Failed to get address information: errid(" +
169 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\"\n");
170 }
171 // Different on platforms: man getsockaddr says "Specifying 0 in
172 // hints.ai_socktype indicates that socket addresses of any type can be
173 // returned". Linux returns SOCK_STREAM, MacOS returns SOCK_DGRAM and win32
174 // returns 0. To be portable we always return 0 in this case, the same as
175 // requested by the user.
176 if (m_hints.ai_socktype == 0)
177 new_res->ai_socktype = 0;
178 // Different on platforms: Ubuntu & MacOS return protocol number,
179 // win32 returns 0. We just return what was requested by the user.
180 new_res->ai_protocol = m_hints.ai_protocol;
181 // Different on platforms: Ubuntu returns set flags, MacOS & win32
182 // return 0. We just return what was requested by the user.
183 new_res->ai_flags = m_hints.ai_flags;
184 // Man getaddrinfo says: "If service is NULL, then the port number
185 // of the returned socket addresses will be left uninitialized." So
186 // we set it definitely to 0.
187 if (m_service.empty())
188 // port for AF_INET6 is also valid for AF_INET
189 reinterpret_cast<sockaddr_in6*>(new_res->ai_addr)->sin6_port = 0;
190
191 TRACE2("syscall ::getaddrinfo() with new_res = ", new_res)
192 // If init() is called the second time then m_res still points to the
193 // previous allocated memory. To avoid a memory leak it must be freed
194 // before pointing to the new allocated memory.
195 this->free_addrinfo();
196 // finaly point to the new address information from the operating system.
197 m_res = new_res;
198}

200// Getter for the assosiated ip address with port
201// ----------------------------------------------
202// e.g. "[2001:db8::2]:50001" or "192.168.254.253:50001".
203Netaddr CAddrinfo::netaddr() const noexcept {
204 // TRACE not usable with chained output.
205 // TRACE2(this, " Executing SSockaddr::get_netaddrp()")
206 Netaddr netaddr;
207 if (m_res == &m_hints)
208 return netaddr; // no information available
209
210 netaddr = to_netaddrp( // noexcept
211 reinterpret_cast<sockaddr_storage*>(m_res->ai_addr));
212
213 return netaddr;
214}
