29// I simply use the system function %inet_pton() to check if the node string is
30// accepted.
31sa_family_t is_netaddr(const std::string& a_node,
32 const int a_addr_family = AF_UNSPEC) noexcept {
33 // clang-format off
34 TRACE("Executing is_netaddr(\"" + a_node + "\", " +
35 (a_addr_family == AF_INET6 ? "AF_INET6" :
36 (a_addr_family == AF_INET ? "AF_INET" : "AF_UNSPEC")) + ")")
37 // clang-format on
38
39 // The shortest numeric netaddress is "[::]".
40 if (a_node.size() < 4) // noexcept
41 return AF_UNSPEC;
42
43 unsigned char buf[sizeof(in6_addr)];
44 switch (a_addr_family) {
45 case AF_INET6: {
46 // front() and back() have undefined behavior with an empty string.
47 // Here its size() is at least 4.
48 if (a_node.front() != '[' || a_node.back() != ']')
49 return AF_UNSPEC;
50 // Remove surounding brackets for inet_pton().
51 // substr() throws exception std::out_of_range if pos > size(), but that
52 // cannot occur due to the guard above, size() is at least 4 here and
53 // pos is fix 1.
54 const std::string node{a_node.substr(1, a_node.size() - 2)};
55 return ::inet_pton(AF_INET6, node.c_str(), buf) == 1 ? AF_INET6
56 : AF_UNSPEC;
57 }
58 case AF_INET:
59 return ::inet_pton(AF_INET, a_node.c_str(), buf) == 1 ? AF_INET
60 : AF_UNSPEC;
61 case AF_UNSPEC:
62 // Recursive call
63 // clang-format off
64 return is_netaddr(a_node, AF_INET6) ? AF_INET6 : false ||
65 is_netaddr(a_node, AF_INET) ? AF_INET : AF_UNSPEC;
66 // clang-format on
67 }
68 return AF_UNSPEC;
69}

80// Constructor for getting an address information with port number string.
81CAddrinfo::CAddrinfo(const std::string& a_node, const std::string& a_service,
82 const int a_family, const int a_socktype,
83 const int a_flags, const int a_protocol)
84 : m_node(a_node), m_service(a_service) {
85 TRACE2(this, " Construct CAddrinfo(..) with arguments")
86
87 // I cannot use the initialization list of the constructor because the
88 // member order in the structure addrinfo is different on Linux, MacOS and
89 // win32. I have to use the member names to initialize them, what's not
90 // possible for structures in the constructors init list.
91 m_hints.ai_flags = a_flags;
92 m_hints.ai_family = a_family;
93 m_hints.ai_socktype = a_socktype;
94 m_hints.ai_protocol = a_protocol;
95 m_hints.ai_addrlen = sizeof(this->ss);
96 m_hints.ai_addr = &this->sa;
97 m_hints.ai_canonname = m_empty_c_str;
98 m_hints.ai_next = nullptr;
99
100 // Correct weak hints:
101 // Always call is_netaddr() with AF_UNSPEC (default argument) to check
102 // all types. We always set AI_NUMERICHOST if it match, no matter what
103 // 'a_family' was reqested by the caller. But if AF_UNSPEC was requested,
104 // the found address family will also be set.
105 const sa_family_t addr_family = is_netaddr(m_node);
106 if (addr_family != AF_UNSPEC) {
107 m_hints.ai_flags |= AI_NUMERICHOST;
108 if (m_hints.ai_family == AF_UNSPEC) {
109 m_hints.ai_family = addr_family;
110 }
111 }
112}

123// Copy constructor
124// ----------------
125CAddrinfo::CAddrinfo(const CAddrinfo& that) : SSockaddr() {
126 // I cannot just copy the structure that m_res pointed to (*m_res =
127 // *that.m_res;). Copy works but MS Windows failed to destruct it with
128 // freeaddrinfo(m_res);. It throws an exception "A non-recoverable error
129 // occurred during a database lookup.". Seems there are also pointer within
130 // the addrinfo structure that are not deeply copied. So we have to go the
131 // hard way with getaddrinfo() and free it with freeaddrinfo(),
132
133 TRACE2(this, " Construct copy CAddrinfo()")
134 m_node = that.m_node;
135 m_service = that.m_service;
136 m_hints = that.m_hints;
137 // Next call is capable to throw exceptions but will not do in this case.
138 // The call was already successful done with initializing the object.
139 // Calling it again with unmodified hints should also always succeed.
140 if (that.m_res != &that.m_hints)
141 this->get_addrinfo();
142}

145// Copy assignment operator
146// ------------------------
147CAddrinfo& CAddrinfo::operator=(CAddrinfo that) noexcept {
148 TRACE2(this, " Executing CAddrinfo::operator=()")
149 // The argument by value ('that') was copied to the stack by the copy
150 // constructor. It contains also a pointer (m_res) to a new allocated
151 // addrinfo.
152 if (m_res != &m_hints && that.m_res != &that.m_hints)
153 std::swap(m_res, that.m_res);
154 else
155 this->free_addrinfo(); // noexcept
156 // The no longer needed current m_res pointer has been swapped to the
157 // stack and its resource will be deallocated by the destructor of the
158 // 'that' object when leaving this function.
159
160 std::swap(m_node, that.m_node);
161 std::swap(m_service, that.m_service);
162 std::swap(m_hints, that.m_hints);
163 std::swap(this->ss, that.ss);
164
165 // by convention, always return *this
166 return *this;
167}

177// Private method to free allocated memory for address information
178// ---------------------------------------------------------------
179void CAddrinfo::free_addrinfo() noexcept {
180 if (m_res != &m_hints) {
181 TRACE2("syscall ::freeaddrinfo() with m_res = ", m_res)
182 umock::netdb_h.freeaddrinfo(m_res);
183 m_res = &m_hints;
184 memset(&this->ss, 0, sizeof(this->ss));
185 }
186}

189// Compare operator ==

191bool CAddrinfo::operator==(const CAddrinfo& a_ai) const noexcept {
192 if (a_ai.m_res->ai_flags == m_res->ai_flags &&
193 a_ai.m_res->ai_family == m_res->ai_family &&
194 a_ai.m_res->ai_socktype == m_res->ai_socktype &&
195 a_ai.m_res->ai_protocol == m_res->ai_protocol &&
196 a_ai.m_res->ai_addrlen == m_res->ai_addrlen &&
197 sockaddrcmp(reinterpret_cast<sockaddr_storage*>(a_ai.m_res->ai_addr),
198 reinterpret_cast<sockaddr_storage*>(m_res->ai_addr)))
199 return true;
200
201 return false;
202}

210// Getter to get an addrinfo and set it to the object
211// --------------------------------------------------
212// Get address information with cached hints.
213void CAddrinfo::get_addrinfo() {
214 TRACE2(this, " Executing CAddrinfo::get_addrinfo()")
215
216 // Temporary working copies: modified node and hints to use for syscall
217 // ::getaddrinfo().
218 std::string node;
219 addrinfo hints = m_hints;
220 hints.ai_addrlen = 0;
221 hints.ai_addr = nullptr;
222 hints.ai_canonname = nullptr;
223 hints.ai_next = nullptr;
224
225 if (is_netaddr(m_node, AF_INET6) == AF_INET6) {
226 // Here we have only ipv6 node strings representing a numeric ip
227 // address (netaddress) that is at least "[::]". Remove surounding
228 // brackets for ::getaddrinfo()
229 node = m_node.substr(1, m_node.length() - 2);
230 } else if (m_node == "::1") {
231 // Special case that isn't a valid netaddress but would be accepted by
232 // ::getaddrinfo(). So I make it also invalid for it.
233 node = "::1(no_brackets)";
234 } else {
235 // Here we have a non numeric node name (no netaddress). Is it a valid
236 // alphanumeric node name? ::getaddrinfo() shall decide it.
237 node = m_node;
238 }
239 const char* c_node = node.empty() ? nullptr : node.c_str();
240 std::string node_out = node.empty() ? "nullptr" : "\"" + node + "\"";
241
242 addrinfo* new_res{nullptr};
243
244 // syscall ::getaddrinfo() with prepared arguments
245 // -----------------------------------------------
246 int ret =
247 umock::netdb_h.getaddrinfo(c_node, m_service.c_str(), &hints, &new_res);
248
249 // Very helpful for debugging to see what is given to ::getaddrinfo()
250 // clang-format off
251 UPNPLIB_LOGINFO << "MSG1114: syscall ::getaddrinfo("
252 << node_out << ", "
253 << "\"" + m_service + "\", "
254 << &hints << ", " << &new_res << ") Using node=\""
255 << m_node << "\", hints.ai_flags="
256 << m_hints.ai_flags << ", hints.ai_family="
257 << m_hints.ai_family
258 << (ret != 0
259 ? ". Get ERROR"
260 : ". Get \"" + to_netaddrp(reinterpret_cast
261 <const sockaddr_storage*>(new_res->ai_addr)) + "\"") << "\n";
262
263 if (ret == EAI_NONAME || ret == EAI_AGAIN) {
264 // Node or service not known (EAI_NONAME) or the name server returned a
265 // temporary failure indication. Try again later (EAI_AGAIN). Maybe an
266 // alphanumeric node name that cannot be resolved (e.g. by DNS)?
267 // Anyway, the user has to decide what to do. Because this depends on
268 // extern available DNS server the error can occur unexpectedly at any
269 // time. We have no influence on it.
270 throw std::runtime_error(UPNPLIB_LOGEXCEPT + "MSG1111: errid(" +
271 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\", " +
272 ((hints.ai_family == AF_UNSPEC) ? "IPv?_" :
273 ((hints.ai_family == AF_INET6) ? "IPv6_" : "IPv4_")) +
274 ((hints.ai_flags & AI_NUMERICHOST) ? "numeric_host=\"" : "alphanum_name=\"") +
275 node + "\", service=\"" +
276 m_service + "\"" +
277 ((hints.ai_flags & AI_PASSIVE) ? ", passive_listen" : "") +
278 ((hints.ai_flags & AI_NUMERICHOST) ? "" : ", (maybe DNS query temporary failed?)"));
279 }
280 // clang-format on
281
282 if (ret != 0) {
283 throw std::invalid_argument(
284 UPNPLIB_LOGEXCEPT +
285 "MSG1037: Failed to get address information: errid(" +
286 std::to_string(ret) + ")=\"" + ::gai_strerror(ret) + "\"");
287 }
288 // Different on platforms: man getsockaddr says "Specifying 0 in
289 // hints.ai_socktype indicates that socket addresses of any type can be
290 // returned". Linux returns SOCK_STREAM, MacOS returns SOCK_DGRAM and win32
291 // returns 0. To be portable we always return 0 in this case, the same as
292 // requested by the user.
293 if (m_hints.ai_socktype == 0)
294 new_res->ai_socktype = 0;
295 // Different on platforms: Ubuntu & MacOS return protocol number,
296 // win32 returns 0. We just return what was requested by the user.
297 new_res->ai_protocol = m_hints.ai_protocol;
298 // Different on platforms: Ubuntu returns set flags, MacOS & win32
299 // return 0. We just return what was requested by the user.
300 new_res->ai_flags = m_hints.ai_flags;
301 // Man getaddrinfo says: "If service is NULL, then the port number
302 // of the returned socket addresses will be left uninitialized." So
303 // we set it definitely to 0.
304 if (m_service.empty())
305 // port for AF_INET6 is also valid for AF_INET
306 reinterpret_cast<sockaddr_in6*>(new_res->ai_addr)->sin6_port = 0;
307
308 TRACE2("syscall ::getaddrinfo() with new_res = ", new_res)
309 // finaly store the address information from the operating system.
310 this->free_addrinfo();
311 m_res = new_res;
312 // Copy socket address from the address info to this socket address:
313 std::memcpy(&this->ss, m_res->ai_addr, m_res->ai_addrlen);
314}
