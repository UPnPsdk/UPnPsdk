112int host_header_is_numeric(
113 char* a_host_port,
114 size_t a_host_port_len
115) {
116 TRACE("Executing host_header_is_numeric()");
117 if (a_host_port_len == 0 || strncmp(a_host_port, "[::]", 4) == 0 ||

166int dispatch_request(
168 SOCKINFO* info,
170 http_parser_t* hparser) {
171 memptr header;
172 size_t min_size;
173 http_message_t* request;
174 MiniServerCallback callback;
175 WebCallback_HostValidate host_validate_callback = 0;
176 void* cookie{};
177 int rc = UPNP_E_SUCCESS;
178 /* If it does not fit in here, it is likely invalid anyway. */
179 char host_port[NAME_SIZE];
180
181 switch (hparser->msg.method) {
182 /* Soap Call */
183 case SOAPMETHOD_POST:
184 case HTTPMETHOD_MPOST:
185 callback = gSoapCallback;
186 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
187 "miniserver %d: got SOAP msg\n", info->socket);
188 break;
189 /* Gena Call */
190 case HTTPMETHOD_NOTIFY:
191 case HTTPMETHOD_SUBSCRIBE:
192 case HTTPMETHOD_UNSUBSCRIBE:
193 callback = gGenaCallback;
194 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
195 "miniserver %d: got GENA msg\n", info->socket);
196 break;
197 /* HTTP server call */
198 case HTTPMETHOD_GET:
199 case HTTPMETHOD_POST:
200 case HTTPMETHOD_HEAD:
201 case HTTPMETHOD_SIMPLEGET:
202 callback = gGetCallback;
203 host_validate_callback = gWebCallback_HostValidate;
204 cookie = gWebCallback_HostValidateCookie;
205 UPnPsdk_LOGINFO("MSG1107") "miniserver socket="
206 << info->socket << ": got WEB server msg.\n";
207 break;
208 default:
209 callback = 0;
210 }
211 if (!callback) {
212 rc = HTTP_INTERNAL_SERVER_ERROR;
213 goto ExitFunction;
214 }
215 request = &hparser->msg;
216 if (UPnPsdk::g_dbug) {
217 getNumericHostRedirection(info->socket, host_port, sizeof host_port);
218 UPnPsdk_LOGINFO("MSG1113") "Redirect host_port=\"" << host_port
219 << "\"\n";
220 }
221 /* check HOST header for an IP number -- prevents DNS rebinding. */
222 if (!httpmsg_find_hdr(request, HDR_HOST, &header)) {
223 rc = UPNP_E_BAD_HTTPMSG;
224 goto ExitFunction;
225 }
226 min_size = header.length < ((sizeof host_port) - 1)
227 ? header.length
228 : (sizeof host_port) - 1;
229 memcpy(host_port, header.buf, min_size);
230 host_port[min_size] = 0;
231 if (host_validate_callback) {
232 rc = host_validate_callback(host_port, cookie);
233 if (rc == UPNP_E_BAD_HTTPMSG) {
234 goto ExitFunction;
235 }
236 } else if (!host_header_is_numeric(host_port, min_size)) {
237 if (!gAllowLiteralHostRedirection) {
238 rc = UPNP_E_BAD_HTTPMSG;
239 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
240 "Possible DNS Rebind attack prevented.\n");
241 goto ExitFunction;
242 } else {
243 membuffer redir_buf;
244 static const char* redir_fmt = "HTTP/1.1 307 Temporary Redirect\r\n"
245 "Location: http://%s\r\n\r\n";
246 char redir_str[NAME_SIZE];
247 int timeout = HTTP_DEFAULT_TIMEOUT;
248
249 getNumericHostRedirection(info->socket, host_port,
250 sizeof host_port);
251 membuffer_init(&redir_buf);
252 snprintf(redir_str, NAME_SIZE, redir_fmt, host_port);
253 membuffer_append_str(&redir_buf, redir_str);
254 rc = http_SendMessage(info, &timeout, "b", redir_buf.buf,
255 redir_buf.length);
256 membuffer_destroy(&redir_buf);
257 goto ExitFunction;
258 }
259 }
260 callback(hparser, request, info);
261
262ExitFunction:
263 return rc;
264}

284void handle_request(
287 void* args) {
288 TRACE("Executing handle_request()")
289 int http_major_version{1};
290 int http_minor_version{1};
291 int ret_code;
292 mserv_request_t* request_in = (mserv_request_t*)args;
293 SOCKET connfd = request_in->connfd;
294
295 if (UPnPsdk::g_dbug) {
296 UPnPsdk::CSocket_basic sockObj(connfd);
297 sockObj.load();
298 UPnPsdk::SSockaddr local_saObj;
299 sockObj.local_saddr(&local_saObj);
300 UPnPsdk::SSockaddr remote_saObj;
301 remote_saObj = request_in->foreign_sockaddr;
302 UPnPsdk_LOGINFO("MSG1027") "UDevice socket="
303 << connfd << ": READING request on local=\""
304 << local_saObj.netaddrp() << "\" from control point remote=\""
305 << remote_saObj.netaddrp() << "\".\n";
306 }
307
308 /* parser_request_init( &parser ); */ /* LEAK_FIX_MK */
309 http_parser_t parser;
310 http_message_t* hmsg = &parser.msg;
311 SOCKINFO info;
312 ret_code = sock_init_with_ip(
313 &info, connfd,
314 reinterpret_cast<sockaddr*>(&request_in->foreign_sockaddr));
315 if (ret_code != UPNP_E_SUCCESS) {
316 free(request_in);
317 httpmsg_destroy(hmsg);
318 return;
319 }
320
321 /* read */
322 int timeout{HTTP_DEFAULT_TIMEOUT};
323 int http_error_code; // Either negative UPNP_E_XXX error code,
324 // or positve HTTP error code (4XX or 5XX).
325 // Will be initialized by next function.
326 ret_code = http_RecvMessage(&info, &parser, HTTPMETHOD_UNKNOWN, &timeout,
327 &http_error_code);
328 if (ret_code == 0) {
329 UPnPsdk_LOGINFO("MSG1106") "miniserver socket=" << connfd
330 << ": PROCESSING...\n";
331 /* dispatch */
332 http_error_code = dispatch_request(&info, &parser);
333 }
334
335 if (http_error_code > 0) { // only positive HTTP error codes (4XX or 5XX).
336 if (hmsg) {
337 http_major_version = hmsg->major_version;
338 http_minor_version = hmsg->minor_version;
339 }
340 http_SendStatusResponse(&info, http_error_code, http_major_version,
341 http_minor_version);
342 }
343 sock_destroy(&info, SD_BOTH);
344 httpmsg_destroy(hmsg);
345 free(request_in);
346
347 UPnPsdk_LOGINFO("MSG1058") "miniserver socket(" << connfd
348 << "); COMPLETE.\n";
349}

355UPNP_INLINE void schedule_request_job(
357 SOCKET a_connfd,
359 UPnPsdk::SSockaddr& clientAddr) {
360 TRACE("Executing schedule_request_job()")
361 if (UPnPsdk::g_dbug) {
362 UPnPsdk::CSocket_basic sockObj(a_connfd);
363 sockObj.load();
364 UPnPsdk::SSockaddr local_saObj;
365 sockObj.local_saddr(&local_saObj);
366 UPnPsdk::SSockaddr remote_saObj;
367 remote_saObj = clientAddr.ss;
368 UPnPsdk_LOGINFO(
369 "MSG1042") "Schedule UDevice to read incomming request with socket("
370 << a_connfd << ") local=\"" << local_saObj.netaddrp()
371 << "\" remote=\"" << remote_saObj.netaddrp() << "\".\n";
372 }
373
374 ThreadPoolJob job{};
375 mserv_request_t* request{
376 static_cast<mserv_request_t*>(std::malloc(sizeof(mserv_request_t)))};
377 if (request == nullptr) {
378 UPnPsdk_LOGCRIT("MSG1024") "Socket(" << a_connfd
379 << "): out of memory.\n";
380 sock_close(a_connfd);
381 return;
382 }
383 request->connfd = a_connfd;
384 memcpy(&request->foreign_sockaddr, &clientAddr.ss,
385 sizeof(request->foreign_sockaddr));
386 TPJobInit(&job, (start_routine)handle_request, request);
387 TPJobSetFreeFunction(&job, free_handle_request_arg);
388 TPJobSetPriority(&job, MED_PRIORITY);
389 if (ThreadPoolAdd(&gMiniServerThreadPool, &job, NULL) != 0) {
390 UPnPsdk_LOGERR("MSG1025") "Socket("
391 << a_connfd << "): failed to add job to miniserver threadpool.\n";
392 free(request);
393 sock_close(a_connfd);
394 }
395}

415void fdset_if_valid( //
416 SOCKET a_sock,
417 fd_set* a_set
420) {
421 TRACE("Executing fdset_if_valid(): check sockfd=" + std::to_string(a_sock))
422 if (a_sock == INVALID_SOCKET)
423 // This is a defined state and we return silently.
424 return;
425
426 if (a_sock < 3 || a_sock >= FD_SETSIZE) {
427 UPnPsdk_LOGERR("MSG1005")
428 << (a_sock < 0 ? "Invalid" : "Prohibited") << " socket " << a_sock
429 << " not set to be monitored by ::select()"
430 << (a_sock >= 3 ? " because it violates FD_SETSIZE.\n" : ".\n");
431 return;
432 }
433 // Check if socket is valid and bound
434 UPnPsdk::CSocket_basic sockObj(a_sock);
435 try {
436 sockObj.load(); // UPnPsdk::CSocket_basic
437 if (sockObj.local_saddr())
438
439 FD_SET(a_sock, a_set);
440
441 else
442 UPnPsdk_LOGINFO("MSG1002") "Unbound socket "
443 << a_sock << " not set to be monitored by ::select().\n";
444
445 } catch (const std::exception& e) {
446 if (UPnPsdk::g_dbug)
447 std::cerr << e.what();
448 UPnPsdk_LOGCATCH("MSG1009") "Invalid socket "
449 << a_sock << " not set to be monitored by ::select().\n";
450 }
451}

464int web_server_accept(
466 [[maybe_unused]] SOCKET listen_sock,
468 [[maybe_unused]] fd_set& set) {
469#ifndef COMPA_HAVE_WEBSERVER
470 return UPNP_E_NO_WEB_SERVER;
471#else
472 TRACE("Executing web_server_accept()")
473 if (listen_sock == INVALID_SOCKET || !FD_ISSET(listen_sock, &set)) {
474 UPnPsdk_LOGINFO("MSG1012") "Socket("
475 << listen_sock << ") invalid or not in file descriptor set.\n";
476 return UPNP_E_SOCKET_ERROR;
477 }
478
479 UPnPsdk::SSockaddr ctrlpnt_saObj;
480 socklen_t ctrlpntLen = sizeof(ctrlpnt_saObj.ss);
481
482 // accept a network request connection
483 SOCKET conn_sock =
484 umock::sys_socket_h.accept(listen_sock, &ctrlpnt_saObj.sa, &ctrlpntLen);
485 if (conn_sock == INVALID_SOCKET) {
486 UPnPsdk_LOGERR("MSG1022") "Error in ::accept(): "
487 << std::strerror(errno) << ".\n";
488 return UPNP_E_SOCKET_ACCEPT;
489 }
490
491 if (UPnPsdk::g_dbug) {
492 // Some helpful status information.
493 UPnPsdk::CSocket_basic listen_sockObj(listen_sock);
494 listen_sockObj.load();
495 UPnPsdk::SSockaddr listen_saObj;
496 listen_sockObj.local_saddr(&listen_saObj);
497
498 UPnPsdk::CSocket_basic conn_sockObj(conn_sock);
499 conn_sockObj.load();
500 UPnPsdk::SSockaddr conn_saObj;
501 conn_sockObj.local_saddr(&conn_saObj);
502
503 UPnPsdk_LOGINFO("MSG1023") "Listening socket("
504 << listen_sock << ") on \"" << listen_saObj.netaddrp()
505 << "\" accept connection socket(" << conn_sock << ") local=\""
506 << conn_saObj.netaddrp() << "\" to remote=\""
507 << ctrlpnt_saObj.netaddrp() << "\".\n";
508 }
509
510 // Schedule a job to manage the UPnP request from the remote control point.
511 schedule_request_job(conn_sock, ctrlpnt_saObj);
512
513 return UPNP_E_SUCCESS;
514#endif /* COMPA_HAVE_WEBSERVER */
515}

556int receive_from_stopSock(
557 SOCKET ssock,
558 fd_set* set
560) {
561 TRACE("Executing receive_from_stopSock()")
562 constexpr char shutdown_str[]{"ShutDown"};
563
564 if (!FD_ISSET(ssock, set))
565 return 0; // Nothing to do for this socket
566
567 UPnPsdk::sockaddr_t clientAddr{};
568 socklen_t clientLen{sizeof(clientAddr)}; // May be modified
569
570 // The receive buffer is one byte greater with '\0' than the max receiving
571 // bytes so the received message will always be terminated.
572 char receiveBuf[sizeof(shutdown_str) + 1]{};
573 char buf_ntop[INET6_ADDRSTRLEN];
574
575 // receive from
576 SSIZEP_T byteReceived = umock::sys_socket_h.recvfrom(
577 ssock, receiveBuf, sizeof(shutdown_str), 0, &clientAddr.sa, &clientLen);
578 if (byteReceived == SOCKET_ERROR ||
579 inet_ntop(AF_INET, &clientAddr.sin.sin_addr, buf_ntop,
580 sizeof(buf_ntop)) == nullptr) {
581 UPnPsdk_LOGCRIT("MSG1038") "Failed to receive data from socket "
582 << ssock << ". Stop miniserver.\n";
583 return 1;
584 }
585
586 // integer of "127.0.0.1" is 2130706433.
587 if (clientAddr.sin.sin_addr.s_addr != htonl(2130706433) ||
588 strcmp(receiveBuf, shutdown_str) != 0) //
589 {
590 char nullstr[]{"\\0"};
591 if (byteReceived == 0 || receiveBuf[byteReceived - 1] != '\0')
592 nullstr[0] = '\0';
593 UPnPsdk_LOGERR("MSG1039") "Received \""
594 << receiveBuf << nullstr << "\" from " << buf_ntop << ":"
595 << ntohs(clientAddr.sin.sin_port)
596 << ", must be \"ShutDown\\0\" from 127.0.0.1:*. Don't "
597 "stopping miniserver.\n";
598 return 0;
599 }
600
601 UPnPsdk_LOGINFO("MSG1040") "On socket "
602 << ssock << " received ordinary datagram \"" << receiveBuf
603 << "\\0\" from " << buf_ntop << ":" << ntohs(clientAddr.sin.sin_port)
604 << ". Stop miniserver.\n";
605 return 1;
606}

619void RunMiniServer(
623 MiniServerSockArray* miniSock) {
624 UPnPsdk_LOGINFO("MSG1085") "Executing...\n";
625
626#ifdef COMPA_HAVE_WEBSERVER
627 // Move the current valid socket objects to this scope. They are owner of
628 // the raw socket file descriptor and manage/close it when leaving this
629 // scope.
630 UPnPsdk::CSocket sockLlaObj;
631 if (miniSock->pSockLlaObj != nullptr) {
632 sockLlaObj = std::move(*miniSock->pSockLlaObj);
633 miniSock->pSockLlaObj = &sockLlaObj;
634 }
635 UPnPsdk::CSocket sockGuaObj;
636 if (miniSock->pSockGuaObj != nullptr) {
637 sockGuaObj = std::move(*miniSock->pSockGuaObj);
638 miniSock->pSockGuaObj = &sockGuaObj;
639 }
640 UPnPsdk::CSocket sockIp4Obj;
641 if (miniSock->pSockIp4Obj != nullptr) {
642 sockIp4Obj = std::move(*miniSock->pSockIp4Obj);
643 miniSock->pSockIp4Obj = &sockIp4Obj;
644 }
645#endif
646
647 fd_set expSet;
648 fd_set rdSet;
649 int stopSock = 0;
650
651 // On MS Windows INVALID_SOCKET is unsigned -1 = 18446744073709551615 so we
652 // get maxMiniSock with this big number even if there is only one
653 // INVALID_SOCKET. Incrementing it at the end results in 0. To be portable
654 // we must not assume INVALID_SOCKET to be -1. --Ingo
655 SOCKET maxMiniSock = 0;
656 maxMiniSock = //
657 std::max(maxMiniSock, miniSock->miniServerSock4 == INVALID_SOCKET
658 ? 0
659 : miniSock->miniServerSock4);
660 maxMiniSock = //
661 std::max(maxMiniSock, miniSock->miniServerSock6 == INVALID_SOCKET
662 ? 0
663 : miniSock->miniServerSock6);
664 maxMiniSock = //
665 std::max(maxMiniSock, miniSock->miniServerSock6UlaGua == INVALID_SOCKET
666 ? 0
667 : miniSock->miniServerSock6UlaGua);
668 maxMiniSock = //
669 std::max(maxMiniSock, miniSock->miniServerStopSock == INVALID_SOCKET
670 ? 0
671 : miniSock->miniServerStopSock);
672 maxMiniSock = //
673 std::max(maxMiniSock, miniSock->ssdpSock4 == INVALID_SOCKET
674 ? 0
675 : miniSock->ssdpSock4);
676 maxMiniSock = //
677 std::max(maxMiniSock, miniSock->ssdpSock6 == INVALID_SOCKET
678 ? 0
679 : miniSock->ssdpSock6);
680 maxMiniSock = //
681 std::max(maxMiniSock, miniSock->ssdpSock6UlaGua == INVALID_SOCKET
682 ? 0
683 : miniSock->ssdpSock6UlaGua);
684#ifdef COMPA_HAVE_CTRLPT_SSDP
685 maxMiniSock = //
686 std::max(maxMiniSock, miniSock->ssdpReqSock4 == INVALID_SOCKET
687 ? 0
688 : miniSock->ssdpReqSock4);
689 maxMiniSock = //
690 std::max(maxMiniSock, miniSock->ssdpReqSock6 == INVALID_SOCKET
691 ? 0
692 : miniSock->ssdpReqSock6);
693#endif
694 ++maxMiniSock;
695
696 gMServState = MSERV_RUNNING;
697 while (!stopSock) {
698 FD_ZERO(&rdSet);
699 FD_ZERO(&expSet);
700 /* FD_SET()'s */
701 FD_SET(miniSock->miniServerStopSock, &expSet);
702 FD_SET(miniSock->miniServerStopSock, &rdSet);
703 fdset_if_valid(miniSock->miniServerSock4, &rdSet);
704 fdset_if_valid(miniSock->miniServerSock6, &rdSet);
705 fdset_if_valid(miniSock->miniServerSock6UlaGua, &rdSet);
706 fdset_if_valid(miniSock->ssdpSock4, &rdSet);
707 fdset_if_valid(miniSock->ssdpSock6, &rdSet);
708 fdset_if_valid(miniSock->ssdpSock6UlaGua, &rdSet);
709#ifdef COMPA_HAVE_CTRLPT_SSDP
710 fdset_if_valid(miniSock->ssdpReqSock4, &rdSet);
711 fdset_if_valid(miniSock->ssdpReqSock6, &rdSet);
712#endif
713
714 /* select() */
715 int ret = umock::sys_socket_h.select(static_cast<int>(maxMiniSock),
716 &rdSet, NULL, &expSet, NULL);
717
718 if (ret == SOCKET_ERROR) {
719 if (errno == EINTR) {
720 // A signal was caught, not for us. We ignore it and
721 continue;
722 }
723 if (errno == EBADF) {
724 // A closed socket file descriptor was given in one of the
725 // sets. For details look at
726 // REF:_[Should_I_assert_fail_on_select()_EBADF?](https://stackoverflow.com/q/28015859/5014688)
727 // It is difficult to determine here what file descriptor
728 // in rdSet or expSet is invalid. So I ensure that only valid
729 // socket fds are given by checking them with fdset_if_valid()
730 // before calling select(). Doing this I have to
731 continue;
732 }
733 // All other errors EINVAL and ENOMEM are critical and cannot
734 // continue run mininserver.
735 UPnPsdk_LOGCRIT("MSG1021") "Error in ::select(): "
736 << std::strerror(errno) << ".\n";
737 break;
738 }
739
740 // Accept requested connection from a remote control point and run the
741 // connection in a new thread. Due to side effects with threading we
742 // need to avoid lazy evaluation with chained || because all
743 // web_server_accept() must be called.
744 // if (ret1 == UPNP_E_SUCCESS || ret2 == UPNP_E_SUCCESS ||
745 // ret3 == UPNP_E_SUCCESS) {
746 [[maybe_unused]] int ret1 =
747 web_server_accept(miniSock->miniServerSock4, rdSet);
748 [[maybe_unused]] int ret2 =
749 web_server_accept(miniSock->miniServerSock6, rdSet);
750 [[maybe_unused]] int ret3 =
751 web_server_accept(miniSock->miniServerSock6UlaGua, rdSet);
752#ifdef COMPA_HAVE_CTRLPT_SSDP
753 ssdp_read(&miniSock->ssdpReqSock4, &rdSet);
754 ssdp_read(&miniSock->ssdpReqSock6, &rdSet);
755#endif
756 ssdp_read(&miniSock->ssdpSock4, &rdSet);
757 ssdp_read(&miniSock->ssdpSock6, &rdSet);
758 ssdp_read(&miniSock->ssdpSock6UlaGua, &rdSet);
759 // }
760
761 // Check if we have received a packet from
762 // localhost(127.0.0.1) that will stop the miniserver.

1048int StartMiniServer([[maybe_unused]] in_port_t* listen_port4,
1049 [[maybe_unused]] in_port_t* listen_port6,
1050 [[maybe_unused]] in_port_t* listen_port6UlaGua) {
1051 UPnPsdk_LOGINFO("MSG1068") "Executing...\n";
1052 constexpr int max_count{10000};
1053 MiniServerSockArray* miniSocket;
1054 ThreadPoolJob job;
1055 int ret_code{UPNP_E_INTERNAL_ERROR};
1056
1057 memset(&job, 0, sizeof(job));
1058
1059 if (gMServState != MSERV_IDLE) {
1060 /* miniserver running. */
1061 UPnPsdk_LOGERR("MSG1087") "Cannot start. Miniserver is running.\n";
1062 return UPNP_E_INTERNAL_ERROR;
1063 }
1064
1065 miniSocket = (MiniServerSockArray*)malloc(sizeof(MiniServerSockArray));
1066 if (!miniSocket) {
1067 return UPNP_E_OUTOF_MEMORY;
1068 }
1069 InitMiniServerSockArray(miniSocket);
1070
1071#ifdef COMPA_HAVE_WEBSERVER
1072 // These socket objects must be valid until the miniserver is successful
1073 // started. They will be moved to the running miniserver thread.
1074 UPnPsdk::CSocket sockLlaObj;
1075 miniSocket->pSockLlaObj = &sockLlaObj;
1076 UPnPsdk::CSocket sockGuaObj;
1077 miniSocket->pSockGuaObj = &sockGuaObj;
1078 UPnPsdk::CSocket sockIp4Obj;
1079 miniSocket->pSockIp4Obj = &sockIp4Obj;
1080
1081 /* V4 and V6 http listeners. */
1082 ret_code = get_miniserver_sockets(miniSocket, *listen_port4, *listen_port6,
1083 *listen_port6UlaGua);
1084 if (ret_code != UPNP_E_SUCCESS) {
1085 free(miniSocket);
1086 return ret_code;
1087 }
1088#endif
1089
1090 /* Stop socket (To end miniserver processing). */
1091 ret_code = get_miniserver_stopsock(miniSocket);
1092 if (ret_code != UPNP_E_SUCCESS) {
1093 sock_close(miniSocket->miniServerSock4);
1094 sock_close(miniSocket->miniServerSock6);
1095 sock_close(miniSocket->miniServerSock6UlaGua);
1096 free(miniSocket);
1097 return ret_code;
1098 }
1099#if defined(COMPA_HAVE_CTRLPT_SSDP) || defined(COMPA_HAVE_DEVICE_SSDP)
1100 /* SSDP socket for discovery/advertising. */
1101 ret_code = umock::pupnp_ssdp.get_ssdp_sockets(miniSocket);
1102 if (ret_code != UPNP_E_SUCCESS) {
1103 sock_close(miniSocket->miniServerSock4);
1104 sock_close(miniSocket->miniServerSock6);
1105 sock_close(miniSocket->miniServerSock6UlaGua);
1106 sock_close(miniSocket->miniServerStopSock);
1107 free(miniSocket);
1108 return ret_code;
1109 }
1110#endif
1111 // Run miniserver in a new thread.
1112 TPJobInit(&job, (start_routine)RunMiniServer_f, (void*)miniSocket);
1113 TPJobSetPriority(&job, MED_PRIORITY);
1114 TPJobSetFreeFunction(&job, (free_routine)free);
1115 ret_code = ThreadPoolAddPersistent(&gMiniServerThreadPool, &job, NULL);
1116 if (ret_code != 0) {
1117 sock_close(miniSocket->miniServerSock4);
1118 sock_close(miniSocket->miniServerSock6);
1119 sock_close(miniSocket->miniServerSock6UlaGua);
1120 sock_close(miniSocket->miniServerStopSock);
1121 sock_close(miniSocket->ssdpSock4);
1122 sock_close(miniSocket->ssdpSock6);
1123 sock_close(miniSocket->ssdpSock6UlaGua);
1124#ifdef COMPA_HAVE_CTRLPT_SSDP
1125 sock_close(miniSocket->ssdpReqSock4);
1126 sock_close(miniSocket->ssdpReqSock6);
1127#endif
1128 free(miniSocket);
1129 return UPNP_E_OUTOF_MEMORY;
1130 }
1131 /* Wait for miniserver to start. */
1132 int count{0};
1133 while (gMServState != (MiniServerState)MSERV_RUNNING && count < max_count) {
1134 /* 0.05s */
1135 imillisleep(50);
1136 count++;
1137 }
1138 if (count >= max_count) {
1139 /* Took it too long to start that thread. */
1140 sock_close(miniSocket->miniServerSock4);
1141 sock_close(miniSocket->miniServerSock6);
1142 sock_close(miniSocket->miniServerSock6UlaGua);
1143 sock_close(miniSocket->miniServerStopSock);
1144 sock_close(miniSocket->ssdpSock4);
1145 sock_close(miniSocket->ssdpSock6);
1146 sock_close(miniSocket->ssdpSock6UlaGua);
1147#ifdef COMPA_HAVE_CTRLPT_SSDP
1148 sock_close(miniSocket->ssdpReqSock4);
1149 sock_close(miniSocket->ssdpReqSock6);
1150#endif
1151 return UPNP_E_INTERNAL_ERROR;
1152 }
1153#ifdef COMPA_HAVE_WEBSERVER
1154 *listen_port4 = miniSocket->miniServerPort4;
1155 *listen_port6 = miniSocket->miniServerPort6;
1156 *listen_port6UlaGua = miniSocket->miniServerPort6UlaGua;
1157#endif
1158
1159 return UPNP_E_SUCCESS;
1160}
