17// Free function to check for a netaddress without port
18// ----------------------------------------------------
19// I simply use the system function %inet_pton() to check if the node string is
20// accepted.
21sa_family_t is_netaddr(const std::string& a_node,
22 const int a_addr_family) noexcept {
23 // clang-format off
24 TRACE("Executing is_netaddr(\"" + a_node + "\", " +
25 (a_addr_family == AF_INET6 ? "AF_INET6" :
26 (a_addr_family == AF_INET ? "AF_INET" : "AF_UNSPEC")) + ")")
27 // clang-format on
28
29 // The shortest numeric netaddress is "[::]".
30 if (a_node.size() < 4) // noexcept
31 return AF_UNSPEC;
32
33 unsigned char buf[sizeof(in6_addr)];
34 switch (a_addr_family) {
35 case AF_INET6: {
36 // front() and back() have undefined behavior with an empty string.
37 // Here its size() is at least 4.
38 if (a_node.front() != '[' || a_node.back() != ']')
39 return AF_UNSPEC;
40 // Remove surounding brackets for inet_pton().
41 // substr() throws exception std::out_of_range if pos > size(), but that
42 // cannot occur due to the guard above, size() is at least 4 here and
43 // pos is fix 1.
44 const std::string node{a_node.substr(1, a_node.size() - 2)};
45 return ::inet_pton(AF_INET6, node.c_str(), buf) == 1 ? AF_INET6
46 : AF_UNSPEC;
47 }
48 case AF_INET:
49 return ::inet_pton(AF_INET, a_node.c_str(), buf) == 1 ? AF_INET
50 : AF_UNSPEC;
51 case AF_UNSPEC:
52 // Recursive call
53 // clang-format off
54 return is_netaddr(a_node, AF_INET6) ? AF_INET6 : false ||
55 is_netaddr(a_node, AF_INET) ? AF_INET : AF_UNSPEC;
56 // clang-format on
57 }
58 return AF_UNSPEC;
59}

62// Free function to check if a string is a valid port number
63// ---------------------------------------------------------
64bool is_port(const std::string& a_port_str) noexcept {
65 TRACE("Executing is_port() with port=\"" + a_port_str + "\"")
66
67 // Only non empty strings
68 // and strings with max. 5 characters are valid (uint16_t has max. 65535)
69 if (a_port_str.empty() || a_port_str.length() > 5)
70 return false;
71
72 // Now we check if the string are all digit characters
73 for (char ch : a_port_str) {
74 if (!::isdigit(static_cast<unsigned char>(ch)))
75 return false;
76 }
77 // Valid positive number but is it within the port range (uint16_t)?
78 // stoi may throw std::invalid_argument if no conversion could be performed
79 // or std::out_of_range. But with the prechecked number string this should
80 // never be thrown.
81 return (std::stoi(a_port_str) <= 65535) ? true : false;
82}

95// Assignment operator to set a netaddress
96// ---------------------------------------
97void Netaddr::operator=(const std::string& a_addr_str) noexcept {
98 TRACE2(this, " Executing Netaddr=(), assign netaddress")
99
100 m_netaddrp.clear(); // noexcept
101
102 try {
103 if (is_netaddr(a_addr_str) != AF_UNSPEC) { // noexcept
104 // Here we have a valid netaddress without port.
105 m_netaddrp = a_addr_str + ":0"; // maybe throw std::length_error?
106 return;
107 }
108 // Following a valid netaddress must have a port.
109
110 // Look for a port part or an ambiguous netaddress.
111 size_t pos = a_addr_str.find_last_of(':'); // noexcept
112 if (pos == std::string::npos || a_addr_str == "[::1") {
113 // No port detected. The address string cannot be a netaddress.
114 return;
115 }
116
117 // Split address string into possible netaddress and port and check
118 // them. substr() throws exception std::out_of_range if pos > size().
119 // pos is 0-based, size() is 1-based so with ":" is size()==1, pos==0,
120 // pos+1==size() does not throw.
121 std::string netaddr = a_addr_str.substr(0, pos);
122 std::string port = a_addr_str.substr(pos + 1);
123 if (is_netaddr(netaddr) == AF_UNSPEC) {
124 // Unspecified netaddress
125 netaddr.clear();
126 }
127 if (!is_port(port)) { // noexcept
128 // Unspecified port
129 port = "0";
130 }
131
132 // Set netaddress with port.
133 if (!(netaddr.empty() && port == "0")) { // noexcept
134 m_netaddrp = netaddr + ":" + port; // maybe throw std::length_error?
135 }
136 return;
137
138 } catch (const std::length_error&) {
139 // If a resulting string is too long then I just return with an
140 // unspecified netaddress set.
141 return;
142 }
143}
