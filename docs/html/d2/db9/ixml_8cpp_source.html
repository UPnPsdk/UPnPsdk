51static void copy_with_escape(
53 ixml_membuf* buf,
55 const char* p) {
56 size_t i;
57 size_t plen;
58
59 if (p == NULL)
60 return;
61 plen = strlen(p);
62 for (i = (size_t)0; i < plen; ++i) {
63 switch (p[i]) {
64 case '<':
65 ixml_membuf_append_str(buf, "&lt;");
66 break;
67 case '>':
68 ixml_membuf_append_str(buf, "&gt;");
69 break;
70 case '&':
71 ixml_membuf_append_str(buf, "&amp;");
72 break;
73 case '\'':
74 ixml_membuf_append_str(buf, "&apos;");
75 break;
76 case '\"':
77 ixml_membuf_append_str(buf, "&quot;");
78 break;
79 default:
80 ixml_membuf_append(buf, &p[i]);
81 break;
82 }
83 }
84}

90static void ixmlPrintDomTreeRecursive(
92 IXML_Node* nodeptr,
94 ixml_membuf* buf) {
95 const char* nodeName = NULL;
96 const char* nodeValue = NULL;
97 IXML_Node *child = NULL, *sibling = NULL;
98
99 if (nodeptr != NULL) {
100 nodeName = (const char*)ixmlNode_getNodeName(nodeptr);
101 nodeValue = ixmlNode_getNodeValue(nodeptr);
102
103 switch (ixmlNode_getNodeType(nodeptr)) {
104 case eTEXT_NODE:
105 copy_with_escape(buf, nodeValue);
106 break;
107
108 case eCDATA_SECTION_NODE:
109 ixml_membuf_append_str(buf, "<![CDATA[");
110 ixml_membuf_append_str(buf, nodeValue);
111 ixml_membuf_append_str(buf, "]]>");
112 break;
113
114 case ePROCESSING_INSTRUCTION_NODE:
115 ixml_membuf_append_str(buf, "<?");
116 ixml_membuf_append_str(buf, nodeName);
117 ixml_membuf_append_str(buf, " ");
118 copy_with_escape(buf, nodeValue);
119 ixml_membuf_append_str(buf, "?>\n");
120 break;
121
122 case eDOCUMENT_NODE:
123 ixmlPrintDomTreeRecursive(ixmlNode_getFirstChild(nodeptr), buf);
124 break;
125
126 case eATTRIBUTE_NODE:
127 ixml_membuf_append_str(buf, nodeName);
128 ixml_membuf_append_str(buf, "=\"");
129 copy_with_escape(buf, nodeValue);
130 ixml_membuf_append_str(buf, "\"");
131 if (nodeptr->nextSibling != NULL) {
132 ixml_membuf_append_str(buf, " ");
133 ixmlPrintDomTreeRecursive(nodeptr->nextSibling, buf);
134 }
135 break;
136
137 case eELEMENT_NODE:
138 ixml_membuf_append_str(buf, "<");
139 ixml_membuf_append_str(buf, nodeName);
140 if (nodeptr->firstAttr != NULL) {
141 ixml_membuf_append_str(buf, " ");
142 ixmlPrintDomTreeRecursive(nodeptr->firstAttr, buf);
143 }
144 child = ixmlNode_getFirstChild(nodeptr);
145 if (child != NULL && ixmlNode_getNodeType(child) == eELEMENT_NODE) {
146 ixml_membuf_append_str(buf, ">\r\n");
147 } else {
148 ixml_membuf_append_str(buf, ">");
149 }
150 /* output the children */
151 ixmlPrintDomTreeRecursive(ixmlNode_getFirstChild(nodeptr), buf);
152
153 /* Done with children. Output the end tag. */
154 ixml_membuf_append_str(buf, "</");
155 ixml_membuf_append_str(buf, nodeName);
156
157 sibling = ixmlNode_getNextSibling(nodeptr);
158 if (sibling != NULL &&
159 ixmlNode_getNodeType(sibling) == eTEXT_NODE) {
160 ixml_membuf_append_str(buf, ">");
161 } else {
162 ixml_membuf_append_str(buf, ">\r\n");
163 }
164 ixmlPrintDomTreeRecursive(ixmlNode_getNextSibling(nodeptr), buf);
165 break;
166
167 default:
168 IxmlPrintf(__FILE__, __LINE__, "ixmlPrintDomTreeRecursive",
169 "Warning, unknown node type %d\n",
170 (int)ixmlNode_getNodeType(nodeptr));
171 break;
172 }
173 }
