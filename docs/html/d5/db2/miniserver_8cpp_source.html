206int dispatch_request(
208 SOCKINFO* info,
210 http_parser_t* hparser) {
211 memptr header;
212 size_t min_size;
213 http_message_t* request;
214 MiniServerCallback callback;
215 WebCallback_HostValidate host_validate_callback = 0;
216 void* cookie{};
217 int rc = UPNP_E_SUCCESS;
218 /* If it does not fit in here, it is likely invalid anyway. */
219 char host_port[NAME_SIZE];
220
221 switch (hparser->msg.method) {
222 /* Soap Call */
223 case SOAPMETHOD_POST:
224 case HTTPMETHOD_MPOST:
225 callback = gSoapCallback;
226 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
227 "miniserver %d: got SOAP msg\n", info->socket);
228 break;
229 /* Gena Call */
230 case HTTPMETHOD_NOTIFY:
231 case HTTPMETHOD_SUBSCRIBE:
232 case HTTPMETHOD_UNSUBSCRIBE:
233 callback = gGenaCallback;
234 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
235 "miniserver %d: got GENA msg\n", info->socket);
236 break;
237 /* HTTP server call */
238 case HTTPMETHOD_GET:
239 case HTTPMETHOD_POST:
240 case HTTPMETHOD_HEAD:
241 case HTTPMETHOD_SIMPLEGET:
242 callback = gGetCallback;
243 host_validate_callback = gWebCallback_HostValidate;
244 cookie = gWebCallback_HostValidateCookie;
245 UPNPLIB_LOGINFO "MSG1107: miniserver socket="
246 << info->socket << ": got WEB server msg.\n";
247 break;
248 default:
249 callback = 0;
250 }
251 if (!callback) {
252 rc = HTTP_INTERNAL_SERVER_ERROR;
253 goto ExitFunction;
254 }
255 request = &hparser->msg;
256 if (upnplib::g_dbug) {
257 getNumericHostRedirection(info->socket, host_port, sizeof host_port);
258 UPNPLIB_LOGINFO "MSG1113: Redirect host_port=\"" << host_port << "\"\n";
259 }
260 /* check HOST header for an IP number -- prevents DNS rebinding. */
261 if (!httpmsg_find_hdr(request, HDR_HOST, &header)) {
262 rc = UPNP_E_BAD_HTTPMSG;
263 goto ExitFunction;
264 }
265 min_size = header.length < ((sizeof host_port) - 1)
266 ? header.length
267 : (sizeof host_port) - 1;
268 memcpy(host_port, header.buf, min_size);
269 host_port[min_size] = 0;
270 if (host_validate_callback) {
271 rc = host_validate_callback(host_port, cookie);
272 if (rc == UPNP_E_BAD_HTTPMSG) {
273 goto ExitFunction;
274 }
275 } else if (!host_header_is_numeric(host_port, min_size)) {
276 if (!gAllowLiteralHostRedirection) {
277 rc = UPNP_E_BAD_HTTPMSG;
278 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
279 "Possible DNS Rebind attack prevented.\n");
280 goto ExitFunction;
281 } else {
282 membuffer redir_buf;
283 static const char* redir_fmt = "HTTP/1.1 307 Temporary Redirect\r\n"
284 "Location: http://%s\r\n\r\n";
285 char redir_str[NAME_SIZE];
286 int timeout = HTTP_DEFAULT_TIMEOUT;
287
288 getNumericHostRedirection(info->socket, host_port,
289 sizeof host_port);
290 membuffer_init(&redir_buf);
291 snprintf(redir_str, NAME_SIZE, redir_fmt, host_port);
292 membuffer_append_str(&redir_buf, redir_str);
293 rc = http_SendMessage(info, &timeout, "b", redir_buf.buf,
294 redir_buf.length);
295 membuffer_destroy(&redir_buf);
296 goto ExitFunction;
297 }
298 }
299 callback(hparser, request, info);
300
301ExitFunction:
302 return rc;
303}

323void handle_request(
325 void* args) { // Expected to be mserv_request_t*
326 SOCKINFO info;
327 int http_error_code;
328 int ret_code;
329 int major = 1;
330 int minor = 1;
331 http_parser_t parser;
332 http_message_t* hmsg = NULL;
333 int timeout = HTTP_DEFAULT_TIMEOUT;
334 mserv_request_t* request_in = (mserv_request_t*)args;
335 SOCKET connfd = request_in->connfd;
336
337 UPNPLIB_LOGINFO "MSG1027: Miniserver socket "
338 << connfd << ": READING request from client...\n";
339 /* parser_request_init( &parser ); */ /* LEAK_FIX_MK */
340 hmsg = &parser.msg;
341 ret_code = sock_init_with_ip(&info, connfd,
342 (sockaddr*)&request_in->foreign_sockaddr);
343 if (ret_code != UPNP_E_SUCCESS) {
344 free(request_in);
345 httpmsg_destroy(hmsg);
346 return;
347 }
348
349 /* read */
350 ret_code = http_RecvMessage(&info, &parser, HTTPMETHOD_UNKNOWN, &timeout,
351 &http_error_code);
352 if (ret_code != 0) {
353 goto error_handler;
354 }
355
356 UPNPLIB_LOGINFO "MSG1106: miniserver socket=" << connfd
357 << ": PROCESSING...\n";
358 /* dispatch */
359 http_error_code = dispatch_request(&info, &parser);
360 if (http_error_code != 0) {
361 goto error_handler;
362 }
363 http_error_code = 0;
364
365error_handler:
366 if (http_error_code > 0) {
367 if (hmsg) {
368 major = hmsg->major_version;
369 minor = hmsg->minor_version;
370 }
371 // BUG! Don't try to send a status response to a remote client with
372 // http error (e.g. 400) if we have a socket error. It doesn't make
373 // sense. --Ingo
374 http_SendStatusResponse(&info, http_error_code, major, minor);
375 }
376 sock_destroy(&info, SD_BOTH);
377 httpmsg_destroy(hmsg);
378 free(request_in);
379
380 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
381 "miniserver %d: COMPLETE\n", connfd);
382}

390 SOCKET connfd,
392 sockaddr* clientAddr) {
393 TRACE("Executing schedule_request_job()")
394 UPNPLIB_LOGINFO "MSG1042: Schedule request job to host "
395 << upnplib::to_netaddrp(
396 reinterpret_cast<const sockaddr_storage*>(clientAddr))
397 << " with socket " << connfd << ".\n";
398
399 ThreadPoolJob job{};
400 mserv_request_t* request{
401 static_cast<mserv_request_t*>(std::malloc(sizeof(mserv_request_t)))};
402
403 if (request == nullptr) {
404 UPNPLIB_LOGCRIT "MSG1024: Socket " << connfd << ": out of memory.\n";
405 sock_close(connfd);
406 return;
407 }
408

410 memcpy(&request->foreign_sockaddr, clientAddr,
411 sizeof(request->foreign_sockaddr));
412 TPJobInit(&job, (start_routine)handle_request, request);
413 TPJobSetFreeFunction(&job, free_handle_request_arg);
414 TPJobSetPriority(&job, MED_PRIORITY);
415 if (ThreadPoolAdd(&gMiniServerThreadPool, &job, NULL) != 0) {
416 UPNPLIB_LOGERR "MSG1025: Socket " << connfd
417 << ": cannot schedule request.\n";
418 free(request);
419 sock_close(connfd);
420 return;
421 }

442void fdset_if_valid( //
443 SOCKET a_sock,
444 fd_set* a_set
447) {
448 UPNPLIB_LOGINFO "MSG1086: Check sockfd=" << a_sock << ".\n";
449 if (a_sock == INVALID_SOCKET)
450 // This is a defined state and we return silently.
451 return;
452
453 if (a_sock < 3 || a_sock >= FD_SETSIZE) {
454 UPNPLIB_LOGERR "MSG1005: "
455 << (a_sock < 0 ? "Invalid" : "Prohibited") << " socket " << a_sock
456 << " not set to be monitored by ::select()"
457 << (a_sock >= 3 ? " because it violates FD_SETSIZE.\n" : ".\n");
458 return;
459 }
460 // Check if socket is valid and bound
461 try {
462 upnplib::CSocket_basic sockObj(a_sock);
463 sockObj.init();
464 if (sockObj.is_bound())
465
466 FD_SET(a_sock, a_set);
467
468 else
469 UPNPLIB_LOGINFO "MSG1002: Unbound socket "
470 << a_sock << " not set to be monitored by ::select().\n";
471
472 } catch (const std::exception& e) {
473 if (upnplib::g_dbug)
474 std::clog << e.what();
475 UPNPLIB_LOGCATCH "MSG1009: Invalid socket "
476 << a_sock << " not set to be monitored by ::select().\n";
477 }
478}

491int web_server_accept(
493 [[maybe_unused]] SOCKET lsock,
495 [[maybe_unused]] fd_set& set) {
496#ifndef COMPA_HAVE_WEBSERVER
497 return UPNP_E_NO_WEB_SERVER;
498#else
499 TRACE("Executing web_server_accept()")
500 if (lsock == INVALID_SOCKET || !FD_ISSET(lsock, &set)) {
501 UPNPLIB_LOGINFO "MSG1012: Socket("
502 << lsock << ") invalid or not in file descriptor set.\n";
503 return UPNP_E_SOCKET_ERROR;
504 }
505
506 SOCKET asock;
507 upnplib::sockaddr_t clientAddr;
508 socklen_t clientLen = sizeof(clientAddr);
509
510 // accept a network request connection
511 asock = umock::sys_socket_h.accept(lsock, &clientAddr.sa, &clientLen);
512 if (asock == INVALID_SOCKET) {
513 UPNPLIB_LOGERR "MSG1022: Error in ::accept(): " << std::strerror(errno)
514 << ".\n";
515 return UPNP_E_SOCKET_ACCEPT;
516 }
517
518 // Schedule a job to manage a UPnP request from a remote host.
519 char buf_ntop[INET6_ADDRSTRLEN + 7];
520 inet_ntop(AF_INET, &clientAddr.sin.sin_addr, buf_ntop, sizeof(buf_ntop));
521 UPNPLIB_LOGINFO "MSG1023: Connected to host "
522 << buf_ntop << ":" << ntohs(clientAddr.sin.sin_port) << " with socket "
523 << asock << ".\n";
524 schedule_request_job(asock, &clientAddr.sa);
525
526 return UPNP_E_SUCCESS;
527#endif /* COMPA_HAVE_WEBSERVER */
528}

569int receive_from_stopSock(
570 SOCKET ssock,
571 fd_set* set
573) {
574 TRACE("Executing receive_from_stopSock()")
575 constexpr char shutdown_str[]{"ShutDown"};
576
577 if (!FD_ISSET(ssock, set))
578 return 0; // Nothing to do for this socket
579
580 upnplib::sockaddr_t clientAddr{};
581 socklen_t clientLen{sizeof(clientAddr)}; // May be modified
582
583 // The receive buffer is one byte greater with '\0' than the max receiving
584 // bytes so the received message will always be terminated.
585 char receiveBuf[sizeof(shutdown_str) + 1]{};
586 char buf_ntop[INET6_ADDRSTRLEN];
587
588 // receive from
589 SSIZEP_T byteReceived = umock::sys_socket_h.recvfrom(
590 ssock, receiveBuf, sizeof(shutdown_str), 0, &clientAddr.sa, &clientLen);
591 if (byteReceived == SOCKET_ERROR ||
592 inet_ntop(AF_INET, &clientAddr.sin.sin_addr, buf_ntop,
593 sizeof(buf_ntop)) == nullptr) {
594 UPNPLIB_LOGCRIT "MSG1038: Failed to receive data from socket "
595 << ssock << ". Stop miniserver.\n";
596 return 1;
597 }
598
599 // 16777343 are netorder bytes of "127.0.0.1"
600 if (clientAddr.sin.sin_addr.s_addr != 16777343 ||
601 strcmp(receiveBuf, shutdown_str) != 0) //
602 {
603 char nullstr[]{"\\0"};
604 if (byteReceived == 0 || receiveBuf[byteReceived - 1] != '\0')
605 nullstr[0] = '\0';
606 UPNPLIB_LOGERR "MSG1039: Received \""
607 << receiveBuf << nullstr << "\" from " << buf_ntop << ":"
608 << ntohs(clientAddr.sin.sin_port)
609 << ", must be \"ShutDown\\0\" from 127.0.0.1:*. Don't "
610 "stopping miniserver.\n";
611 return 0;
612 }
613
614 UPNPLIB_LOGINFO "MSG1040: On socket "
615 << ssock << " received ordinary datagram \"" << receiveBuf
616 << "\\0\" from " << buf_ntop << ":" << ntohs(clientAddr.sin.sin_port)
617 << ". Stop miniserver.\n";
618 return 1;
619}

632void RunMiniServer(
636 MiniServerSockArray* miniSock) {
637 UPNPLIB_LOGINFO "MSG1085: Executing...\n";
638 fd_set expSet;
639 fd_set rdSet;
640 int stopSock = 0;
641
642 // On MS Windows INVALID_SOCKET is unsigned -1 = 18446744073709551615 so we
643 // get maxMiniSock with this big number even if there is only one
644 // INVALID_SOCKET. Incrementing it at the end results in 0. To be portable
645 // we must not assume INVALID_SOCKET to be -1. --Ingo
646 SOCKET maxMiniSock = 0;
647 maxMiniSock = //
648 std::max(maxMiniSock, miniSock->miniServerSock4 == INVALID_SOCKET
649 ? 0
650 : miniSock->miniServerSock4);
651 maxMiniSock = //
652 std::max(maxMiniSock, miniSock->miniServerSock6 == INVALID_SOCKET
653 ? 0
654 : miniSock->miniServerSock6);
655 maxMiniSock = //
656 std::max(maxMiniSock, miniSock->miniServerSock6UlaGua == INVALID_SOCKET
657 ? 0
658 : miniSock->miniServerSock6UlaGua);
659 maxMiniSock = //
660 std::max(maxMiniSock, miniSock->miniServerStopSock == INVALID_SOCKET
661 ? 0
662 : miniSock->miniServerStopSock);
663 maxMiniSock = //
664 std::max(maxMiniSock, miniSock->ssdpSock4 == INVALID_SOCKET
665 ? 0
666 : miniSock->ssdpSock4);
667 maxMiniSock = //
668 std::max(maxMiniSock, miniSock->ssdpSock6 == INVALID_SOCKET
669 ? 0
670 : miniSock->ssdpSock6);
671 maxMiniSock = //
672 std::max(maxMiniSock, miniSock->ssdpSock6UlaGua == INVALID_SOCKET
673 ? 0
674 : miniSock->ssdpSock6UlaGua);
675#ifdef COMPA_HAVE_CTRLPT_SSDP
676 maxMiniSock = //
677 std::max(maxMiniSock, miniSock->ssdpReqSock4 == INVALID_SOCKET
678 ? 0
679 : miniSock->ssdpReqSock4);
680 maxMiniSock = //
681 std::max(maxMiniSock, miniSock->ssdpReqSock6 == INVALID_SOCKET
682 ? 0
683 : miniSock->ssdpReqSock6);
684#endif
685 ++maxMiniSock;
686
687 gMServState = MSERV_RUNNING;
688 while (!stopSock) {
689 FD_ZERO(&rdSet);
690 FD_ZERO(&expSet);
691 /* FD_SET()'s */
692 FD_SET(miniSock->miniServerStopSock, &expSet);
693 FD_SET(miniSock->miniServerStopSock, &rdSet);
694 fdset_if_valid(miniSock->miniServerSock4, &rdSet);
695 fdset_if_valid(miniSock->miniServerSock6, &rdSet);
696 fdset_if_valid(miniSock->miniServerSock6UlaGua, &rdSet);
697 fdset_if_valid(miniSock->ssdpSock4, &rdSet);
698 fdset_if_valid(miniSock->ssdpSock6, &rdSet);
699 fdset_if_valid(miniSock->ssdpSock6UlaGua, &rdSet);
700#ifdef COMPA_HAVE_CTRLPT_SSDP
701 fdset_if_valid(miniSock->ssdpReqSock4, &rdSet);
702 fdset_if_valid(miniSock->ssdpReqSock6, &rdSet);
703#endif
704
705 /* select() */
706 int ret = umock::sys_socket_h.select(static_cast<int>(maxMiniSock),
707 &rdSet, NULL, &expSet, NULL);
708
709 if (ret == SOCKET_ERROR) {
710 if (errno == EINTR) {
711 // A signal was caught, not for us. We ignore it and
712 continue;
713 }
714 if (errno == EBADF) {
715 // A closed socket file descriptor was given in one of the
716 // sets. For details look at
717 // REF:_[Should_I_assert_fail_on_select()_EBADF?](https://stackoverflow.com/q/28015859/5014688)
718 // It is difficult to determine here what file descriptor
719 // in rdSet or expSet is invalid. So I ensure that only valid
720 // socket fds are given by checking them with fdset_if_valid()
721 // before calling select(). Doing this I have to
722 continue;
723 }
724 // All other errors EINVAL and ENOMEM are critical and cannot
725 // continue run mininserver.
726 UPNPLIB_LOGCRIT "MSG1021: Error in ::select(): "
727 << std::strerror(errno) << ".\n";
728 break;
729 }
730
731 // Accept requested connection from a remote control point and run the
732 // connection in a new thread. Due to side effects with threading we
733 // need to avoid lazy evaluation with chained || because all
734 // web_server_accept() must be called.
735 // if (ret1 == UPNP_E_SUCCESS || ret2 == UPNP_E_SUCCESS ||
736 // ret3 == UPNP_E_SUCCESS) {
737 [[maybe_unused]] int ret1 =
738 web_server_accept(miniSock->miniServerSock4, rdSet);
739 [[maybe_unused]] int ret2 =
740 web_server_accept(miniSock->miniServerSock6, rdSet);
741 [[maybe_unused]] int ret3 =
742 web_server_accept(miniSock->miniServerSock6UlaGua, rdSet);
743#ifdef COMPA_HAVE_CTRLPT_SSDP
744 ssdp_read(&miniSock->ssdpReqSock4, &rdSet);
745 ssdp_read(&miniSock->ssdpReqSock6, &rdSet);
746#endif
747 ssdp_read(&miniSock->ssdpSock4, &rdSet);
748 ssdp_read(&miniSock->ssdpSock6, &rdSet);
749 ssdp_read(&miniSock->ssdpSock6UlaGua, &rdSet);
750 // }
751
752 // Check if we have received a packet from
753 // localhost(127.0.0.1) that will stop the miniserver.
754 stopSock = receive_from_stopSock(miniSock->miniServerStopSock, &rdSet);
755 } // while (!stopsock)
756
757 /* Close all sockets. */
758 sock_close(miniSock->miniServerSock4);
759 sock_close(miniSock->miniServerSock6);
760 sock_close(miniSock->miniServerSock6UlaGua);
761 sock_close(miniSock->miniServerStopSock);
762 sock_close(miniSock->ssdpSock4);
763 sock_close(miniSock->ssdpSock6);
764 sock_close(miniSock->ssdpSock6UlaGua);
765#ifdef COMPA_HAVE_CTRLPT_SSDP
766 sock_close(miniSock->ssdpReqSock4);
767 sock_close(miniSock->ssdpReqSock6);
768#endif
769 /* Free minisock. */
770 umock::stdlib_h.free(miniSock);
771 gMServState = MSERV_IDLE;
772
773 return;
774}

788int get_port(
790 SOCKET sockfd,
792 uint16_t* port) {
793 TRACE("Executing get_port(), calls system ::getsockname()")
794 upnplib::sockaddr_t sockinfo{};
795 socklen_t len(sizeof sockinfo); // May be modified by getsockname()
796
797 if (umock::sys_socket_h.getsockname(sockfd, &sockinfo.sa, &len) == -1)
798 // system error (errno etc.) is expected to be unmodified on return.
799 return -1;
800
