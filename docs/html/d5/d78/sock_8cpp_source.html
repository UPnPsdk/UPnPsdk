81int sock_read_unprotected(
83 const SOCKINFO* a_info,
85 char* a_readbuf,
87 [[maybe_unused]] const char* a_writebuf,
89 const size_t a_bufsize,
92 int* a_timeoutSecs) {
93 time_t start_time{time(NULL)};
94 TRACE("Executing sock_read_unprotected()")
95
96 // Also restrict a_bufsize to integer for save later use despite type cast.
97 if (a_info == nullptr || a_readbuf == nullptr || a_bufsize > INT_MAX)
98 return UPNP_E_SOCKET_ERROR;
99 if (a_bufsize == 0)
100 return 0;
101
102 SOCKET sockfd{a_info->socket};
103
104 ::fd_set readSet;
105 ::fd_set writeSet;
106 FD_ZERO(&readSet);
107 FD_ZERO(&writeSet);
108 FD_SET(sockfd, &readSet);
109
110 // a_timeoutSecs == nullptr means default timeout to use.
111 int timeout_secs = (a_timeoutSecs == nullptr) ? upnplib::g_response_timeout
112 : *a_timeoutSecs;
113
114 upnplib::CSocketErr sockerrObj;
115 while (true) {
116 // Due to 'man select' timeout should be considered to be undefined
117 // after select() returns so we must set it on every loop.
118 ::timeval timeout;
119 timeout.tv_sec = timeout_secs;
120 timeout.tv_usec = 0;
121
122 // select() monitors only one socket file descriptor.
123 int retCode = umock::sys_socket_h.select(
124 static_cast<int>(sockfd + 1), &readSet, &writeSet, nullptr,
125 (timeout_secs < 0) ? nullptr : &timeout);
126
127 if (retCode == 0)
128 return UPNP_E_TIMEDOUT;
129 if (retCode == SOCKET_ERROR) {
130 sockerrObj.catch_error();
131 if (sockerrObj == EINTRP)
132 // Signal catched by select(). It is not for us so we
133 continue;
134 return UPNP_E_SOCKET_ERROR;
135 } else
136 /* read */
137 break;
138 }
139
140 TRACE("Read data with syscall ::recv().")
141 SSIZEP_T numBytes = umock::sys_socket_h.recv(
142 sockfd, a_readbuf, static_cast<SIZEP_T>(a_bufsize), 0);
143
144 // Also protect type cast
145 if (numBytes < 0 || numBytes > INT_MAX)
146 return UPNP_E_SOCKET_ERROR;
147
148 /* subtract time used for reading/writing. */
149 if (a_timeoutSecs != nullptr && timeout_secs != 0)
150 *a_timeoutSecs -= static_cast<int>(time(NULL) - start_time);
151
152 return static_cast<int>(numBytes);
153}

166int sock_write_unprotected(
168 const SOCKINFO* a_info,
170 [[maybe_unused]] char* a_readbuf,
172 const char* a_writebuf,
174 const size_t a_bufsize,
177 int* a_timeoutSecs) {
178 time_t start_time{time(NULL)};

180
181 // Also restrict a_bufsize to integer for save later use despite type cast.
182 if (a_info == nullptr || a_writebuf == nullptr || a_bufsize > INT_MAX)
183 return UPNP_E_SOCKET_ERROR;
184 if (a_bufsize == 0)
185 return 0;
186
187 SOCKET sockfd{a_info->socket};
188
189 ::fd_set readSet;
190 ::fd_set writeSet;
191 FD_ZERO(&readSet);
192 FD_ZERO(&writeSet);
193 FD_SET(sockfd, &writeSet);
194
195 // a_timeoutSecs == nullptr means default timeout to use.
196 int timeout_secs = (a_timeoutSecs == nullptr) ? upnplib::g_response_timeout
197 : *a_timeoutSecs;
198
199 upnplib::CSocketErr sockerrObj;
200 while (true) {
201 // Due to 'man select' timeout should be considered to be undefined
202 // after select() returns so we must set it on every loop.
203 ::timeval timeout;
204 timeout.tv_sec = timeout_secs;
205 timeout.tv_usec = 0;
206
207 // select monitors only one socket file descriptor.
208 int retCode = umock::sys_socket_h.select(
209 static_cast<int>(sockfd + 1), &readSet, &writeSet, nullptr,
210 (timeout_secs < 0) ? nullptr : &timeout);
211
212 if (retCode == 0)
213 return UPNP_E_TIMEDOUT;
214 if (retCode == SOCKET_ERROR) {
215 sockerrObj.catch_error();
216 if (sockerrObj == EINTRP)
217 // Signal catched by select(). It is not for us so we
218 continue;
219 return UPNP_E_SOCKET_ERROR;
220 } else
221 /* write. */
222 break;
223 }
224
225 // a_bufsize is restricted from 0 to INT_MAX.
226 ssize_t byte_left{static_cast<ssize_t>(a_bufsize)};
227 ssize_t bytes_sent{};
228
229 TRACE("Write data with syscall ::send().")
230 UPNPLIB_SCOPED_NO_SIGPIPE
231 while (byte_left != 0) {
232 ssize_t num_written = umock::sys_socket_h.send(
233 sockfd, a_writebuf + bytes_sent, static_cast<SIZEP_T>(byte_left),
234 MSG_DONTROUTE);
235 if (num_written == -1 || num_written > INT_MAX) {
236 return UPNP_E_SOCKET_WRITE;
237 }
238 byte_left -= num_written;
239 bytes_sent += num_written;
240 }
241
242 // Also protect type cast
243 if (bytes_sent < 0 || bytes_sent > INT_MAX)
244 return UPNP_E_SOCKET_ERROR;
245
246 /* subtract time used for writing. */
247 if (a_timeoutSecs != nullptr && timeout_secs != 0)
248 *a_timeoutSecs -= static_cast<int>(time(NULL) - start_time);
249
250 return static_cast<int>(bytes_sent);
251}

491int sock_destroy(SOCKINFO* info, int ShutdownMethod) {
492 TRACE("Executing sock_destroy()")
493 int ret{UPNP_E_SUCCESS};
494
495 if (info->socket != INVALID_SOCKET) {
496#ifdef UPNP_ENABLE_OPEN_SSL
497 if (info->ssl) {
498 SSL_shutdown(info->ssl);
499 SSL_free(info->ssl);
500 info->ssl = NULL;
501 }
502#endif
503 upnplib::CSocketErr sockerrObj;
504 if (umock::sys_socket_h.shutdown(info->socket, ShutdownMethod) ==
505 SOCKET_ERROR) {
506 sockerrObj.catch_error();
507 std::string msg = "MSG1010: syscall ::shutdown() returned \"" +
508 sockerrObj.get_error_str() + "\".\n";
509 if (sockerrObj == ENOTCONNP) {
510 // shutdown a not connected connection is not an error.
511 UPNPLIB_LOGINFO << msg;
512 } else {
513 UPNPLIB_LOGERR << msg;
514 ret = UPNP_E_SOCKET_ERROR;
515 }
516 }
517 if (sock_close(info->socket) != 0) {
518 ret = UPNP_E_SOCKET_ERROR;
519 }
520 info->socket = INVALID_SOCKET;
521 }
522
523 return ret;
524}

558int sock_make_blocking(SOCKET sock) {
559// returns 0 if successful, else SOCKET_ERROR.
560#ifdef _WIN32
561 u_long val = 0;
562 // returns 0 if successful, else SOCKET_ERROR.
563 return ioctlsocket(sock, FIONBIO, &val);
564#else
565 int val = fcntl(sock, F_GETFL, 0);
566 if (fcntl(sock, F_SETFL, val & ~O_NONBLOCK) == -1) {
567 return SOCKET_ERROR;
568 }
569 return 0;
570#endif
571}
