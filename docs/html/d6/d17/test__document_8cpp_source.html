40static const char* get_ixml_error_string(IXML_ERRORCODE code) {
41#define CASE(CODE) \
42 case IXML_##CODE: \
43 return #CODE

47 CASE(DOMSTRING_SIZE_ERR);

74int main(int argc, char* argv[]) {
75 int i;
76
77 if (argc < 2) {
78 fprintf(stderr, "Usage: %s [xml files to load]\n", argv[0]);
79 exit(EXIT_FAILURE);
80 }
81
82 for (i = 1; i < argc; i++) {
83 int rc;
84 IXML_Document* doc = NULL;
85 DOMString s;
86 char* p;
87
88 printf("Test \"%s\" \n", argv[i]);
89 printf(" Loading ... ");
90 fflush(stdout);
91
92 rc = ixmlLoadDocumentEx(argv[i], &doc);
93 if (rc != IXML_SUCCESS) {
94 fprintf(stderr,
95 "** error : can't load document %s : "
96 "error %d (%s)\n",
97 argv[i], rc, get_ixml_error_string((IXML_ERRORCODE)rc));
98 exit(EXIT_FAILURE);
99 }
100
101 printf("OK\n");
102
103 printf(" Printing ... ");
104 fflush(stdout);
105
106 s = ixmlPrintDocument(doc);
107 if (s == NULL || s[0] == '\0') {
108 fprintf(stderr, "** error : can't print loaded document %s\n",
109 argv[i]);
110 exit(EXIT_FAILURE);
111 }
112 p = s + strlen(s) - 1;
113 while (isspace(*p) && p > s)
114 p--;
115 if (*s != '<' || *p != '>') {
116 fprintf(stderr,
117 "** error : malformed printed document '%s' :"
118 "%s\n",
119 argv[i], s);
120 exit(EXIT_FAILURE);
121 }
122
123 printf("OK\n");
124
125 ixmlFreeDOMString(s);
126 ixmlDocument_free(doc);
127 }
128
129 exit(EXIT_SUCCESS);
130}
