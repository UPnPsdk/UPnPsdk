10// Class to initialize and shutdown ThreadPools
11// --------------------------------------------
12CThreadPoolInit::CThreadPoolInit(ThreadPool& a_threadpool,
13 const bool a_shutdown,
14 const int a_maxJobsTotal)
15 : m_threadpool(a_threadpool) //
16{
17 TRACE2(this, " Construct CThreadPoolInit()")
18 // Initialize the given Threadpool. nullptr means to use default
19 // attributes.
20 int ret = ThreadPoolInit(&m_threadpool, nullptr);
21 if (ret != 0)
22 throw std::runtime_error(
23 UPNPLIB_LOGEXCEPT
24 "MSG1045: Initializing ThreadPool fails with return number " +
25 std::to_string(ret));
26
27 if (a_shutdown) {
28 TRACE("constructor CThreadPoolInit(): set shutdown flag.")
29 // Prevent to add jobs without error message if set, I test jobs
30 // isolated.
31 m_threadpool.shutdown = 1;
32 } else {
33 // or allow number of threads. Use 0 with check of error message
34 // to stderr to test if threads are used.
35 ret = TPAttrSetMaxJobsTotal(&m_threadpool.attr, a_maxJobsTotal);
36 if (ret != 0)
37 throw std::runtime_error(UPNPLIB_LOGEXCEPT
38 "MSG1046: Setting maximal jobs number "
39 "fails with return number " +
40 std::to_string(ret));
41 }
42}

45 TRACE2(this, " Destruct CThreadPoolInit()")
