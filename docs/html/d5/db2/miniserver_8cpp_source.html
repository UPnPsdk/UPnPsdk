175int dispatch_request(
177 SOCKINFO* info,
179 http_parser_t* hparser) {
180 memptr header;
181 size_t min_size;
182 http_message_t* request;
183 MiniServerCallback callback;
184 WebCallback_HostValidate host_validate_callback = 0;
185 void* cookie{};
186 int rc = UPNP_E_SUCCESS;
187 /* If it does not fit in here, it is likely invalid anyway. */
188 char host_port[NAME_SIZE];
189
190 switch (hparser->msg.method) {
191 /* Soap Call */
192 case SOAPMETHOD_POST:
193 case HTTPMETHOD_MPOST:
194 callback = gSoapCallback;
195 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
196 "miniserver %d: got SOAP msg\n", info->socket);
197 break;
198 /* Gena Call */
199 case HTTPMETHOD_NOTIFY:
200 case HTTPMETHOD_SUBSCRIBE:
201 case HTTPMETHOD_UNSUBSCRIBE:
202 callback = gGenaCallback;
203 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
204 "miniserver %d: got GENA msg\n", info->socket);
205 break;
206 /* HTTP server call */
207 case HTTPMETHOD_GET:
208 case HTTPMETHOD_POST:
209 case HTTPMETHOD_HEAD:
210 case HTTPMETHOD_SIMPLEGET:
211 callback = gGetCallback;
212 host_validate_callback = gWebCallback_HostValidate;
213 cookie = gWebCallback_HostValidateCookie;
214 UPNPLIB_LOGINFO "MSG1107: miniserver socket="
215 << info->socket << ": got WEB server msg.\n";
216 break;
217 default:
218 callback = 0;
219 }
220 if (!callback) {
221 rc = HTTP_INTERNAL_SERVER_ERROR;
222 goto ExitFunction;
223 }
224 request = &hparser->msg;
225 if (upnplib::g_dbug) {
226 getNumericHostRedirection(info->socket, host_port, sizeof host_port);
227 UPNPLIB_LOGINFO "MSG1113: Redirect host_port=\"" << host_port << "\"\n";
228 }
229 /* check HOST header for an IP number -- prevents DNS rebinding. */
230 if (!httpmsg_find_hdr(request, HDR_HOST, &header)) {
231 rc = UPNP_E_BAD_HTTPMSG;
232 goto ExitFunction;
233 }
234 min_size = header.length < ((sizeof host_port) - 1)
235 ? header.length
236 : (sizeof host_port) - 1;
237 memcpy(host_port, header.buf, min_size);
238 host_port[min_size] = 0;
239 if (host_validate_callback) {
240 rc = host_validate_callback(host_port, cookie);
241 if (rc == UPNP_E_BAD_HTTPMSG) {
242 goto ExitFunction;
243 }
244 } else if (!host_header_is_numeric(host_port, min_size)) {
245 if (!gAllowLiteralHostRedirection) {
246 rc = UPNP_E_BAD_HTTPMSG;
247 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
248 "Possible DNS Rebind attack prevented.\n");
249 goto ExitFunction;
250 } else {
251 membuffer redir_buf;
252 static const char* redir_fmt = "HTTP/1.1 307 Temporary Redirect\r\n"
253 "Location: http://%s\r\n\r\n";
254 char redir_str[NAME_SIZE];
255 int timeout = HTTP_DEFAULT_TIMEOUT;
256
257 getNumericHostRedirection(info->socket, host_port,
258 sizeof host_port);
259 membuffer_init(&redir_buf);
260 snprintf(redir_str, NAME_SIZE, redir_fmt, host_port);
261 membuffer_append_str(&redir_buf, redir_str);
262 rc = http_SendMessage(info, &timeout, "b", redir_buf.buf,
263 redir_buf.length);
264 membuffer_destroy(&redir_buf);
265 goto ExitFunction;
266 }
267 }
268 callback(hparser, request, info);
269
270ExitFunction:
271 return rc;
272}

292void handle_request(
294 void* args) { // Expected to be mserv_request_t*
295 SOCKINFO info;
296 int http_error_code;
297 int ret_code;
298 int major = 1;
299 int minor = 1;
300 http_parser_t parser;
301 http_message_t* hmsg = NULL;
302 int timeout = HTTP_DEFAULT_TIMEOUT;
303 mserv_request_t* request_in = (mserv_request_t*)args;
304 SOCKET connfd = request_in->connfd;
305
306 UPNPLIB_LOGINFO "MSG1027: Miniserver socket "
307 << connfd << ": READING request from client...\n";
308 /* parser_request_init( &parser ); */ /* LEAK_FIX_MK */
309 hmsg = &parser.msg;
310 ret_code = sock_init_with_ip(&info, connfd,
311 (sockaddr*)&request_in->foreign_sockaddr);
312 if (ret_code != UPNP_E_SUCCESS) {
313 free(request_in);
314 httpmsg_destroy(hmsg);
315 return;
316 }
317
318 /* read */
319 ret_code = http_RecvMessage(&info, &parser, HTTPMETHOD_UNKNOWN, &timeout,
320 &http_error_code);
321 if (ret_code != 0) {
322 goto error_handler;
323 }
324
325 UPNPLIB_LOGINFO "MSG1106: miniserver socket=" << connfd
326 << ": PROCESSING...\n";
327 /* dispatch */
328 http_error_code = dispatch_request(&info, &parser);
329 if (http_error_code != 0) {
330 goto error_handler;
331 }
332 http_error_code = 0;
333
334error_handler:
335 if (http_error_code > 0) {
336 if (hmsg) {
337 major = hmsg->major_version;
338 minor = hmsg->minor_version;
339 }
340 // BUG! Don't try to send a status response to a remote client with
341 // http error (e.g. 400) if we have a socket error. It doesn't make
342 // sense. --Ingo
343 http_SendStatusResponse(&info, http_error_code, major, minor);
344 }
345 sock_destroy(&info, SD_BOTH);
346 httpmsg_destroy(hmsg);
347 free(request_in);
348
349 UpnpPrintf(UPNP_INFO, MSERV, __FILE__, __LINE__,
350 "miniserver %d: COMPLETE\n", connfd);
351}

357UPNP_INLINE void schedule_request_job(
359 SOCKET connfd,
361 sockaddr* clientAddr) {
362 TRACE("Executing schedule_request_job()")
363 UPNPLIB_LOGINFO "MSG1042: Schedule request job to host "
364 << upnplib::to_netaddrp(
365 reinterpret_cast<const sockaddr_storage*>(clientAddr))
366 << " with socket " << connfd << ".\n";
367
368 ThreadPoolJob job{};
369 mserv_request_t* request{
370 static_cast<mserv_request_t*>(std::malloc(sizeof(mserv_request_t)))};
371
372 if (request == nullptr) {
373 UPNPLIB_LOGCRIT "MSG1024: Socket " << connfd << ": out of memory.\n";
374 sock_close(connfd);
375 return;
376 }
377
378 request->connfd = connfd;
379 memcpy(&request->foreign_sockaddr, clientAddr,
380 sizeof(request->foreign_sockaddr));
381 TPJobInit(&job, (start_routine)handle_request, request);
382 TPJobSetFreeFunction(&job, free_handle_request_arg);
383 TPJobSetPriority(&job, MED_PRIORITY);
384 if (ThreadPoolAdd(&gMiniServerThreadPool, &job, NULL) != 0) {
385 UPNPLIB_LOGERR "MSG1025: Socket " << connfd
386 << ": cannot schedule request.\n";
387 free(request);
388 sock_close(connfd);
389 return;
390 }

411void fdset_if_valid( //
412 SOCKET a_sock,
413 fd_set* a_set
416) {
417 UPNPLIB_LOGINFO "MSG1086: Check sockfd=" << a_sock << ".\n";
418 if (a_sock == INVALID_SOCKET)
419 // This is a defined state and we return silently.
420 return;
421
422 if (a_sock < 3 || a_sock >= FD_SETSIZE) {
423 UPNPLIB_LOGERR "MSG1005: "
424 << (a_sock < 0 ? "Invalid" : "Prohibited") << " socket " << a_sock
425 << " not set to be monitored by ::select()"
426 << (a_sock >= 3 ? " because it violates FD_SETSIZE.\n" : ".\n");
427 return;
428 }
429 // Check if socket is valid and bound
430 try {
431 upnplib::CSocket_basic sockObj(a_sock);
432 sockObj.init();
433 if (sockObj.is_bound())
434
435 FD_SET(a_sock, a_set);
436
437 else
438 UPNPLIB_LOGINFO "MSG1002: Unbound socket "
439 << a_sock << " not set to be monitored by ::select().\n";
440
441 } catch (const std::exception& e) {
442 if (upnplib::g_dbug)
443 std::clog << e.what();
444 UPNPLIB_LOGCATCH "MSG1009: Invalid socket "
445 << a_sock << " not set to be monitored by ::select().\n";
446 }
447}

460int web_server_accept(
462 [[maybe_unused]] SOCKET lsock,
464 [[maybe_unused]] fd_set& set) {
465#ifndef COMPA_HAVE_WEBSERVER
466 return UPNP_E_NO_WEB_SERVER;
467#else
468 TRACE("Executing web_server_accept()")
469 if (lsock == INVALID_SOCKET || !FD_ISSET(lsock, &set)) {
470 UPNPLIB_LOGINFO "MSG1012: Socket("
471 << lsock << ") invalid or not in file descriptor set.\n";
472 return UPNP_E_SOCKET_ERROR;
473 }
474
475 SOCKET asock;
476 upnplib::sockaddr_t clientAddr;
477 socklen_t clientLen = sizeof(clientAddr);
478
479 // accept a network request connection
480 asock = umock::sys_socket_h.accept(lsock, &clientAddr.sa, &clientLen);
481 if (asock == INVALID_SOCKET) {
482 UPNPLIB_LOGERR "MSG1022: Error in ::accept(): " << std::strerror(errno)
483 << ".\n";
484 return UPNP_E_SOCKET_ACCEPT;
485 }
486
487 // Schedule a job to manage a UPnP request from a remote host.
488 char buf_ntop[INET6_ADDRSTRLEN + 7];
489 inet_ntop(AF_INET, &clientAddr.sin.sin_addr, buf_ntop, sizeof(buf_ntop));
490 UPNPLIB_LOGINFO "MSG1023: Connected to host "
491 << buf_ntop << ":" << ntohs(clientAddr.sin.sin_port) << " with socket "
492 << asock << ".\n";
493 schedule_request_job(asock, &clientAddr.sa);
494
495 return UPNP_E_SUCCESS;
496#endif /* COMPA_HAVE_WEBSERVER */
497}

538int receive_from_stopSock(
539 SOCKET ssock,
540 fd_set* set
542) {
543 TRACE("Executing receive_from_stopSock()")
544 constexpr char shutdown_str[]{"ShutDown"};
545
546 if (!FD_ISSET(ssock, set))
547 return 0; // Nothing to do for this socket
548
549 upnplib::sockaddr_t clientAddr{};
550 socklen_t clientLen{sizeof(clientAddr)}; // May be modified
551
552 // The receive buffer is one byte greater with '\0' than the max receiving
553 // bytes so the received message will always be terminated.
554 char receiveBuf[sizeof(shutdown_str) + 1]{};
555 char buf_ntop[INET6_ADDRSTRLEN];
556
557 // receive from
558 SSIZEP_T byteReceived = umock::sys_socket_h.recvfrom(
559 ssock, receiveBuf, sizeof(shutdown_str), 0, &clientAddr.sa, &clientLen);
560 if (byteReceived == SOCKET_ERROR ||
561 inet_ntop(AF_INET, &clientAddr.sin.sin_addr, buf_ntop,
562 sizeof(buf_ntop)) == nullptr) {
563 UPNPLIB_LOGCRIT "MSG1038: Failed to receive data from socket "
564 << ssock << ". Stop miniserver.\n";
565 return 1;
566 }
567
568 // 16777343 are netorder bytes of "127.0.0.1"
569 if (clientAddr.sin.sin_addr.s_addr != 16777343 ||
570 strcmp(receiveBuf, shutdown_str) != 0) //
571 {
572 char nullstr[]{"\\0"};
573 if (byteReceived == 0 || receiveBuf[byteReceived - 1] != '\0')
574 nullstr[0] = '\0';
575 UPNPLIB_LOGERR "MSG1039: Received \""
576 << receiveBuf << nullstr << "\" from " << buf_ntop << ":"
577 << ntohs(clientAddr.sin.sin_port)
578 << ", must be \"ShutDown\\0\" from 127.0.0.1:*. Don't "
579 "stopping miniserver.\n";
580 return 0;
581 }
582
583 UPNPLIB_LOGINFO "MSG1040: On socket "
584 << ssock << " received ordinary datagram \"" << receiveBuf
585 << "\\0\" from " << buf_ntop << ":" << ntohs(clientAddr.sin.sin_port)
586 << ". Stop miniserver.\n";
587 return 1;
588}

601void RunMiniServer(
605 MiniServerSockArray* miniSock) {
606 UPNPLIB_LOGINFO "MSG1085: Executing...\n";
607 fd_set expSet;
608 fd_set rdSet;
609 int stopSock = 0;
610
611 // On MS Windows INVALID_SOCKET is unsigned -1 = 18446744073709551615 so we
612 // get maxMiniSock with this big number even if there is only one
613 // INVALID_SOCKET. Incrementing it at the end results in 0. To be portable
614 // we must not assume INVALID_SOCKET to be -1. --Ingo
615 SOCKET maxMiniSock = 0;
616 maxMiniSock = //
617 std::max(maxMiniSock, miniSock->miniServerSock4 == INVALID_SOCKET
618 ? 0
619 : miniSock->miniServerSock4);
620 maxMiniSock = //
621 std::max(maxMiniSock, miniSock->miniServerSock6 == INVALID_SOCKET
622 ? 0
623 : miniSock->miniServerSock6);
624 maxMiniSock = //
625 std::max(maxMiniSock, miniSock->miniServerSock6UlaGua == INVALID_SOCKET
626 ? 0
627 : miniSock->miniServerSock6UlaGua);
628 maxMiniSock = //
629 std::max(maxMiniSock, miniSock->miniServerStopSock == INVALID_SOCKET
630 ? 0
631 : miniSock->miniServerStopSock);
632 maxMiniSock = //
633 std::max(maxMiniSock, miniSock->ssdpSock4 == INVALID_SOCKET
634 ? 0
635 : miniSock->ssdpSock4);
636 maxMiniSock = //
637 std::max(maxMiniSock, miniSock->ssdpSock6 == INVALID_SOCKET
638 ? 0
639 : miniSock->ssdpSock6);
640 maxMiniSock = //
641 std::max(maxMiniSock, miniSock->ssdpSock6UlaGua == INVALID_SOCKET
642 ? 0
643 : miniSock->ssdpSock6UlaGua);
644#ifdef COMPA_HAVE_CTRLPT_SSDP
645 maxMiniSock = //
646 std::max(maxMiniSock, miniSock->ssdpReqSock4 == INVALID_SOCKET
647 ? 0
648 : miniSock->ssdpReqSock4);
649 maxMiniSock = //
650 std::max(maxMiniSock, miniSock->ssdpReqSock6 == INVALID_SOCKET
651 ? 0
652 : miniSock->ssdpReqSock6);
653#endif
654 ++maxMiniSock;
655
656 gMServState = MSERV_RUNNING;
657 while (!stopSock) {
658 FD_ZERO(&rdSet);
659 FD_ZERO(&expSet);
660 /* FD_SET()'s */
661 FD_SET(miniSock->miniServerStopSock, &expSet);
662 FD_SET(miniSock->miniServerStopSock, &rdSet);
663 fdset_if_valid(miniSock->miniServerSock4, &rdSet);
664 fdset_if_valid(miniSock->miniServerSock6, &rdSet);
665 fdset_if_valid(miniSock->miniServerSock6UlaGua, &rdSet);
666 fdset_if_valid(miniSock->ssdpSock4, &rdSet);
667 fdset_if_valid(miniSock->ssdpSock6, &rdSet);
668 fdset_if_valid(miniSock->ssdpSock6UlaGua, &rdSet);
669#ifdef COMPA_HAVE_CTRLPT_SSDP
670 fdset_if_valid(miniSock->ssdpReqSock4, &rdSet);
671 fdset_if_valid(miniSock->ssdpReqSock6, &rdSet);
672#endif
673
674 /* select() */
675 int ret = umock::sys_socket_h.select(static_cast<int>(maxMiniSock),
676 &rdSet, NULL, &expSet, NULL);
677
678 if (ret == SOCKET_ERROR) {
679 if (errno == EINTR) {
680 // A signal was caught, not for us. We ignore it and
681 continue;
682 }
683 if (errno == EBADF) {
684 // A closed socket file descriptor was given in one of the
685 // sets. For details look at
686 // REF:_[Should_I_assert_fail_on_select()_EBADF?](https://stackoverflow.com/q/28015859/5014688)
687 // It is difficult to determine here what file descriptor
688 // in rdSet or expSet is invalid. So I ensure that only valid
689 // socket fds are given by checking them with fdset_if_valid()
690 // before calling select(). Doing this I have to
691 continue;
692 }
693 // All other errors EINVAL and ENOMEM are critical and cannot
694 // continue run mininserver.
695 UPNPLIB_LOGCRIT "MSG1021: Error in ::select(): "
696 << std::strerror(errno) << ".\n";
697 break;
698 }
699
700 // Accept requested connection from a remote control point and run the
701 // connection in a new thread. Due to side effects with threading we
702 // need to avoid lazy evaluation with chained || because all
703 // web_server_accept() must be called.
704 // if (ret1 == UPNP_E_SUCCESS || ret2 == UPNP_E_SUCCESS ||
705 // ret3 == UPNP_E_SUCCESS) {
706 [[maybe_unused]] int ret1 =
707 web_server_accept(miniSock->miniServerSock4, rdSet);
708 [[maybe_unused]] int ret2 =
709 web_server_accept(miniSock->miniServerSock6, rdSet);
710 [[maybe_unused]] int ret3 =
711 web_server_accept(miniSock->miniServerSock6UlaGua, rdSet);
712#ifdef COMPA_HAVE_CTRLPT_SSDP
713 ssdp_read(&miniSock->ssdpReqSock4, &rdSet);
714 ssdp_read(&miniSock->ssdpReqSock6, &rdSet);
715#endif
716 ssdp_read(&miniSock->ssdpSock4, &rdSet);
717 ssdp_read(&miniSock->ssdpSock6, &rdSet);
718 ssdp_read(&miniSock->ssdpSock6UlaGua, &rdSet);
719 // }
720
721 // Check if we have received a packet from
722 // localhost(127.0.0.1) that will stop the miniserver.
723 stopSock = receive_from_stopSock(miniSock->miniServerStopSock, &rdSet);
724 } // while (!stopsock)
725
726 /* Close all sockets. */
727 sock_close(miniSock->miniServerSock4);
728 sock_close(miniSock->miniServerSock6);
729 sock_close(miniSock->miniServerSock6UlaGua);
730 sock_close(miniSock->miniServerStopSock);
731 sock_close(miniSock->ssdpSock4);
732 sock_close(miniSock->ssdpSock6);
733 sock_close(miniSock->ssdpSock6UlaGua);
734#ifdef COMPA_HAVE_CTRLPT_SSDP
735 sock_close(miniSock->ssdpReqSock4);
736 sock_close(miniSock->ssdpReqSock6);
737#endif
738 /* Free minisock. */
739 umock::stdlib_h.free(miniSock);
740 gMServState = MSERV_IDLE;
741
742 return;
743}

757int get_port(
759 SOCKET sockfd,
761 uint16_t* port) {
762 TRACE("Executing get_port(), calls system ::getsockname()")
763 upnplib::sockaddr_t sockinfo{};
764 socklen_t len(sizeof sockinfo); // May be modified by getsockname()
765
766 if (umock::sys_socket_h.getsockname(sockfd, &sockinfo.sa, &len) == -1)
767 // system error (errno etc.) is expected to be unmodified on return.
768 return -1;
769
770 switch (sockinfo.ss.ss_family) {
771 case AF_INET:
772 *port = ntohs(sockinfo.sin.sin_port);
773 break;
774 case AF_INET6:
775 *port = ntohs(sockinfo.sin6.sin6_port);
776 break;
777 default:
778 // system error (errno etc.) is expected to be unmodified on return.
779 return -1;
780 }
781 UPNPLIB_LOGINFO "MSG1063: sockfd=" << sockfd << ", port=" << *port << ".\n";
782
783 return 0;
784}

983int StartMiniServer([[maybe_unused]] in_port_t* listen_port4,
984 [[maybe_unused]] in_port_t* listen_port6,
985 [[maybe_unused]] in_port_t* listen_port6UlaGua) {
986 UPNPLIB_LOGINFO "MSG1068: Executing...\n";
987 constexpr int max_count{10000};
988 MiniServerSockArray* miniSocket;
989 ThreadPoolJob job;
990 int ret_code{UPNP_E_INTERNAL_ERROR};
991
992 memset(&job, 0, sizeof(job));
993
994 if (gMServState != MSERV_IDLE) {
995 /* miniserver running. */
996 UPNPLIB_LOGERR "MSG1087: Cannot start. Miniserver is running.\n";
997 return UPNP_E_INTERNAL_ERROR;
998 }
999 miniSocket = (MiniServerSockArray*)malloc(sizeof(MiniServerSockArray));
1000 if (!miniSocket) {
1001 return UPNP_E_OUTOF_MEMORY;
1002 }
1003 InitMiniServerSockArray(miniSocket);
1004
1005 // Instantiate socket objects and point to them in miniSocket
1006 upnplib::CSocket Sock6LlaObj(AF_INET6, SOCK_STREAM);
1007 miniSocket->MiniSvrSock6LlaObj = &Sock6LlaObj;
1008 upnplib::CSocket Sock6UadObj(AF_INET6, SOCK_STREAM);
1009 miniSocket->MiniSvrSock6UadObj = &Sock6UadObj;
1010 upnplib::CSocket Sock4Obj(AF_INET, SOCK_STREAM);
1011 miniSocket->MiniSvrSock4Obj = &Sock4Obj;
1012
1013#ifdef COMPA_HAVE_WEBSERVER
1014 if (*listen_port4 == 0 || *listen_port6 == 0 || *listen_port6UlaGua == 0) {
1015 // Create a simple random number generator for port numbers. We need
1016 // this because we do not reuse addresses before TIME_WAIT has expired
1017 // (socket option SO_REUSEADDR = false). We need to use different
1018 // socket addresses and that is already given with different port
1019 // numbers.
1020 std::random_device rd; // obtain a random number from hardware
1021 std::minstd_rand random(rd()); // seed the generator
1022 std::uniform_int_distribution<in_port_t> portno(
1023 APPLICATION_LISTENING_PORT, 65535); // used range
1024
1025 in_port_t listen_port = portno(random);
1026 if (*listen_port4 == 0)
1027 *listen_port4 = listen_port;
1028 if (*listen_port6 == 0)
1029 *listen_port6 = listen_port;
1030 if (*listen_port6UlaGua == 0)
1031 *listen_port6UlaGua = listen_port;
1032 }
1033 /* V4 and V6 http listeners. */
1034 ret_code = get_miniserver_sockets(miniSocket, *listen_port4, *listen_port6,
1035 *listen_port6UlaGua);
1036 if (ret_code != UPNP_E_SUCCESS) {
1037 free(miniSocket);
1038 return ret_code;
1039 }
1040#endif
1041
1042 /* Stop socket (To end miniserver processing). */
1043 ret_code = get_miniserver_stopsock(miniSocket);
1044 if (ret_code != UPNP_E_SUCCESS) {
1045 sock_close(miniSocket->miniServerSock4);
1046 sock_close(miniSocket->miniServerSock6);
1047 sock_close(miniSocket->miniServerSock6UlaGua);
1048 free(miniSocket);
1049 return ret_code;
1050 }
1051#if defined(COMPA_HAVE_CTRLPT_SSDP) || defined(COMPA_HAVE_DEVICE_SSDP)
1052 /* SSDP socket for discovery/advertising. */
1053 ret_code = umock::pupnp_ssdp.get_ssdp_sockets(miniSocket);
1054 if (ret_code != UPNP_E_SUCCESS) {
1055 sock_close(miniSocket->miniServerSock4);
1056 sock_close(miniSocket->miniServerSock6);
1057 sock_close(miniSocket->miniServerSock6UlaGua);
1058 sock_close(miniSocket->miniServerStopSock);
1059 free(miniSocket);
1060 return ret_code;
1061 }
1062#endif
1063 TPJobInit(&job, (start_routine)RunMiniServer_f, (void*)miniSocket);
1064 TPJobSetPriority(&job, MED_PRIORITY);
1065 TPJobSetFreeFunction(&job, (free_routine)free);
1066 ret_code = ThreadPoolAddPersistent(&gMiniServerThreadPool, &job, NULL);
1067 if (ret_code != 0) {
1068 sock_close(miniSocket->miniServerSock4);
1069 sock_close(miniSocket->miniServerSock6);
1070 sock_close(miniSocket->miniServerSock6UlaGua);
1071 sock_close(miniSocket->miniServerStopSock);
1072 sock_close(miniSocket->ssdpSock4);
1073 sock_close(miniSocket->ssdpSock6);
1074 sock_close(miniSocket->ssdpSock6UlaGua);
1075#ifdef COMPA_HAVE_CTRLPT_SSDP
1076 sock_close(miniSocket->ssdpReqSock4);
1077 sock_close(miniSocket->ssdpReqSock6);
1078#endif
1079 free(miniSocket);
1080 return UPNP_E_OUTOF_MEMORY;
1081 }
1082 /* Wait for miniserver to start. */
1083 int count{0};
1084 while (gMServState != (MiniServerState)MSERV_RUNNING && count < max_count) {
1085 /* 0.05s */
1086 imillisleep(50);
1087 count++;
1088 }
1089 if (count >= max_count) {
1090 /* Took it too long to start that thread. */
1091 sock_close(miniSocket->miniServerSock4);
1092 sock_close(miniSocket->miniServerSock6);
1093 sock_close(miniSocket->miniServerSock6UlaGua);
1094 sock_close(miniSocket->miniServerStopSock);
1095 sock_close(miniSocket->ssdpSock4);
1096 sock_close(miniSocket->ssdpSock6);
1097 sock_close(miniSocket->ssdpSock6UlaGua);
1098#ifdef COMPA_HAVE_CTRLPT_SSDP
1099 sock_close(miniSocket->ssdpReqSock4);
1100 sock_close(miniSocket->ssdpReqSock6);
1101#endif
1102 return UPNP_E_INTERNAL_ERROR;
1103 }
1104#ifdef COMPA_HAVE_WEBSERVER
1105 *listen_port4 = miniSocket->miniServerPort4;
1106 *listen_port6 = miniSocket->miniServerPort6;
1107 *listen_port6UlaGua = miniSocket->miniServerPort6UlaGua;
1108#endif
1109
1110 return UPNP_E_SUCCESS;
1111}

1115 "Inside StopMiniServer()\n");
1116
1117 socklen_t socklen = sizeof(struct sockaddr_in);
1118 SOCKET sock;

1120 char buf[256] = "ShutDown";

1139 ssdpAddr.sin_family = (sa_family_t)AF_INET;

1141 ssdpAddr.sin_port = htons(miniStopSockPort);

1145 if (gMServState == (MiniServerState)MSERV_IDLE) {
1146 break;
1147 }
1148 isleep(1u);
1149 }
