46int map_str_to_int(const char* name, size_t name_len,
47 const str_int_entry* table, int num_entries,
48 int case_sensitive) {
49 TRACE("Executing map_str_to_int()");
50 // num_entries must not be negative due to unsigned type cast later.
51 if (name_len <= 0 || table == nullptr || num_entries <= 0)
52 return -1;
53
54 // Create a std::array container from plain C array.
55 std::array<str_int_entry, container_size> tbl{};
56 if (tbl.size() < static_cast<size_t>(num_entries)) {
57 UPNPLIB_LOGCRIT "MSG1029: std::array container["
58 << tbl.size() << "] is to small, requested " << num_entries
59 << " entries. This program error MUST be fixed! Program is stable "
60 "but may have ignored runtime conditions.\n";
61 return -1;
62 }
63 size_t i{0};
64 for (; i < static_cast<size_t>(num_entries); i++) {
65 tbl[i] = table[i];
66 }
67 // Fill remaining entries with dummy values needed for binary search.
68 constexpr char str_zzzz[]{"ZZZZ"};
69 for (; i < tbl.size(); i++) {
70 tbl[i].name = str_zzzz;
71 tbl[i].id = INT_MIN + 8;
72 }
73
74 return UPnPsdk::str_to_int(name, tbl, case_sensitive);
75}
