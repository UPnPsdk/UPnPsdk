118static int SetServiceTable(
120 int serviceType,
122 const char* UDN,
124 const char* serviceId,
126 const char* serviceTypeS,
128 struct TvService* out) {
129 int i = 0;
130
131 strcpy(out->UDN, UDN);
132 strcpy(out->ServiceId, serviceId);
133 strcpy(out->ServiceType, serviceTypeS);
134
135 switch (serviceType) {
136 case TV_SERVICE_CONTROL:
137 out->VariableCount = TV_CONTROL_VARCOUNT;
138 for (i = 0; i < tv_service_table[TV_SERVICE_CONTROL].VariableCount;
139 i++) {
140 tv_service_table[TV_SERVICE_CONTROL].VariableName[i] =

143 tvc_varval[i];

145 tvc_varval_def[i]);
146 }
147 break;
148 case TV_SERVICE_PICTURE:
149 out->VariableCount = TV_PICTURE_VARCOUNT;
150 for (i = 0; i < tv_service_table[TV_SERVICE_PICTURE].VariableCount;

162 }

221int TvDeviceStateTableInit(char* DescDocURL) {
222 IXML_Document* DescDoc = NULL;
223 int ret = UPNP_E_SUCCESS;
224 char* servid_ctrl = NULL;
225 char* evnturl_ctrl = NULL;
226 char* ctrlurl_ctrl = NULL;
227 char* servid_pict = NULL;
228 char* evnturl_pict = NULL;
229 char* ctrlurl_pict = NULL;
230 char* udn = NULL;
231
232 /*Download description document */
233 if (UpnpDownloadXmlDoc(DescDocURL, &DescDoc) != UPNP_E_SUCCESS) {
234 SampleUtil_Print("TvDeviceStateTableInit -- Error Parsing %s\n",
235 DescDocURL);
236 ret = UPNP_E_INVALID_DESC;
237 goto error_handler;
238 }
239 udn = SampleUtil_GetFirstDocumentItem(DescDoc, "UDN");
240 /* Find the Tv Control Service identifiers */
241 if (!SampleUtil_FindAndParseService(
242 DescDoc, DescDocURL, TvServiceType[TV_SERVICE_CONTROL],
243 &servid_ctrl, &evnturl_ctrl, &ctrlurl_ctrl)) {
244 SampleUtil_Print("TvDeviceStateTableInit -- Error: Could not "
245 "find Service: %s\n",
246 TvServiceType[TV_SERVICE_CONTROL]);
247 ret = UPNP_E_INVALID_DESC;
248 goto error_handler;
249 }
250 /* set control service table */
251 SetServiceTable(TV_SERVICE_CONTROL, udn, servid_ctrl,
252 TvServiceType[TV_SERVICE_CONTROL],
253 &tv_service_table[TV_SERVICE_CONTROL]);
254
255 /* Find the Tv Picture Service identifiers */
256 if (!SampleUtil_FindAndParseService(
257 DescDoc, DescDocURL, TvServiceType[TV_SERVICE_PICTURE],
258 &servid_pict, &evnturl_pict, &ctrlurl_pict)) {
259 SampleUtil_Print("TvDeviceStateTableInit -- Error: Could not "
260 "find Service: %s\n",
261 TvServiceType[TV_SERVICE_PICTURE]);
262 ret = UPNP_E_INVALID_DESC;
263 goto error_handler;
264 }
265 /* set picture service table */
266 SetServiceTable(TV_SERVICE_PICTURE, udn, servid_pict,
267 TvServiceType[TV_SERVICE_PICTURE],
268 &tv_service_table[TV_SERVICE_PICTURE]);
269
270error_handler:
271 /* clean up */
272 if (udn)
273 free(udn);
274 if (servid_ctrl)
275 free(servid_ctrl);
276 if (evnturl_ctrl)
277 free(evnturl_ctrl);
278 if (ctrlurl_ctrl)
279 free(ctrlurl_ctrl);
280 if (servid_pict)
281 free(servid_pict);
282 if (evnturl_pict)
283 free(evnturl_pict);
284 if (ctrlurl_pict)
285 free(ctrlurl_pict);
286 if (DescDoc)
287 ixmlDocument_free(DescDoc);
288
289 return (ret);
290}

391int TvDeviceHandleActionRequest(UpnpActionRequest* ca_event) {
392 /* Defaults if action not found. */
393 int action_found = 0;
394 int i = 0;
395 int service = -1;
396 int retCode = 0;
397 const char* errorString = NULL;
398 const char* devUDN = NULL;
399 const char* serviceID = NULL;
400 const char* actionName = NULL;
401 IXML_Document* actionResult = NULL;
402
403 UpnpActionRequest_set_ErrCode(ca_event, 0);
404 UpnpActionRequest_set_ActionResult(ca_event, NULL);
405
406 devUDN = UpnpString_get_String(UpnpActionRequest_get_DevUDN(ca_event));
407 serviceID =
408 UpnpString_get_String(UpnpActionRequest_get_ServiceID(ca_event));
409 actionName =
410 UpnpString_get_String(UpnpActionRequest_get_ActionName(ca_event));
411 if (strcmp(devUDN, tv_service_table[TV_SERVICE_CONTROL].UDN) == 0 &&
412 strcmp(serviceID, tv_service_table[TV_SERVICE_CONTROL].ServiceId) ==
413 0) {
414 /* Request for action in the TvDevice Control Service. */
415 service = TV_SERVICE_CONTROL;
416 } else if (strcmp(devUDN, tv_service_table[TV_SERVICE_PICTURE].UDN) == 0 &&
417 strcmp(serviceID,
418 tv_service_table[TV_SERVICE_PICTURE].ServiceId) == 0) {
419 /* Request for action in the TvDevice Picture Service. */
420 service = TV_SERVICE_PICTURE;
421 }
422 /* Find and call appropriate procedure based on action name.
423 * Each action name has an associated procedure stored in the
424 * service table. These are set at initialization. */
425 for (i = 0;
426 i < TV_MAXACTIONS && tv_service_table[service].ActionNames[i] != NULL;
427 i++) {
428 if (!strcmp(actionName, tv_service_table[service].ActionNames[i])) {
429 if (!strcmp(tv_service_table[TV_SERVICE_CONTROL]
430 .VariableStrVal[TV_CONTROL_POWER],
431 "1") ||
432 !strcmp(actionName, "PowerOn")) {
433 retCode = tv_service_table[service].actions[i](
434 UpnpActionRequest_get_ActionRequest(ca_event),
435 &actionResult, &errorString);
436 UpnpActionRequest_set_ActionResult(ca_event, actionResult);
437 } else {
438 errorString = "Power is Off";
439 retCode = UPNP_E_INTERNAL_ERROR;
440 }
441 action_found = 1;
442 break;
443 }
444 }
445
446 if (!action_found) {
447 UpnpActionRequest_set_ActionResult(ca_event, NULL);
448 UpnpActionRequest_strcpy_ErrStr(ca_event, "Invalid Action");
449 UpnpActionRequest_set_ErrCode(ca_event, 401);
450 } else {
451 if (retCode == UPNP_E_SUCCESS) {
452 UpnpActionRequest_set_ErrCode(ca_event, UPNP_E_SUCCESS);
453 } else {
454 /* copy the error string */
455 UpnpActionRequest_strcpy_ErrStr(ca_event, errorString);
456 switch (retCode) {
457 case UPNP_E_INVALID_PARAM:
458 UpnpActionRequest_set_ErrCode(ca_event, 402);
459 break;
460 case UPNP_E_INTERNAL_ERROR:
461 default:
462 UpnpActionRequest_set_ErrCode(ca_event, 501);
463 break;
464 }
465 }
466 }
467
468 return UpnpActionRequest_get_ErrCode(ca_event);
469}

1220int TvDeviceCallbackEventHandler(Upnp_EventType EventType, const void* Event,
1221 [[maybe_unused]] void* Cookie) {
1222 switch (EventType) {
1223 case UPNP_EVENT_SUBSCRIPTION_REQUEST:
1224 TvDeviceHandleSubscriptionRequest((UpnpSubscriptionRequest*)Event);
1225 break;
1226 case UPNP_CONTROL_GET_VAR_REQUEST:
1227 TvDeviceHandleGetVarRequest((UpnpStateVarRequest*)Event);
1228 break;
1229 case UPNP_CONTROL_ACTION_REQUEST:
1230 TvDeviceHandleActionRequest((UpnpActionRequest*)Event);
1231 break;
1232 /* ignore these cases, since this is not a control point */
1233 case UPNP_DISCOVERY_ADVERTISEMENT_ALIVE:
1234 case UPNP_DISCOVERY_SEARCH_RESULT:
1235 case UPNP_DISCOVERY_SEARCH_TIMEOUT:
1236 case UPNP_DISCOVERY_ADVERTISEMENT_BYEBYE:
1237 case UPNP_CONTROL_ACTION_COMPLETE:
1238 case UPNP_CONTROL_GET_VAR_COMPLETE:
1239 case UPNP_EVENT_RECEIVED:
1240 case UPNP_EVENT_RENEWAL_COMPLETE:
1241 case UPNP_EVENT_SUBSCRIBE_COMPLETE:
1242 case UPNP_EVENT_UNSUBSCRIBE_COMPLETE:
1243 break;
1244 default:
1245 SampleUtil_Print("Error in TvDeviceCallbackEventHandler: "
1246 "unknown event type %d\n",
1247 EventType);
1248 }
1249 /* Print a summary of the event received */
1250 SampleUtil_PrintEvent(EventType, Event);
1251
1252 return 0;
1253}
