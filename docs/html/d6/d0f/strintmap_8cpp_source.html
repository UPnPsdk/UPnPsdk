43int map_str_to_int(const char* name, size_t name_len,
44 const str_int_entry* table, int num_entries,
45 int case_sensitive) {
46 TRACE("Executing map_str_to_int()");
47 if (name == nullptr || table == nullptr)
48 return -1;
49
50 int top, mid, bot;
51 int cmp;
52 memptr name_ptr;
53
54 name_ptr.buf = (char*)name;
55 name_ptr.length = name_len;
56
57 top = 0;
58 bot = num_entries - 1;
59
60 while (top <= bot) {
61 mid = (top + bot) / 2;
62 if (case_sensitive) {
63 /*cmp = strcmp( name, table[mid].name ); */
64 cmp = memptr_cmp(&name_ptr, table[mid].name);
65 } else {
66 /*cmp = strcasecmp( name, table[mid].name ); */
67 cmp = memptr_cmp_nocase(&name_ptr, table[mid].name);
68 }
69
70 if (cmp > 0) {
71 top = mid + 1; /* look below mid */
72 } else if (cmp < 0) {
73 bot = mid - 1; /* look above mid */
74 } else { /* cmp == 0 */
75 return mid; /* match; return table index */
76 }
77 }
78
79 return -1; /* header name not found */
80}
