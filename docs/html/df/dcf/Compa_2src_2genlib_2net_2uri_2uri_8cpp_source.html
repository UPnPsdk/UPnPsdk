104int replace_escaped(
106 char* in,
108 size_t index,
111 size_t* max) {
112 int tempInt = 0;
113 char tempChar = 0;
114 size_t i = (size_t)0;
115 size_t j = (size_t)0;
116
117 if (in[index] == '%' && isxdigit(in[index + (size_t)1]) &&
118 isxdigit(in[index + (size_t)2])) {
119 /* Note the "%2x", makes sure that we convert a maximum of two
120 * characters. */
121#ifdef _WIN32
122 if (sscanf_s(&in[index + (size_t)1],
123#else
124 if (sscanf(&in[index + (size_t)1],
125#endif
126 "%2x", (unsigned int*)&tempInt) != 1) {
127 return 0;
128 }
129 tempChar = (char)tempInt;
130 for (i = index + (size_t)3, j = index; j < *max; i++, j++) {
131 in[j] = tempChar;
132 if (i < *max) {
133 tempChar = in[i];
134 } else {
135 tempChar = 0;
136 }
137 }
138 *max -= (size_t)2;
139 return 1;
140 } else {
141 return 0;
142 }
143}

149int create_url_list(memptr* a_url_list, URL_list* a_out) {
150 if (!a_out)
151 return 0;
152 memset(a_out, 0, sizeof(*a_out));
153 if (!a_url_list)
154 return 0;
155
156 std::string_view url_list_sv =
157 std::string_view(a_url_list->buf, a_url_list->length);
158
159 if (url_list_sv.empty() || url_list_sv == "<>")
160 return 0;
161
162 // Verify correct delimiter '<', '><', '>', and count URLs.
163 auto it{url_list_sv.begin()};
164 size_t url_count{1};
165 auto ch0 = *it;
166 for (++it; it < url_list_sv.end() - 1; it++) {
167 if (*it == '<' || (*it == '>' && *++it != '<'))
168 return UPNP_E_INVALID_URL;
169 if (*it == '<')
170 url_count++;
171 }
172 auto ch1 = *it;
173 // Check begin and end delimiter. They must be '<'...'>'.
174 // To be compatible with pUPnP we have different error messages.
175 if ((ch0 == '<' && ch1 != '>') || (ch0 != '<' && ch1 == '>'))
176 return UPNP_E_INVALID_URL;
177 if (!(ch0 == '<' && ch1 == '>'))
178 return 0;
179
180 // Allocate memory and copy the serialized urls to it.
181 auto url_list_size = url_list_sv.size(); // There is no terminating '\0'.
182 char* base_urls = static_cast<char*>(malloc(url_list_size + 1));
183 if (!base_urls)
184 return UPNP_E_OUTOF_MEMORY;
185 // copy and terminate url_list ("<url><url>\0").
186 base_urls[url_list_sv.copy(base_urls, url_list_size)] = '\0';
187 std::string_view base_urls_sv = std::string_view(base_urls, url_list_size);
188
189 // Allocate memory for the parsed urls.
190 uri_type* parsed_urls =
191 static_cast<uri_type*>(malloc(sizeof(uri_type) * url_count));
192 if (!parsed_urls) {
193 free(base_urls);
194 return UPNP_E_OUTOF_MEMORY;
195 }
196
197 // Create views into the base_urls to its components (scheme, hostport,
198 // path, etc.) and store them into the allocated buffer as members of a
199 // trivial C array.
200 size_t store_idx{}; // We have 'uri_count' for URIs correct delimited with
201 // '<url><url>' but the URIs may be wrong. These are not
202 // parsed and returned. To count only the successful
203 // parsed URIs I use this index.
204 std::string_view base_url_sv;
205 uri_type splitted_url;
206 for (size_t i{}; i < url_count; i++) {
207 size_t pos = base_urls_sv.find_first_of('>');
208 base_url_sv = base_urls_sv.substr(0, pos); // Extract current URL.
209 base_url_sv.remove_prefix(1); // Remove leading '<' from current URL.
210 base_urls_sv.remove_prefix(pos + 1); // Remove current URL from list.
211
212 // parse_uri
213 int return_code =
214 parse_uri(base_url_sv.data(), base_url_sv.size(), &splitted_url);
215
216 if (return_code == HTTP_SUCCESS &&
217 splitted_url.hostport.text.size != 0) {
218 // No errors detected, cache the parsed result. Only URLs with
219 // network addresses are considered.
220 parsed_urls[store_idx++] = splitted_url;
221
222 } else if (return_code == UPNP_E_OUTOF_MEMORY) {
223 free(base_urls);
224 free(parsed_urls);
225 return UPNP_E_OUTOF_MEMORY;
226 }
227 }
228
229 // All values are available now. Fill the destination structure.
230 a_out->parsedURLs = parsed_urls;
231 a_out->URLs = base_urls;
232 a_out->size = store_idx;
233
234 if (store_idx > INT_MAX) // Paranoia checking due to cast.
235 return UPNP_E_OUTOF_MEMORY;
236 return static_cast<int>(store_idx);
237}

240int copy_URL_list(URL_list* in, URL_list* out) {
241 out->URLs = nullptr;
242 out->parsedURLs = nullptr;
243 out->size = 0u;
244
245 size_t len = strlen(in->URLs) + 1;
246 out->URLs = static_cast<char*>(malloc(len));
247 if (!out->URLs)
248 return UPNP_E_OUTOF_MEMORY;
249 // out->parsedURLs = static_cast<uri_type*>(calloc(in->size,
250 // sizeof(uri_type)));
251 out->parsedURLs =
252 static_cast<uri_type*>(malloc(sizeof(uri_type) * in->size));
253 if (!out->parsedURLs) {
254 free(out->URLs);
255 return UPNP_E_OUTOF_MEMORY;
256 }
257
258 memcpy(out->URLs, in->URLs, len);
259 for (size_t i{0}; i < in->size; i++) {
260 // Having pointer within the coppied structure a deep copy is needed.
261 out->parsedURLs[i].type = in->parsedURLs[i].type;
262 copy_token(&in->parsedURLs[i].scheme, in->URLs,
263 &out->parsedURLs[i].scheme, out->URLs);
264 out->parsedURLs[i].path_type = in->parsedURLs[i].path_type;
265 copy_token(&in->parsedURLs[i].pathquery, in->URLs,
266 &out->parsedURLs[i].pathquery, out->URLs);
267 copy_token(&in->parsedURLs[i].fragment, in->URLs,
268 &out->parsedURLs[i].fragment, out->URLs);
269 copy_token(&in->parsedURLs[i].hostport.text, in->URLs,
270 &out->parsedURLs[i].hostport.text, out->URLs);
271 memcpy(&out->parsedURLs[i].hostport.IPaddress,
272 &in->parsedURLs[i].hostport.IPaddress,
273 sizeof(struct sockaddr_storage));
274 }
275 out->size = in->size;
276
277 return HTTP_SUCCESS;
278}

311int token_string_casecmp(token* in1, const char* in2) {
312 size_t in2_length = strlen(in2);
313 if (in1->buff == nullptr && in2_length == 0)
314 return 0;
315 else if (in1->buff == nullptr && in2_length != 0)
316 return -1;
317 else if (in1->size < in2_length)
318 return -1;
319 else if (in1->size > in2_length)
320 return 1;
321
322 return strncasecmp(in1->buff, in2, in1->size);
323}

326 if (in1->size < in2->size)
327 return -1;
328 else if (in1->size > in2->size)
329 return 1;
330 else
331 return memcmp(in1->buff, in2->buff, in1->size);

347char* resolve_rel_url(char* base_url, char* rel_url) {
348 uri_type base;
349 uri_type rel;
350 int rv;
351 size_t len_rel;
352 size_t len_base;
353 size_t len;
354 char* out;
355 char* out_finger;
356 char* path;
357 size_t i;
358 size_t prefix;
359 std::string pathObj;
360
361 if ((!base_url || *base_url == '\0') && (!rel_url || *rel_url == '\0'))
362 return nullptr;
363 if (!base_url && rel_url)
364 return strdup(rel_url);
365 if (base_url && (!rel_url || *rel_url == '\0'))
366 return strdup(base_url);
367
368 len_rel = strlen(rel_url);
369 if (parse_uri(rel_url, len_rel, &rel) != HTTP_SUCCESS)
370 return NULL;
371
372 if (rel.type == Absolute)
373 return strdup(rel_url);
374
375 len_base = strlen(base_url);
376 if ((parse_uri(base_url, len_base, &base) != HTTP_SUCCESS) ||
377 (base.type != Absolute))
378 return NULL;
379
380 if (len_rel == (size_t)0)
381 return strdup(base_url);
382
383 len = len_base + len_rel + (size_t)2;
384 out = (char*)malloc(len);
385 if (out == NULL)
386 return NULL;
387 memset(out, 0, len);
388 out_finger = out;
389
390 /* scheme */
391 rv = snprintf(out_finger, len, "%.*s:", (int)base.scheme.size,
392 base.scheme.buff);
393 if (rv < 0 || rv >= (int)len)
394 goto error;
395 out_finger += rv;
396 len -= (size_t)rv;
397
398 /* authority */
399 if (rel.hostport.text.size > (size_t)0) {
400 rv = snprintf(out_finger, len, "%s", rel_url);
401 if (rv < 0 || rv >= (int)len)
402 goto error;
403 return out;
404 }
405 if (base.hostport.text.size > (size_t)0) {
406 rv = snprintf(out_finger, len, "//%.*s", (int)base.hostport.text.size,
407 base.hostport.text.buff);
408 if (rv < 0 || rv >= (int)len)
409 goto error;
410 out_finger += rv;
411 len -= (size_t)rv;
412 }
413
414 /* path */
415 path = out_finger;
416 if (rel.path_type == ABS_PATH) {
417 rv = snprintf(out_finger, len, "%s", rel_url);
418 } else if (base.pathquery.size == (size_t)0) {
419 rv = snprintf(out_finger, len, "/%s", rel_url);
420 } else {
421 if (rel.pathquery.size == (size_t)0) {
422 rv = snprintf(out_finger, len, "%.*s", (int)base.pathquery.size,
423 base.pathquery.buff);
424 } else {
425 if (len < base.pathquery.size)
426 goto error;
427 i = 0;
428 prefix = 1;
429 while (i < base.pathquery.size) {
430 out_finger[i] = base.pathquery.buff[i];
431 switch (base.pathquery.buff[i++]) {
432 case '/':
433 prefix = i;
434 /* fall-through */
435 default:
436 continue;
437 case '?': /* query */
438 if (rel.pathquery.buff[0] == '?')
439 prefix = --i;
440 }
441 break;
442 }
443 out_finger += prefix;
444 len -= prefix;
445 rv = snprintf(out_finger, len, "%.*s", (int)rel.pathquery.size,
446 rel.pathquery.buff);
447 }
448 if (rv < 0 || rv >= (int)len)
449 goto error;
450 out_finger += rv;
451 len -= (size_t)rv;
452
453 /* fragment */
454 if (rel.fragment.size > (size_t)0)
455 rv = snprintf(out_finger, len, "#%.*s", (int)rel.fragment.size,
456 rel.fragment.buff);
457 else if (base.fragment.size > (size_t)0)
458 rv = snprintf(out_finger, len, "#%.*s", (int)base.fragment.size,
459 base.fragment.buff);
460 else
461 rv = 0;
462 }
463 if (rv < 0 || rv >= (int)len)
464 goto error;
465 out_finger += rv;
466 len -= (size_t)rv;
467
468 pathObj = std::string(path, (size_t)(out_finger - path));
469 UPnPsdk::remove_dot_segments(pathObj);
470 pathObj.copy(path, pathObj.size());
471 path[pathObj.size()] = '\0';
472
473 return out;
474
475error:
476 free(out);
477 return NULL;
478}
