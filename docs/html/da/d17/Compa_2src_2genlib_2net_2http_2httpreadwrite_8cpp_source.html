103int Check_Connect_And_Wait_Connection(
104 const SOCKET a_sock,
105 const int connect_res
107) {
108 TRACE("Executing Check_Connect_And_Wait_Connection()")
109
110 if (connect_res == 0)
111 return 0;
112#ifdef _WIN32
113 if (umock::winsock2_h.WSAGetLastError() != WSAEWOULDBLOCK)
114 return -1;
115#else
116 if (errno != EINPROGRESS)
117 return -1;
118#endif
119
120 fd_set fdSet;
121 FD_ZERO(&fdSet);
122 FD_SET(a_sock, &fdSet);
123 timeval tmvTimeout = {DEFAULT_TCP_CONNECT_TIMEOUT, 0};
124 int result{SOCKET_ERROR};
125
126 result =
127 umock::sys_socket_h.select(a_sock + 1, NULL, &fdSet, NULL, &tmvTimeout);
128 switch (result) {
129 case SOCKET_ERROR:
130 return -1;
131 case 0:
132 /* timeout */
133 return -1;
134 }
135 int valopt{};
136 socklen_t len{sizeof(valopt)};
137 if (umock::sys_socket_h.getsockopt(a_sock, SOL_SOCKET, SO_ERROR,
138 (void*)&valopt, &len) < 0)
139 /* failed to read delayed error */
140 return -1;
141 if (valopt)
142 /* delayed error = valopt */
143 // TODO: Return more detailed error codes, e.g. valopt == 111:
144 // ECONNREFUSED "Connection refused" if there is a remote host but no
145 // server service listening.
146 return -1;
147
148 return 0;
149}

170int get_hoststr(
171 const char* url_str,
172 const char**
173 hoststr,
175 size_t* hostlen
177) {
178 TRACE("Executing get_hoststr()")
179 const char* start;
180 const char* finish;
181 int ret_code = UPNP_E_INVALID_URL;
182
183 // strlen() must be at least 2 char otherwise compiler complains boundary
184 // error with next strstr() statement.
185 if (strlen(url_str) < 2)
186 goto end_function;
187
188 start = strstr(url_str, "//");
189 if (!start)
190 goto end_function;
191
192 start += 2;
193 finish = strchr(start, '/');
194 if (finish) {
195 *hostlen = static_cast<size_t>(finish - start);
196 } else {
197 *hostlen = strlen(start);
198 }
199 *hoststr = start;
200
201 ret_code = UPNP_E_SUCCESS;
202
203end_function:
204 return ret_code;
205}

245int MakeGenericMessage(http_method_t method, const char* url_str,
246 membuffer* request, uri_type* url, int contentLength,
247 const char* contentType, const UpnpString* headers) {
248 int ret_code = 0;
249 size_t hostlen{};
250 const char* hoststr;
251
252 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__, "URL: %s method: %d\n",
253 url_str, method);
254 ret_code = http_FixStrUrl(url_str, strlen(url_str), url);
255 if (ret_code != UPNP_E_SUCCESS)
256 return ret_code;
257 /* make msg */
258 membuffer_init(request);
259 ret_code = http_MakeMessage(request, 1, 1, "Q", method, url->pathquery.buff,
260 url->pathquery.size);
261 /* add request headers if specified, otherwise use default headers */
262 if (ret_code == 0) {
263 if (headers) {
264 ret_code = http_MakeMessage(request, 1, 1, "s",
265 UpnpString_get_String(headers));
266 } else {
267 ret_code = get_hoststr(url_str, &hoststr, &hostlen);
268 if (ret_code != UPNP_E_SUCCESS)
269 return ret_code;
270 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__,
271 "HOSTNAME : %s Length : %" PRIzu "\n", hoststr, hostlen);
272 ret_code = http_MakeMessage(request, 1, 1,
273 "s"
274 "bcDCU",
275 "HOST: ", hoststr, hostlen);
276 }
277 }
278
279 /* add the content-type header */
280 if (ret_code == 0 && contentType) {
281 ret_code = http_MakeMessage(request, 1, 1, "T", contentType);
282 }
283 /* add content-length header. */
284 if (ret_code == 0) {
285 if (contentLength >= 0)
286 ret_code =
287 http_MakeMessage(request, 1, 1, "Nc", (off_t)contentLength);
288 else if (contentLength == UPNP_USING_CHUNKED)
289 ret_code = http_MakeMessage(request, 1, 1, "Kc");
290 else if (contentLength == UPNP_UNTIL_CLOSE)
291 ret_code = http_MakeMessage(request, 1, 1, "c");
292 else
293 ret_code = UPNP_E_INVALID_PARAM;
294 }
295 if (ret_code != 0) {
296 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__,
297 "HTTP Makemessage failed\n");
298 membuffer_destroy(request);
299 return ret_code;
300 }
301 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__,
302 "HTTP Buffer:\n%s\n"
303 "----------END--------\n",
304 request->buf);
305
306 return UPNP_E_SUCCESS;
307}

332int ReadResponseLineAndHeaders(
334 SOCKINFO* info,
336 http_parser_t* parser,
338 int* timeout_secs,
340 int* http_error_code) {
341 parse_status_t status;
342 int num_read;
343 char buf[2 * 1024];
344 int done = 0;
345 int ret_code = 0;
346
347 /*read response line */
348 status = parser_parse_responseline(parser);
349 switch (status) {
350 case PARSE_OK:
351 done = 1;
352 break;
353 case PARSE_INCOMPLETE:
354 done = 0;
355 break;
356 default:
357 /*error */
358 return status;
359 }
360 while (!done) {
361 num_read = sock_read(info, buf, sizeof(buf), timeout_secs);
362 if (num_read > 0) {
363 /* append data to buffer */
364 ret_code =
365 membuffer_append(&parser->msg.msg, buf, (size_t)num_read);
366 if (ret_code != 0) {
367 /* set failure status */
368 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
369 return PARSE_FAILURE;
370 }
371 status = parser_parse_responseline(parser);
372 switch (status) {
373 case PARSE_OK:
374 done = 1;
375 break;
376 case PARSE_INCOMPLETE:
377 done = 0;
378 break;
379 default:
380 /*error */
381 return status;
382 }
383 } else if (num_read == 0) {
384 /* partial msg */
385 *http_error_code = HTTP_BAD_REQUEST; /* or response */
386 return UPNP_E_BAD_HTTPMSG;
387 } else {
388 *http_error_code = parser->http_error_code;
389 return num_read;
390 }
391 }
392 status = parser_parse_headers(parser);
393 if ((status == (parse_status_t)PARSE_OK) &&
394 (parser->position == (parser_pos_t)POS_ENTITY))
395 done = 1;
396 else if (status == (parse_status_t)PARSE_INCOMPLETE)
397 done = 0;
398 else
399 /*error */
400 return status;
401 /*read headers */
402 while (!done) {
403 num_read = sock_read(info, buf, sizeof(buf), timeout_secs);
404 if (num_read > 0) {
405 /* append data to buffer */
406 ret_code =
407 membuffer_append(&parser->msg.msg, buf, (size_t)num_read);
408 if (ret_code != 0) {
409 /* set failure status */
410 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
411 return PARSE_FAILURE;
412 }
413 status = parser_parse_headers(parser);
414 if (status == (parse_status_t)PARSE_OK &&
415 parser->position == (parser_pos_t)POS_ENTITY)
416 done = 1;
417 else if (status == (parse_status_t)PARSE_INCOMPLETE)
418 done = 0;
419 else
420 /*error */
421 return status;
422 } else if (num_read == 0) {
423 /* partial msg */
424 *http_error_code = HTTP_BAD_REQUEST; /* or response */
425 return UPNP_E_BAD_HTTPMSG;
426 } else {
427 *http_error_code = parser->http_error_code;
428 return num_read;
429 }
430 }
431
432 return PARSE_OK;
433}

501static int private_connect(
502 const SOCKET sockfd,
503 const sockaddr* const
504 serv_addr,
505 const socklen_t addrlen
506) {
507 TRACE("Executing private_connect(), blocking " +
508 std::string(unblock_tcp_connections ? "false" : "true"))
509
510 if (unblock_tcp_connections) {
511 // This is never used due to SDK specification. It is only presevered
512 // for historical reasons.
513
514 int ret{SOCKET_ERROR};
515 // returns 0 if successful, else SOCKET_ERROR.
516 ret = umock::pupnp_sock.sock_make_no_blocking(sockfd);
517 if (ret == 0) {
518 // ret is needed for Check_Connect_And_Wait_Connection(),
519 // returns 0 if successful, else -1.
520 ret = umock::sys_socket_h.connect(sockfd, serv_addr, addrlen);
521 // returns 0 if successful, else -1.
522 ret = Check_Connect_And_Wait_Connection(sockfd, ret);
523
524 // Always make_blocking() to revert make_no_blocking() above.
525 // returns 0 if successful, else SOCKET_ERROR.
526 ret = ret | umock::pupnp_sock.sock_make_blocking(sockfd);
527 }
528
529 return ret == 0 ? 0 : SOCKET_ERROR;
530
531 } else { // tcp_connection is blocking
532
533 UPnPsdk::CSocketErr serrObj;
534 if (umock::sys_socket_h.connect(sockfd, serv_addr, addrlen) != 0) {
535 serrObj.catch_error();
536 UPnPsdk_LOGERR("MSG1020") "failed to connect() socket("
537 << sockfd << "): " << serrObj.error_str() << "\n";
538
539 return SOCKET_ERROR;
540 }
541
542 return 0;
543 }

558int http_FixUrl(uri_type* url, uri_type* fixed_url) {
559 const char* temp_path = "/";
560
561 *fixed_url = *url;
562#ifdef UPnPsdk_HAVE_OPENSSL
563 if (token_string_casecmp(&fixed_url->scheme, "http") != 0 &&
564 token_string_casecmp(&fixed_url->scheme, "https") != 0) {
565 return UPNP_E_INVALID_URL;
566 }
567#else
568 if (token_string_casecmp(&fixed_url->scheme, "http") != 0) {
569 return UPNP_E_INVALID_URL;
570 }
571#endif
572 if (fixed_url->hostport.text.size == (size_t)0) {
573 return UPNP_E_INVALID_URL;
574 }
575 /* set pathquery to "/" if it is empty */
576 if (fixed_url->pathquery.size == (size_t)0) {
577 fixed_url->pathquery.buff = temp_path;
578 fixed_url->pathquery.size = (size_t)1;
579 }
580
581 return UPNP_E_SUCCESS;
582}

628int http_RecvMessage(SOCKINFO* info, http_parser_t* parser,
629 http_method_t request_method, int* timeout_secs,
630 int* http_error_code) {
631 TRACE("Executing http_RecvMessage()")
632 int ret{UPNP_E_SUCCESS};
633 int line{};
634 parse_status_t status;
635 int num_read{};
636 int ok_on_close{};
637 char* buf;
638 size_t buf_len{1024};
639
640 *http_error_code = HTTP_INTERNAL_SERVER_ERROR;
641 buf = (char*)malloc(buf_len);
642 if (!buf) {
643 ret = UPNP_E_OUTOF_MEMORY;
644 goto ExitFunction;
645 }
646 if (request_method == (http_method_t)HTTPMETHOD_UNKNOWN) {
647 parser_request_init(parser);
648 } else {
649 parser_response_init(parser, request_method);
650 }
651
652 while (1) {
653 /* Double the bet if needed */
654 /* We should have already exited the loop if the value was <= 0
655 * so the cast is safe */
656 if ((size_t)num_read >= buf_len) {
657 free(buf);
658 buf_len = 2 * buf_len;
659 buf = (char*)malloc(buf_len);
660 if (!buf) {
661 ret = UPNP_E_OUTOF_MEMORY;
662 goto ExitFunction;
663 }
664 }
665 num_read = sock_read(info, buf, buf_len, timeout_secs);
666 if (num_read > 0) {
667 /* got data */
668 status = parser_append(parser, buf, (size_t)num_read);
669 switch (status) {
670 case PARSE_SUCCESS:
671 UPnPsdk_LOGINFO("MSG1031") "<<< (RECVD) <<<\n"
672 << parser->msg.msg.buf << "UPnPsdk -----------------\n";
673 if (g_maxContentLength > (size_t)0 &&
674 parser->content_length > (unsigned int)g_maxContentLength) {
675 *http_error_code = HTTP_REQ_ENTITY_TOO_LARGE;
676 line = __LINE__;
677 ret = UPNP_E_OUTOF_BOUNDS;
678 goto ExitFunction;
679 }
680 line = __LINE__;
681 ret = 0;
682 goto ExitFunction;
683 case PARSE_FAILURE:
684 case PARSE_NO_MATCH:
685 *http_error_code = parser->http_error_code;
686 line = __LINE__;
687 ret = UPNP_E_BAD_HTTPMSG;
688 goto ExitFunction;
689 case PARSE_INCOMPLETE_ENTITY:
690 /* read until close */
691 ok_on_close = 1;
692 break;
693 case PARSE_CONTINUE_1:
694 /* Web post request. */
695 line = __LINE__;
696 ret = PARSE_SUCCESS;
697 goto ExitFunction;
698 default:
699 break;
700 }
701 } else if (num_read == 0) {
702 UPnPsdk_LOGINFO("MSG1047") "<<< (RECVD) <<<\n"
703 << parser->msg.msg.buf << "\n-----------------\n";
704 if (ok_on_close) {
705 line = __LINE__;
706 ret = 0;
707 goto ExitFunction;
708 } else {
709 *http_error_code = HTTP_BAD_REQUEST; /* or response */
710 line = __LINE__;
711 UPnPsdk_LOGERR("MSG1005") "Line " << line
712 << ": partial message.\n";
713 ret = UPNP_E_BAD_HTTPMSG;
714 goto ExitFunction;
715 }
716 } else {
717 *http_error_code = parser->http_error_code;
718 line = __LINE__;
719 ret = num_read;
720 goto ExitFunction;
721 }
722 }
723
724ExitFunction:
725 free(buf);
726 if (ret != UPNP_E_SUCCESS) {
727 UPnPsdk_LOGERR("MSG1048") "error "
728 << ret << " on line " << line
729 << ", http_error_code=" << *http_error_code << ".\n";
730 }
731
732 return ret;
733}

736 TRACE("Executing http_SendMessage()")
737#ifdef COMPA_HAVE_WEBSERVER
738 SendInstruction* Instr{nullptr};
739 char* file_buf{nullptr};
740 char* ChunkBuf{nullptr};
741 /* 10 byte allocated for chunk header. */
742 size_t num_read;
743 off_t amount_to_be_read{};
744 size_t Data_Buf_Size{WEB_SERVER_BUF_SIZE};
745 int I_fmt_processed = 0;
746#endif /* COMPA_HAVE_WEBSERVER */

748 char c;
749 int RetVal = UPNP_E_SUCCESS;

751 UPnPsdk::SSockaddr local_saObj;

767#ifdef COMPA_HAVE_WEBSERVER

769 I_fmt_processed = 1;

785
786 } else if (c == 'f') {
787 /* file name */
788 char* filename = va_arg(argp, char*);
789 FILE* Fp{nullptr};
790
791 if (Instr && Instr->IsVirtualFile)
792 Fp = (FILE*)(virtualDirCallback.open)(
793 filename, UPNP_READ, Instr->Cookie, Instr->RequestCookie);
794 else
795#ifdef _WIN32
796 // returns error code direct
797 if (umock::stdio_h.fopen_s(&Fp, filename, "rb") != 0)
798 Fp = nullptr;
799#else
800 Fp = umock::stdio_h.fopen(filename, "rb");
801#endif
802 if (Fp == nullptr) {
803 RetVal = UPNP_E_FILE_READ_ERROR;
804 goto ExitFunction;
805 }
806 if (Instr && Instr->IsRangeActive && Instr->IsVirtualFile) {
807 if (virtualDirCallback.seek(Fp, Instr->RangeOffset, SEEK_CUR,
808 Instr->Cookie,
809 Instr->RequestCookie) != 0) {
810 RetVal = UPNP_E_FILE_READ_ERROR;
811 goto Cleanup_File;
812 }
813 } else if (Instr && Instr->IsRangeActive) {
814 if (fseeko(Fp, Instr->RangeOffset, SEEK_CUR) != 0) {
815 RetVal = UPNP_E_FILE_READ_ERROR;
816 goto Cleanup_File;
817 }
818 }
819 while (amount_to_be_read) {
820 if (Instr) {
821 size_t n = amount_to_be_read >= (off_t)Data_Buf_Size
822 ? Data_Buf_Size
823 : (size_t)amount_to_be_read;
824 if (Instr->IsVirtualFile) {
825 int nr = virtualDirCallback.read(Fp, file_buf, n,
826 Instr->Cookie,
827 Instr->RequestCookie);
828 if (nr < 0) {
829 RetVal = UPNP_E_FILE_READ_ERROR;
830 goto Cleanup_File;
831 }
832 // Type cast is save, 'nr' cannot be negative.
833 num_read = static_cast<size_t>(nr);
834 } else {
835 num_read = umock::stdio_h.fread(file_buf, 1u, n, Fp);
836 if (umock::stdio_h.ferror(Fp)) {
837 RetVal = UPNP_E_FILE_READ_ERROR;
838 umock::stdio_h.clearerr(Fp);
839 goto Cleanup_File;
840 }
841 }
842 amount_to_be_read -= (off_t)num_read;
843 if (Instr->ReadSendSize < 0) {
844 /* read until close */
845 amount_to_be_read = (off_t)Data_Buf_Size;
846 }
847 } else {
848 num_read = umock::stdio_h.fread(file_buf, (size_t)1,
849 Data_Buf_Size, Fp);
850 if (umock::stdio_h.ferror(Fp)) {
851 RetVal = UPNP_E_FILE_READ_ERROR;
852 umock::stdio_h.clearerr(Fp);
853 goto Cleanup_File;
854 }
855 }
856 int num_write;
857 if (num_read == 0u) {
858 /* EOF so no more to send. */
859 if (Instr && Instr->IsChunkActive) {
860 const char* str = "0\r\n\r\n";
861 num_write = sock_write(info, str, strlen(str), TimeOut);
862 } else {
863 RetVal = UPNP_E_FILE_READ_ERROR;
864 }
865 goto Cleanup_File;
866 }
867 /* Create chunk for the current buffer. */
868 if (Instr && Instr->IsChunkActive) {
869 /* Copy CRLF at the end of the chunk */
870 memcpy(file_buf + num_read, "\r\n", (size_t)2);
871 /* Hex length for the chunk size. */
872 char Chunk_Header[CHUNK_HEADER_SIZE]{};
873 int rc = snprintf(Chunk_Header, sizeof(Chunk_Header),
874 "%" PRIzx "\r\n", num_read);
875 if (rc < 0 ||
876 static_cast<unsigned int>(rc) >= sizeof(Chunk_Header)) {
877 RetVal = UPNP_E_INTERNAL_ERROR;
878 goto Cleanup_File;
879 }
880 /* Copy the chunk size header */
881 memcpy(file_buf - strlen(Chunk_Header), Chunk_Header,
882 strlen(Chunk_Header));
883 /* on the top of the buffer. */
884 /*file_buf[num_read+strlen(Chunk_Header)]
885 * = NULL; */
886 /*upnpprintf("Sending
887 * %s\n",file_buf-strlen(Chunk_Header));*/
888 num_write = sock_write(
889 info, file_buf - strlen(Chunk_Header),
890 num_read + strlen(Chunk_Header) + 2u, TimeOut);
891 size_t num_written = static_cast<size_t>(num_write);
892 if (num_write <= 0 ||
893 num_written != num_read + strlen(Chunk_Header) + 2u)
894 /* Send error nothing we can do. */
895 goto Cleanup_File;
896 } else {
897 /* write data */
898 num_write = sock_write(info, file_buf, num_read, TimeOut);
899
900 UPnPsdk_LOGINFO(
901 "MSG1104") ">>> (SENT) >>> UDevice response_out "
902 "from local \""
903 << local_saObj.netaddrp()
904 << "\" to (\"HOST:\" in following message) ...\n"
905 << std::string(file_buf, static_cast<size_t>(num_write))
906 << "\nUPnPsdk num_written=" << num_write
907 << ".\nUPnPsdk ------------\n";
908
909 /* Send error nothing we can do */
910 size_t num_written = static_cast<size_t>(num_write);
911 if (num_write <= 0 || num_written != num_read) {
912 goto Cleanup_File;
913 }
914 }
915 } /* while */
916
917 Cleanup_File:
918 if (Instr && Instr->IsVirtualFile) {
919 virtualDirCallback.close(Fp, Instr->Cookie,
920 Instr->RequestCookie);
921 } else {
922 umock::stdio_h.fclose(Fp);
923 }
924 goto ExitFunction;
925 } else
926#endif /* COMPA_HAVE_WEBSERVER */
927
928 if (c == 'b') {
929 /* Message to send is given in a memory buffer */
930 char* buf = va_arg(argp, char*);
931 size_t buf_length = va_arg(argp, size_t);
932 if (buf_length > 0u) {
933 // Write data.
934 int num_write = sock_write(info, buf, buf_length, TimeOut);
935
936 UPnPsdk_LOGINFO(
937 "MSG1105") ">>> (SENT) >>> UDevice response_out "
938 "from local \""
939 << local_saObj.netaddrp()
940 << "\" to (\"HOST:\" in following message) ...\n"
941 << std::string(buf, buf_length)
942 << "\nUPnPsdk buf_length=" << buf_length
943 << ", num_written=" << num_write
944 << ".\nUPnPsdk ------------\n";
945
946 if (num_write < 0) {
947 RetVal = num_write;
948 goto ExitFunction;
949 }
950 // 'num_write' may be negive in case of error. This will be
951 // casted to very big positive numbers that also never
952 // equals to buf_length, means the cast is save.
953 if (static_cast<size_t>(num_write) != buf_length) {
954 RetVal = UPNP_E_SOCKET_WRITE;
955 goto ExitFunction;
956 }
957 }
958 }
959 } // while
960
961ExitFunction:
962 va_end(argp);
963#ifdef COMPA_HAVE_WEBSERVER
964 free(ChunkBuf);
965#endif
966 return RetVal;
967}

1369int http_ReadHttpResponse(void* Handle, char* buf, size_t* size, int timeout) {
1370 http_connection_handle_t* handle = (http_connection_handle_t*)Handle;
1371 parse_status_t status;
1372 int num_read;
1373 int ok_on_close = 0;
1374 char tempbuf[2 * 1024];
1375 int ret_code = 0;
1376
1377 if (!handle || !size || (*size > 0 && !buf)) {
1378 if (size)
1379 *size = 0;
1380 return UPNP_E_INVALID_PARAM;
1381 }
1382 /* first parse what has already been gotten */
1383 if (handle->response.position != POS_COMPLETE)
1384 status = parser_parse_entity(&handle->response);
1385 else
1386 status = PARSE_SUCCESS;
1387 if (status == PARSE_INCOMPLETE_ENTITY)
1388 /* read until close */
1389 ok_on_close = 1;
1390 else if ((status != PARSE_SUCCESS) && (status != PARSE_CONTINUE_1) &&
1391 (status != PARSE_INCOMPLETE)) {
1392 /*error */
1393 *size = 0;
1394 return UPNP_E_BAD_RESPONSE;
1395 }
1396 /* read more if necessary entity */
1397 while (handle->response.msg.amount_discarded + *size >
1398 handle->response.msg.entity.length &&
1399 !handle->cancel && handle->response.position != POS_COMPLETE) {
1400 num_read =
1401 sock_read(&handle->sock_info, tempbuf, sizeof(tempbuf), &timeout);
1402 if (num_read > 0) {
1403 /* append data to buffer */
1404 ret_code = membuffer_append(&handle->response.msg.msg, tempbuf,
1405 (size_t)num_read);
1406 if (ret_code != 0) {
1407 /* set failure status */
1408 handle->response.http_error_code = HTTP_INTERNAL_SERVER_ERROR;
1409 *size = 0;
1410 return PARSE_FAILURE;
1411 }
1412 status = parser_parse_entity(&handle->response);
1413 if (status == PARSE_INCOMPLETE_ENTITY) {
1414 /* read until close */
1415 ok_on_close = 1;
1416 } else if ((status != PARSE_SUCCESS) &&
1417 (status != PARSE_CONTINUE_1) &&
1418 (status != PARSE_INCOMPLETE)) {
1419 /*error */
1420 *size = 0;
