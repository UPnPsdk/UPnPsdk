74void GenaAutoRenewSubscription(
76 void* input) {
77 job_arg* arg = (job_arg*)input;
78 UpnpEventSubscribe* sub_struct = (UpnpEventSubscribe*)arg->Event;
79 void* cookie;
80 Upnp_FunPtr callback_fun;
81 struct Handle_Info* handle_info;
82 int send_callback = 0;
83 int eventType = 0;
84 int timeout = 0;
85 int errCode = 0;
86
87 if constexpr (AUTO_RENEW_TIME == 0) {
88 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__, "GENA SUB EXPIRED");
89 UpnpEventSubscribe_set_ErrCode(sub_struct, UPNP_E_SUCCESS);
90 send_callback = 1;
91 eventType = UPNP_EVENT_SUBSCRIPTION_EXPIRED;
92 } else {
93 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__, "GENA AUTO RENEW");
94 timeout = UpnpEventSubscribe_get_TimeOut(sub_struct);
95 errCode = genaRenewSubscription(
96 arg->handle, UpnpEventSubscribe_get_SID(sub_struct), &timeout);
97 UpnpEventSubscribe_set_ErrCode(sub_struct, errCode);
98 UpnpEventSubscribe_set_TimeOut(sub_struct, timeout);
99 if (errCode != UPNP_E_SUCCESS && errCode != GENA_E_BAD_SID &&
100 errCode != GENA_E_BAD_HANDLE) {
101 send_callback = 1;
102 eventType = UPNP_EVENT_AUTORENEWAL_FAILED;
103 }
104 }
105
106 if (send_callback) {
107 HandleReadLock();
108 if (GetHandleInfo(arg->handle, &handle_info) != HND_CLIENT) {
109 HandleUnlock();
110 free_subscribe_arg(arg);
111 goto end_function;
112 }
113 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__, "HANDLE IS VALID");
114
115 /* make callback */
116 callback_fun = handle_info->Callback;
117 cookie = handle_info->Cookie;
118 HandleUnlock();
119 callback_fun((Upnp_EventType)eventType, arg->Event, cookie);
120 }
121
122 free_subscribe_arg(arg);
123
124end_function:
125 return;
126}

134int ScheduleGenaAutoRenew(
136 int client_handle,
138 int TimeOut,
140 GenlibClientSubscription* sub) {
141 UpnpEventSubscribe* RenewEvent = NULL;
142 job_arg* arg = NULL;
143 int return_code = GENA_SUCCESS;
144 ThreadPoolJob job;
145
146 memset(&job, 0, sizeof(job));
147
148 if (TimeOut == UPNP_INFINITE) {
149 return_code = GENA_SUCCESS;
150 goto end_function;
151 }
152
153 RenewEvent = UpnpEventSubscribe_new();
154 if (RenewEvent == NULL) {
155 return_code = UPNP_E_OUTOF_MEMORY;
156 goto end_function;
157 }
158
159 arg = (job_arg*)malloc(sizeof(job_arg));
160 if (arg == NULL) {
161 free(RenewEvent);
162 return_code = UPNP_E_OUTOF_MEMORY;
163 goto end_function;
164 }
165 memset(arg, 0, sizeof(job_arg));
166
167 /* schedule expire event */
168 UpnpEventSubscribe_set_ErrCode(RenewEvent, UPNP_E_SUCCESS);
169 UpnpEventSubscribe_set_TimeOut(RenewEvent, TimeOut);
170 UpnpEventSubscribe_set_SID(RenewEvent,
171 GenlibClientSubscription_get_SID(sub));
172 UpnpEventSubscribe_set_PublisherUrl(
173 RenewEvent, GenlibClientSubscription_get_EventURL(sub));
174
175 arg->handle = client_handle;
176 arg->Event = RenewEvent;
177
178 TPJobInit(&job, (start_routine)GenaAutoRenewSubscription, arg);
179 TPJobSetFreeFunction(&job, (free_routine)free_subscribe_arg);
180 TPJobSetPriority(&job, MED_PRIORITY);
181
182 /* Schedule the job */
183 return_code =
184 TimerThreadSchedule(&gTimerThread, TimeOut - AUTO_RENEW_TIME, REL_SEC,
185 &job, SHORT_TERM, &(arg->eventId));
186 if (return_code != UPNP_E_SUCCESS) {
187 free_subscribe_arg(arg);
188 goto end_function;
189 }
190
191 GenlibClientSubscription_set_RenewEventId(sub, arg->eventId);
192
193 return_code = GENA_SUCCESS;
194
195end_function:
196
197 return return_code;
198}

206int gena_unsubscribe(
208 const UpnpString* url,
210 const UpnpString* sid,
212 http_parser_t* response) {
213 int return_code;
214 uri_type dest_url;
215 membuffer request;
216
217 /* parse url */
218 return_code = http_FixStrUrl(UpnpString_get_String(url),
219 UpnpString_get_Length(url), &dest_url);
220 if (return_code != 0) {
221 return return_code;
222 }
223
224 /* make request msg */
225 membuffer_init(&request);
226 request.size_inc = 30;
227 return_code = http_MakeMessage(&request, 1, 1,
228 "q"
229 "ssc"
230 "Uc",
231 HTTPMETHOD_UNSUBSCRIBE, &dest_url,
232 "SID: ", UpnpString_get_String(sid));
233
234 /* Not able to make the message so destroy the existing buffer */
235 if (return_code != 0) {
236 membuffer_destroy(&request);
237
238 return return_code;
239 }
240
241 /* send request and get reply */
242 return_code = http_RequestAndResponse(
243 &dest_url, request.buf, request.length, HTTPMETHOD_UNSUBSCRIBE,
244 HTTP_DEFAULT_TIMEOUT, response);
245 membuffer_destroy(&request);
246 if (return_code != 0) {
247 httpmsg_destroy(&response->msg);
248 }
249
250 if (return_code == 0 && response->msg.status_code != HTTP_OK) {
251 return_code = UPNP_E_UNSUBSCRIBE_UNACCEPTED;
252 httpmsg_destroy(&response->msg);
253 }
254
255 return return_code;
256}

263int gena_subscribe(
265 const UpnpString* url,
267 int* timeout,
270 const UpnpString* renewal_sid,
272 UpnpString* sid) {
273 int return_code;
274 int parse_ret = 0;
275 int local_timeout = CP_MINIMUM_SUBSCRIPTION_TIME;
276 memptr sid_hdr;
277 memptr timeout_hdr;
278 char timeout_str[25];
279 membuffer request;
280 uri_type dest_url;
281 http_parser_t response;
282 int rc = 0;
283
284 UpnpString_clear(sid);
285
286 /* request timeout to string */
287 if (timeout == NULL) {
288 timeout = &local_timeout;
289 }
290 if (*timeout < 0) {
291 memset(timeout_str, 0, sizeof(timeout_str));
292 strncpy(timeout_str, "infinite", sizeof(timeout_str) - 1);
293 } else if (*timeout < CP_MINIMUM_SUBSCRIPTION_TIME) {
294 rc = snprintf(timeout_str, sizeof(timeout_str), "%d",
295 CP_MINIMUM_SUBSCRIPTION_TIME);
296 } else {
297 rc = snprintf(timeout_str, sizeof(timeout_str), "%d", *timeout);
298 }
299 if (rc < 0 || (unsigned int)rc >= sizeof(timeout_str))
300 return UPNP_E_OUTOF_MEMORY;
301
302 /* parse url */
303 return_code = http_FixStrUrl(UpnpString_get_String(url),
304 UpnpString_get_Length(url), &dest_url);
305 if (return_code != 0) {
306 return return_code;
307 }
308
309 /* make request msg */
310 membuffer_init(&request);
311 request.size_inc = 30;
312 if (renewal_sid) {
313 /* renew subscription */
314 return_code =
315 http_MakeMessage(&request, 1, 1,
316 "q"
317 "ssc"
318 "sscc",
319 HTTPMETHOD_SUBSCRIBE, &dest_url,
320 "SID: ", UpnpString_get_String(renewal_sid),
321 "TIMEOUT: Second-", timeout_str);
322 } else {
323 /* subscribe */
324 if (dest_url.hostport.IPaddress.ss_family == AF_INET6) {
325 struct sockaddr_in6* DestAddr6 =
326 (struct sockaddr_in6*)&dest_url.hostport.IPaddress;
327 return_code = http_MakeMessage(
328 &request, 1, 1,
329 "q"
330 "sssdsc"
331 "sc"
332 "sscc",
333 HTTPMETHOD_SUBSCRIBE, &dest_url, "CALLBACK: <http://[",
334 (IN6_IS_ADDR_LINKLOCAL(&DestAddr6->sin6_addr) ||
335 strlen(gIF_IPV6_ULA_GUA) == 0)
336 ? gIF_IPV6
337 : gIF_IPV6_ULA_GUA,
338 "]:",
339 (IN6_IS_ADDR_LINKLOCAL(&DestAddr6->sin6_addr) ||
340 strlen(gIF_IPV6_ULA_GUA) == 0)
341 ? LOCAL_PORT_V6
342 : LOCAL_PORT_V6_ULA_GUA,
343 "/>", "NT: upnp:event", "TIMEOUT: Second-", timeout_str);
344 } else {
345 return_code = http_MakeMessage(
346 &request, 1, 1,
347 "q"
348 "sssdsc"
349 "sc"
350 "sscc",
351 HTTPMETHOD_SUBSCRIBE, &dest_url, "CALLBACK: <http://", gIF_IPV4,
352 ":", LOCAL_PORT_V4, "/>", "NT: upnp:event", "TIMEOUT: Second-",
353 timeout_str);
354 }
355 }
356 if (return_code != 0) {
357 return return_code;
358 }
359
360 /* send request and get reply */
361 return_code = http_RequestAndResponse(&dest_url, request.buf,
362 request.length, HTTPMETHOD_SUBSCRIBE,
363 HTTP_DEFAULT_TIMEOUT, &response);
364 membuffer_destroy(&request);
365
366 if (return_code != 0) {
367 httpmsg_destroy(&response.msg);
368
369 return return_code;
370 }
371 if (response.msg.status_code != HTTP_OK) {
372 httpmsg_destroy(&response.msg);
373
374 return UPNP_E_SUBSCRIBE_UNACCEPTED;
375 }
376
377 /* get SID and TIMEOUT */
378 if (httpmsg_find_hdr(&response.msg, HDR_SID, &sid_hdr) == NULL ||
379 sid_hdr.length == 0 ||
380 httpmsg_find_hdr(&response.msg, HDR_TIMEOUT, &timeout_hdr) == NULL ||
381 timeout_hdr.length == 0) {
382 httpmsg_destroy(&response.msg);
383
384 return UPNP_E_BAD_RESPONSE;
385 }
386
387 /* save timeout */
388 parse_ret =
389 matchstr(timeout_hdr.buf, timeout_hdr.length, "%iSecond-%d%0", timeout);
390 if (parse_ret == PARSE_OK) {
391 /* nothing to do */
392 } else if (memptr_cmp_nocase(&timeout_hdr, "Second-infinite") == 0) {
393 *timeout = -1;
394 } else {
395 httpmsg_destroy(&response.msg);
396
397 return UPNP_E_BAD_RESPONSE;
398 }
399
400 /* save SID */
401 UpnpString_set_StringN(sid, sid_hdr.buf, sid_hdr.length);
402 if (UpnpString_get_String(sid) == NULL) {
403 httpmsg_destroy(&response.msg);
404
405 return UPNP_E_OUTOF_MEMORY;
406 }
407 httpmsg_destroy(&response.msg);
408
409 return UPNP_E_SUCCESS;
410}

414int genaUnregisterClient(UpnpClient_Handle client_handle) {
415 GenlibClientSubscription* sub_copy = GenlibClientSubscription_new();
416 int return_code = UPNP_E_SUCCESS;
417 struct Handle_Info* handle_info = NULL;
418 http_parser_t response;
419
420 while (1) {
421 HandleLock();
422
423 if (GetHandleInfo(client_handle, &handle_info) != HND_CLIENT) {
424 HandleUnlock();
425 return_code = GENA_E_BAD_HANDLE;
426 goto exit_function;
427 }
428 if (handle_info->ClientSubList == NULL) {
429 return_code = UPNP_E_SUCCESS;
430 break;
431 }
432 GenlibClientSubscription_assign(sub_copy, handle_info->ClientSubList);
433 RemoveClientSubClientSID(&handle_info->ClientSubList,
434 GenlibClientSubscription_get_SID(sub_copy));
435
436 HandleUnlock();
437
438 return_code = gena_unsubscribe(
439 GenlibClientSubscription_get_EventURL(sub_copy),
440 GenlibClientSubscription_get_ActualSID(sub_copy), &response);
441 if (return_code == 0) {
442 httpmsg_destroy(&response.msg);
443 }
444 free_client_subscription(sub_copy);
445 }
446
447 freeClientSubList(handle_info->ClientSubList);
448 HandleUnlock();
449
450exit_function:
451 GenlibClientSubscription_delete(sub_copy);
452 return return_code;
453}

500int genaSubscribe(UpnpClient_Handle client_handle,
501 const UpnpString* PublisherURL, int* TimeOut,
502 UpnpString* out_sid) {
503 int return_code = GENA_SUCCESS;
504 GenlibClientSubscription* newSubscription = GenlibClientSubscription_new();
505 uuid_upnp uid;
506 Upnp_SID temp_sid;
507 Upnp_SID temp_sid2;
508 UpnpString* ActualSID = UpnpString_new();
509 UpnpString* EventURL = UpnpString_new();
510 struct Handle_Info* handle_info;
511 int rc = 0;
512
513 memset(temp_sid, 0, sizeof(temp_sid));
514 memset(temp_sid2, 0, sizeof(temp_sid2));
515
516 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__, "GENA SUBSCRIBE BEGIN");
517
518 UpnpString_clear(out_sid);
519
520 HandleReadLock();
521 /* validate handle */
522 if (GetHandleInfo(client_handle, &handle_info) != HND_CLIENT) {
523 return_code = GENA_E_BAD_HANDLE;
524 SubscribeLock();
525 goto error_handler;
526 }
527 HandleUnlock();
528
529 /* subscribe */
530 SubscribeLock();
531 return_code = gena_subscribe(PublisherURL, TimeOut, NULL, ActualSID);
532 HandleLock();
533 if (return_code != UPNP_E_SUCCESS) {
534 UpnpPrintf(UPNP_CRITICAL, GENA, __FILE__, __LINE__,
535 "SUBSCRIBE FAILED in transfer error code: %d "
536 "returned\n",
537 return_code);
538 goto error_handler;
539 }
540
541 if (GetHandleInfo(client_handle, &handle_info) != HND_CLIENT) {
542 return_code = GENA_E_BAD_HANDLE;
543 goto error_handler;
544 }
545
546 /* generate client SID */
547 uuid_create(&uid);
548 upnp_uuid_unpack(&uid, temp_sid);
549 rc = snprintf(temp_sid2, sizeof(temp_sid2), "uuid:%s", temp_sid);
550 if (rc < 0 || (unsigned int)rc >= sizeof(temp_sid2)) {
551 return_code = UPNP_E_OUTOF_MEMORY;
552 goto error_handler;
553 }
554 UpnpString_set_String(out_sid, temp_sid2);
555
556 /* create event url */
557 UpnpString_assign(EventURL, PublisherURL);
558
559 /* fill subscription */
560 if (newSubscription == NULL) {
561 return_code = UPNP_E_OUTOF_MEMORY;
562 goto error_handler;
563 }
564 GenlibClientSubscription_set_RenewEventId(newSubscription, -1);
565 GenlibClientSubscription_set_SID(newSubscription, out_sid);
566 GenlibClientSubscription_set_ActualSID(newSubscription, ActualSID);
567 GenlibClientSubscription_set_EventURL(newSubscription, EventURL);
568 GenlibClientSubscription_set_Next(newSubscription,
569 handle_info->ClientSubList);
570 handle_info->ClientSubList = newSubscription;
571
572 /* schedule expiration event */
573 return_code =
574 ScheduleGenaAutoRenew(client_handle, *TimeOut, newSubscription);
575
576error_handler:
577 UpnpString_delete(ActualSID);
578 UpnpString_delete(EventURL);
579 if (return_code != UPNP_E_SUCCESS)
580 GenlibClientSubscription_delete(newSubscription);
581 HandleUnlock();
582 SubscribeUnlock();
583
584 return return_code;
585}

587int genaRenewSubscription(UpnpClient_Handle client_handle,
588 const UpnpString* in_sid, int* TimeOut) {
589 int return_code = GENA_SUCCESS;
590 GenlibClientSubscription* sub = NULL;
591 GenlibClientSubscription* sub_copy = GenlibClientSubscription_new();
592 struct Handle_Info* handle_info;
593 UpnpString* ActualSID = UpnpString_new();
594 ThreadPoolJob tempJob;
595
596 HandleLock();
597
598 /* validate handle and sid */
599 if (GetHandleInfo(client_handle, &handle_info) != HND_CLIENT) {
600 HandleUnlock();
601
602 return_code = GENA_E_BAD_HANDLE;
603 goto exit_function;
604 }
605
606 sub = GetClientSubClientSID(handle_info->ClientSubList, in_sid);
607 if (sub == NULL) {
608 HandleUnlock();
609
610 return_code = GENA_E_BAD_SID;
611 goto exit_function;
612 }
613
614 /* remove old events */
615 if (TimerThreadRemove(&gTimerThread,
616 GenlibClientSubscription_get_RenewEventId(sub),
617 &tempJob) == 0) {
618 tempJob.free_func(tempJob.arg);
619 }
620
621 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__,
622 "REMOVED AUTO RENEW EVENT");
623
624 GenlibClientSubscription_set_RenewEventId(sub, -1);
625 GenlibClientSubscription_assign(sub_copy, sub);
626
627 HandleUnlock();
628
629 return_code = gena_subscribe(
630 GenlibClientSubscription_get_EventURL(sub_copy), TimeOut,
631 GenlibClientSubscription_get_ActualSID(sub_copy), ActualSID);
632
633 HandleLock();
634
635 if (GetHandleInfo(client_handle, &handle_info) != HND_CLIENT) {
636 HandleUnlock();
637 return_code = GENA_E_BAD_HANDLE;
638 goto exit_function;
639 }
640
641 /* we just called GetHandleInfo, so we don't check for return value */
642 /*GetHandleInfo(client_handle, &handle_info); */
643 if (return_code != UPNP_E_SUCCESS) {
644 /* network failure (remove client sub) */
645 RemoveClientSubClientSID(&handle_info->ClientSubList, in_sid);
646 free_client_subscription(sub_copy);
647 HandleUnlock();
648 goto exit_function;
649 }
650
651 /* get subscription */
652 sub = GetClientSubClientSID(handle_info->ClientSubList, in_sid);
653 if (sub == NULL) {
654 free_client_subscription(sub_copy);
655 HandleUnlock();
656 return_code = GENA_E_BAD_SID;
657 goto exit_function;
658 }
659
660 /* store actual sid */
661 GenlibClientSubscription_set_ActualSID(sub, ActualSID);
662
663 /* start renew subscription timer */
664 return_code = ScheduleGenaAutoRenew(client_handle, *TimeOut, sub);
665 if (return_code != GENA_SUCCESS) {
666 RemoveClientSubClientSID(&handle_info->ClientSubList,
667 GenlibClientSubscription_get_SID(sub));
668 }
669 free_client_subscription(sub_copy);
670 HandleUnlock();
671
672exit_function:
673 UpnpString_delete(ActualSID);
674 GenlibClientSubscription_delete(sub_copy);
675 return return_code;
676}

678void gena_process_notification_event(SOCKINFO* info, http_message_t* event) {
679 UpnpEvent* event_struct = UpnpEvent_new();
680 IXML_Document* ChangedVars = NULL;
681 int eventKey;
682 token sid;
683 GenlibClientSubscription* subscription = NULL;
684 struct Handle_Info* handle_info;
685 void* cookie;
686 Upnp_FunPtr callback;
687 UpnpClient_Handle client_handle;
688 UpnpClient_Handle client_handle_start;
689 int err_ret = HTTP_PRECONDITION_FAILED;
690
691 memptr sid_hdr;
692 memptr nt_hdr, nts_hdr;
693 memptr seq_hdr;
694
695 /* get SID */
696 if (httpmsg_find_hdr(event, HDR_SID, &sid_hdr) == NULL) {
697 error_respond(info, HTTP_PRECONDITION_FAILED, event);
698 goto exit_function;
699 }
700 sid.buff = sid_hdr.buf;
701 sid.size = sid_hdr.length;
702
703 /* get event key */
704 if (httpmsg_find_hdr(event, HDR_SEQ, &seq_hdr) == NULL ||
705 matchstr(seq_hdr.buf, seq_hdr.length, "%d%0", &eventKey) != PARSE_OK) {
706 error_respond(info, HTTP_BAD_REQUEST, event);
707 goto exit_function;
708 }
709
710 /* get NT and NTS headers */
711 if (httpmsg_find_hdr(event, HDR_NT, &nt_hdr) == NULL ||
712 httpmsg_find_hdr(event, HDR_NTS, &nts_hdr) == NULL) {
713 error_respond(info, HTTP_BAD_REQUEST, event);
714 goto exit_function;
715 }
716
717 /* verify NT and NTS headers */
718 if (memptr_cmp(&nt_hdr, "upnp:event") != 0 ||
719 memptr_cmp(&nts_hdr, "upnp:propchange") != 0) {
720 error_respond(info, HTTP_PRECONDITION_FAILED, event);
721 goto exit_function;
722 }
723
724 /* parse the content (should be XML) */
725 if (!has_xml_content_type(event) || event->msg.length == 0 ||
726 ixmlParseBufferEx(event->entity.buf, &ChangedVars) != IXML_SUCCESS) {
727 error_respond(info, HTTP_BAD_REQUEST, event);
728 goto exit_function;
729 }
730
731 HandleLock();
732
733 /* get client info */
734 if (GetClientHandleInfo(&client_handle_start, &handle_info) != HND_CLIENT) {
735 error_respond(info, HTTP_PRECONDITION_FAILED, event);
736 HandleUnlock();
737 goto exit_function;
738 }
739
740 HandleUnlock();
741
742 for (client_handle = client_handle_start; client_handle < NUM_HANDLE;
743 client_handle++) {
744 HandleLock();
745
746 /* get client info */
747 if (GetHandleInfo(client_handle, &handle_info) != HND_CLIENT) {
748 HandleUnlock();
749 continue;
750 }
751
752 /* get subscription based on SID */
753 subscription = GetClientSubActualSID(handle_info->ClientSubList, &sid);
754 if (subscription == NULL) {
755 if (eventKey == 0) {
756 /* wait until we've finished processing a
757 * subscription */
758 /* (if we are in the middle) */
759 /* this is to avoid mistakenly rejecting the
760 * first event if we */
761 /* receive it before the subscription response
