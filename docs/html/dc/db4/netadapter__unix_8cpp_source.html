27void CNetadapter::load() {
28 TRACE2(this, " Executing load()")
29
30 // Get system adapters addresses.
31 this->free_ifaddrs();
32 if (umock::ifaddrs_h.getifaddrs(&m_ifa_first) != 0) {
33 throw std::runtime_error(
34 UPnPsdk_LOGEXCEPT +
35 "MSG1119: Failed to get information from the network adapters: " +
36 std::string(std::strerror(errno)) + '\n');
37 }
38
39 for (m_ifa_current = m_ifa_first; m_ifa_current != nullptr;
40 m_ifa_current = m_ifa_current->ifa_next) {
41 if (is_valid_if())
42 break; // Does not increment m_ifa_current
43 }
44}

117inline bool CNetadapter::is_valid_if() const noexcept {
118 // Accept IFF_LOOPBACK or up AF_INET6/AF_INET interfaces with address
119 // (e.g. not bonded) and that support MULTICAST.
120 if (m_ifa_current == nullptr || m_ifa_current->ifa_addr == nullptr)
121 return false;
122
123 if (m_ifa_current->ifa_flags & IFF_LOOPBACK &&
124 (m_ifa_current->ifa_addr->sa_family == AF_INET6 ||
125 m_ifa_current->ifa_addr->sa_family == AF_INET))
126 return true;
127
128 if ((m_ifa_current->ifa_addr->sa_family == AF_INET6 ||
129 m_ifa_current->ifa_addr->sa_family == AF_INET) &&
130 m_ifa_current->ifa_flags & IFF_UP &&
131 m_ifa_current->ifa_flags & IFF_MULTICAST)
132 return true;
133
134 return false;
135}
