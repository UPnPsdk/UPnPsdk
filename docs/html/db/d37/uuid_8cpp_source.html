62inline void format_uuid_v1(uuid_upnp* uid, uint16_t clock_seq,
63 uuid_time_t timestamp, uuid_node_t node) {
64 /* Construct a version 1 uuid with the information we've gathered
65 * plus a few constants. */
66 uid->time_low = (uint32_t)(timestamp & 0xFFFFFFFF);
67 uid->time_mid = (uint16_t)((timestamp >> 32) & 0xFFFF);
68 uid->time_hi_and_version = (uint16_t)((timestamp >> 48) & 0x0FFF);
69 uid->time_hi_and_version |= (1 << 12);
70 uid->clock_seq_low = (uint8_t)(clock_seq & 0xFF);
71 uid->clock_seq_hi_and_reserved = (uint8_t)((clock_seq & 0x3F00) >> 8);
72 uid->clock_seq_hi_and_reserved |= 0x80;
73 memcpy(&uid->node, &node, sizeof uid->node);
74}

97inline void write_state(uint16_t clockseq, uuid_time_t timestamp,
98 uuid_node_t node) {
99 static uuid_time_t next_save;
100
101 if (!stateInited) {
102 next_save = timestamp;
103 stateInited = 1;
104 };
105 /* always save state to volatile shared state. */
106 st.cs = clockseq;
107 st.ts = timestamp;
108 st.node = node;
109 if (timestamp >= next_save) {
110 /* schedule next save for 10 seconds from now. */
111 next_save = timestamp + (10 * 10 * 1000 * 1000);
112 };
113}

120inline void get_current_time(uuid_time_t* timestamp) {
121 uuid_time_t time_now;
122 static uuid_time_t time_last;
123 static uint16_t uuids_this_tick;
124 static int inited = 0;
125
126 if (!inited) {
127 uuids_this_tick = UUIDS_PER_TICK;
128 inited = 1;
129 };
130 while (1) {
131 get_system_time(&time_now);
132 /* if clock reading changed since last UUID generated... */
133 if (time_last != time_now) {
134 /* reset count of uuids gen'd with this clock reading.
135 */
136 uuids_this_tick = 0;
137 break;
138 };
139 if (uuids_this_tick < UUIDS_PER_TICK) {
140 uuids_this_tick++;
141 break;
142 };
143 /* going too fast for our clock; spin. */
144 };
145 /* add the count of uuids to low order bits of the clock reading. */
146 *timestamp = time_now + uuids_this_tick;
147 time_last = *timestamp;
148}

154inline uint16_t true_random() {
155 static int inited = 0;
156 uuid_time_t time_now;
157
158 if (!inited) {
159 get_system_time(&time_now);
160 time_now = time_now / UUIDS_PER_TICK;
161 srand((unsigned int)(((time_now >> 32) ^ time_now) & 0xffffffff));
162 inited = 1;
163 };
164
165 return (uint16_t)(rand());
166}

171inline void format_uuid_v3(uuid_upnp* uid, unsigned char hash[16]) {
172 /* Construct a version 3 uuid with the (pseudo-)random number plus a few
173 * constants. */
174 memcpy(uid, hash, sizeof(uuid_upnp));
175 /* convert UUID to local byte order. */
176 uid->time_low = ntohl(uid->time_low);
177 uid->time_mid = ntohs(uid->time_mid);
178 uid->time_hi_and_version = ntohs(uid->time_hi_and_version);
179 /* put in the variant and version bits. */
180 uid->time_hi_and_version &= 0x0FFF;
181 uid->time_hi_and_version |= (3 << 12);
182 uid->clock_seq_hi_and_reserved &= 0x3F;
183 uid->clock_seq_hi_and_reserved |= 0x80;
184}

194int uuid_create(uuid_upnp* uid) {
195 uuid_time_t timestamp;
196 uuid_time_t last_time;
197 uint16_t clockseq;
198 uuid_node_t node;
199 uuid_node_t last_node;
200 int f;
201
202 /* acquire system wide lock so we're alone. */
203 UUIDLock();
204 /* get current time. */
205 get_current_time(&timestamp);
206 /* get node ID. */
207 get_ieee_node_identifier(&node);
208 /* get saved state from NV storage. */
209 f = read_state(&clockseq, &last_time, &last_node);
210 /* if no NV state, or if clock went backwards, or node ID changed
211 * (e.g., net card swap) change clockseq. */
212 if (!f || memcmp(&node, &last_node, sizeof(uuid_node_t)))
213 clockseq = true_random();
214 else if (timestamp < last_time)
215 clockseq++;
216 /* stuff fields into the UUID. */
217 format_uuid_v1(uid, clockseq, timestamp, node);
218 /* save the state for next time. */
219 write_state(clockseq, timestamp, node);
220 UUIDUnlock();
221
222 return 1;
223}

233void uuid_create_from_name(
235 uuid_upnp* uid,
238 uuid_upnp nsid,
240 void* name,
242 int namelen) {
243 MD5_CTX c;
244 unsigned char hash[16];
245 uuid_upnp net_nsid; /* context UUID in network byte order */
246
247 /* put name space ID in network byte order so it hashes the same no
248 * matter what endian machine we're on. */
249 net_nsid = nsid;
250 net_nsid.time_low = htonl(net_nsid.time_low);
251 net_nsid.time_mid = htons(net_nsid.time_mid);
252 net_nsid.time_hi_and_version = htons(net_nsid.time_hi_and_version);
253 MD5Init(&c);
254 MD5Update(&c, (unsigned char*)&net_nsid, sizeof(uuid_upnp));
255 MD5Update(&c, name, (unsigned int)namelen);
256 MD5Final(hash, &c);
257 /* the hash is in network byte order at this point. */
258 format_uuid_v3(uid, hash);
259}
