18// Free function to get the address string from a sockaddr structure
19// -----------------------------------------------------------------
29std::string to_netaddr(const ::sockaddr_storage* const a_sockaddr) noexcept {
30 // TRACE("Executing to_netaddr()") // not usable in chained output.
31 char addrbuf[INET6_ADDRSTRLEN]{};
32
33 switch (a_sockaddr->ss_family) {
34 case AF_UNSPEC:
35 return "";
36
37 // There is no need to test the return value of ::inet_ntop() because its
38 // two possible errors are implicit managed by this code.
39 case AF_INET6: {
40 const ::sockaddr_in6* sin6 =
41 reinterpret_cast<const ::sockaddr_in6*>(a_sockaddr);
42 ::inet_ntop(AF_INET6, sin6->sin6_addr.s6_addr, addrbuf,
43 sizeof(addrbuf));
44 // Next throws 'std::length_error' if the length of the constructed
45 // string would exceed max_size(). This should never happen with given
46 // length of addrbuf.
47 if (sin6->sin6_scope_id > 0)
48 return '[' + std::string(addrbuf) + '%' +
49 std::to_string(sin6->sin6_scope_id) + ']';
50 return '[' + std::string(addrbuf) + ']';
51 }
52 case AF_INET:
53 ::inet_ntop(AF_INET,
54 &reinterpret_cast<const ::sockaddr_in*>(a_sockaddr)
55 ->sin_addr.s_addr,
56 addrbuf, sizeof(addrbuf));
57 // Next throws 'std::length_error' if the length of the constructed
58 // string would exceed max_size(). This should never happen with given
59 // length of addrbuf.
60 return std::string(addrbuf);
61
62 default:
63 UPNPLIB_LOGERR "MSG1036: Unsupported address family "
64 << static_cast<int>(a_sockaddr->ss_family) << ".\n";
65 }
66 return "";
67}

99// Free function to logical compare two sockaddr structures
100// --------------------------------------------------------
111bool sockaddrcmp(const ::sockaddr_storage* a_ss1,
112 const ::sockaddr_storage* a_ss2) noexcept {
113 // To have a logical equal socket address we compare the address family, the
114 // ip address and the port.
115 // Throws no exception.
116 if (a_ss1 == nullptr && a_ss2 == nullptr)
117 return true;
118 if (a_ss1 == nullptr || a_ss2 == nullptr)
119 return false;
120
121 switch (a_ss1->ss_family) {
122 case AF_UNSPEC:
123 if (a_ss2->ss_family != AF_UNSPEC)
124 return false;
125 break;
126
127 case AF_INET6: {
128 // We compare ipv6 addresses which are stored in a 16 byte array
129 // (unsigned char s6_addr[16]). So we have to use memcmp() for
130 // comparison.
131 const unsigned char* const s6_addr1 =
132 reinterpret_cast<const ::sockaddr_in6*>(a_ss1)->sin6_addr.s6_addr;
133 const unsigned char* const s6_addr2 =
134 reinterpret_cast<const ::sockaddr_in6*>(a_ss2)->sin6_addr.s6_addr;
135
136 if (a_ss2->ss_family != AF_INET6 ||
137 ::memcmp(s6_addr1, s6_addr2, sizeof(in6_addr)) != 0 ||
138 reinterpret_cast<const ::sockaddr_in6*>(a_ss1)->sin6_port !=
139 reinterpret_cast<const ::sockaddr_in6*>(a_ss2)->sin6_port)
140 return false;
141 } break;
142
143 case AF_INET:
144 if (a_ss2->ss_family != AF_INET ||
145 reinterpret_cast<const ::sockaddr_in*>(a_ss1)->sin_addr.s_addr !=
146 reinterpret_cast<const ::sockaddr_in*>(a_ss2)
147 ->sin_addr.s_addr ||
148 reinterpret_cast<const ::sockaddr_in*>(a_ss1)->sin_port !=
149 reinterpret_cast<const ::sockaddr_in*>(a_ss2)->sin_port)
150 return false;
151 break;
152
153 default:
154 return false;
155 }
156
157 return true;
158}

163// Free function to get the address string with port from a sockaddr structure
164// ---------------------------------------------------------------------------
165std::string to_netaddrp(const ::sockaddr_storage* const a_sockaddr) noexcept {
166 // TRACE("Executing to_addrport_str()") // not usable in chained output.
167 //
168 // sin_port and sin6_port are on the same memory location (union of the
169 // structures) so I can use it for AF_INET and AF_INET6.
170 // 'std::to_string()' may throw 'std::bad_alloc' from the std::string
171 // constructor. It is a fatal error that violates the promise to noexcpt
172 // and immediately terminates the propgram. This is intentional because the
173 // error cannot be handled.
174 return (a_sockaddr->ss_family != AF_INET6 &&
175 a_sockaddr->ss_family != AF_INET)
176 ? to_netaddr(a_sockaddr) // let it handle the error.
177 : to_netaddr(a_sockaddr) + ":" +
178 std::to_string(ntohs(
179 reinterpret_cast<const ::sockaddr_in6*>(a_sockaddr)
180 ->sin6_port));
181}

221// Assignment operator= to set socket address from string,
222// -------------------------------------------------------
223// For port conversion:
224// Don't use '::htons' (with colons) instead of 'htons', MacOS don't like it.
225// 'sin6_port' is also 'sin_port' due to union.
226void SSockaddr::operator=(const std::string& a_addr_str) {
227 // Valid input examles: "[2001:db8::1]", "[2001:db8::1]:50001",
228 // "192.168.1.1", "192.168.1.1:50001".
229 TRACE2(this, " Executing SSockaddr::operator=()")
230
231 // An empty address string clears the address storage.
232 if (a_addr_str.empty()) {
233 ::memset(&m_sa_union, 0, sizeof(m_sa_union));
234 return;
235 }
236
237 if (a_addr_str.front() == '[') { // IPv6 address
238 if (a_addr_str.back() == ']') { // IPv6 address without port
239 this->handle_ipv6(a_addr_str);
240
241 } else { // IPv6 with port
242 // Split address and port
243 size_t pos = a_addr_str.find("]:");
244 this->handle_ipv6(a_addr_str.substr(0, pos + 1));
245 sin6.sin6_port = htons(to_port(a_addr_str.substr(pos + 2)));
246 }
247 } else { // IPv4 address or port
248 size_t pos = a_addr_str.find_first_of(':');
249 if (pos == 0 && a_addr_str.size() > 1 && ss.ss_family != AF_UNSPEC) {
250 // Special case: only port with leading ':' is valid to change port
251 // of existing socket address.
252 sin6.sin6_port = htons(to_port(a_addr_str.substr(pos + 1)));
253
254 } else if (pos != std::string::npos) { // ':' found means ipv4 with port
255 this->handle_ipv4(a_addr_str.substr(0, pos));
256 sin6.sin6_port = htons(to_port(a_addr_str.substr(pos + 1)));
257
258 } else { // IPv4 without port or port only
259 if (a_addr_str.find_first_of(".") != std::string::npos) {
260 this->handle_ipv4(a_addr_str); // IPv4 address without port
261
262 } else { // port only
263 sin6.sin6_port = htons(to_port(a_addr_str));
264 }
265 }
266 }

344 TRACE2(this, " Executing SSockaddr::handle_ipv6()")
345 // remove surounding brackets
346 std::string addr_str = a_addr_str.substr(1, a_addr_str.length() - 2);

351 "MSG1043: Invalid netaddress \"" +
352 a_addr_str + "\".");
