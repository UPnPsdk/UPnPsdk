47int main(int argc, char* argv[]) {
48 TRACE("Executing main()");

55
56 SampleUtil_Initialize(linux_print);
57 /* Parse options */
58 for (int i = 1; i < argc; i++) {
59 if (strcmp(argv[i], "-i") == 0) {
60 iface = argv[++i];
61 } else if (strcmp(argv[i], "-help") == 0) {
62 SampleUtil_Print("Usage: %s -i interface -help (this message)\n",
63 argv[0]);
64 SampleUtil_Print(
65 "\tinterface: interface address of the control point\n"
66 "\t\te.g.: eth0\n");
67 return 1;
68 }
69 }
70
71 // Start TV device and the control point
72 int rc_dev = device_main(argc, argv);
73 if (rc_dev != UPNP_E_SUCCESS) {
74 SampleUtil_Print("Error starting UPnP TV Device\n");
75 return rc_dev;
76 }
77 int rc_cpt = TvCtrlPointStart(iface, NULL, 1);
78 if (rc_cpt != TV_SUCCESS) {
79 SampleUtil_Print("Error starting UPnP TV Control Point\n");
80 return rc_cpt;
81 }
82
83 /* start a command loop thread */
84 if (pthread_create(&cmdloop_thread, NULL, TvCtrlPointCommandLoop, NULL) !=
85 0) {
86 return UPNP_E_INTERNAL_ERROR;
87 }
88#ifdef _MSC_VER
89 pthread_join(cmdloop_thread, NULL);
90#else
91 /* Catch Ctrl-C and properly shutdown */
92 sigemptyset(&sigs_to_catch);
93 sigaddset(&sigs_to_catch, SIGINT);
94 sigwait(&sigs_to_catch, &sig);
95 SampleUtil_Print("Shutting down on signal %d...\n", sig);
96#endif
97
98 rc_dev = TvDeviceStop();
99 rc_cpt = TvCtrlPointStop();
100 if (rc_dev != UPNP_E_SUCCESS)
101 return rc_dev;
102 return rc_cpt;
103}
