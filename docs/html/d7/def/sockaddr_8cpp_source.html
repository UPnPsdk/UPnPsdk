17// Free function to get the address string from a sockaddr structure
18// -----------------------------------------------------------------
28std::string to_netaddr(const ::sockaddr_storage& a_sockaddr) noexcept {
29 // TRACE("Executing to_netaddr()") // not usable in chained output.
30
31 // Accept nameinfo only for supported address families.
32 switch (a_sockaddr.ss_family) {
33 case AF_INET6:
34 case AF_INET:
35 break;
36 case AF_UNSPEC:
37 return "";
38 default:
39 UPnPsdk_LOGERR "MSG1129: Unsupported address family "
40 << std::to_string(a_sockaddr.ss_family)
41 << ". Continue with empty netaddress \"\".\n";
42 return "";
43 }
44
45 char addrStr[INET6_ADDRSTRLEN]{};
46 int ret = ::getnameinfo(reinterpret_cast<const sockaddr*>(&a_sockaddr),
47 sizeof(a_sockaddr), addrStr, sizeof(addrStr),
48 nullptr, 0, NI_NUMERICHOST);
49 if (ret != 0) {
50 UPnPsdk_LOGERR "MSG1036: Failed to get netaddress with address family "
51 << std::to_string(a_sockaddr.ss_family) << ": "
52 << ::gai_strerror(ret)
53 << ". Continue with empty netaddress \"\".\n";
54 return "";
55 }
56
57 // Next throws 'std::length_error' if the length of the constructed
58 // std::string would exceed max_size(). This should never happen with given
59 // lengths of addrStr (promise noexcept).
60 if (a_sockaddr.ss_family == AF_INET6)
61 return '[' + std::string(addrStr) + ']';
62
63 return std::string(addrStr);
64}

78std::string to_netaddrp(const ::sockaddr_storage& a_sockaddr) noexcept {
79 // TRACE("Executing to_addrport_str()") // not usable in chained output.
80 //
81 // sin_port and sin6_port are on the same memory location (union of the
82 // structures) so I can use it for AF_INET and AF_INET6.
83 // 'std::to_string()' may throw 'std::bad_alloc' from the std::string
84 // constructor. It is a fatal error that violates the promise to noexcept
85 // and immediately terminates the propgram. This is intentional because the
86 // error cannot be handled.
87 std::string netaddr{to_netaddr(a_sockaddr)}; // noexcept, has limited length
88 return netaddr.empty()
89 ? ""
90 : netaddr + ":" +
91 std::to_string(ntohs(
92 reinterpret_cast<const ::sockaddr_in6*>(&a_sockaddr)
93 ->sin6_port));
94}

97// Free function to logical compare two sockaddr structures
98// --------------------------------------------------------
109bool sockaddrcmp(const ::sockaddr_storage* a_ss1,
110 const ::sockaddr_storage* a_ss2) noexcept {
111 // To have a logical equal socket address we compare the address family,
112 // the ip address, the port and the scope.
113 // Throws no exception.
114 if (a_ss1 == nullptr && a_ss2 == nullptr)
115 return true;
116 if (a_ss1 == nullptr || a_ss2 == nullptr)
117 return false;
118
119 switch (a_ss1->ss_family) {
120 case AF_UNSPEC:
121 if (a_ss2->ss_family != AF_UNSPEC)
122 return false;
123 break;
124
125 case AF_INET6: {
126 // I compare ipv6 addresses which are stored in a 16 byte array
127 // (unsigned char s6_addr[16]). So we have to use memcmp() for
128 // comparison.
129 const ::sockaddr_in6* const s6_addr1 =
130 reinterpret_cast<const ::sockaddr_in6*>(a_ss1);
131 const ::sockaddr_in6* const s6_addr2 =
132 reinterpret_cast<const ::sockaddr_in6*>(a_ss2);
133
134 if (a_ss2->ss_family != AF_INET6 ||
135 ::memcmp(&s6_addr1->sin6_addr, &s6_addr2->sin6_addr,
136 sizeof(in6_addr)) != 0 ||
137 s6_addr1->sin6_port != s6_addr2->sin6_port ||
138 s6_addr1->sin6_scope_id != s6_addr2->sin6_scope_id)
139 return false;
140 } break;
141
142 case AF_INET: {
143 const ::sockaddr_in* const s_addr1 =
144 reinterpret_cast<const ::sockaddr_in*>(a_ss1);
145 const ::sockaddr_in* const s_addr2 =
146 reinterpret_cast<const ::sockaddr_in*>(a_ss2);
147
148 if (a_ss2->ss_family != AF_INET ||
149 s_addr1->sin_addr.s_addr != s_addr2->sin_addr.s_addr ||
150 s_addr1->sin_port != s_addr2->sin_port)
151 return false;
152 } break;
153
154 default:
155 return false;
156 }
157
158 return true;
159}

164// Free function to get the port number from a string
165// --------------------------------------------------
166in_port_t to_port(const std::string& a_port_str) {
167 TRACE("Executing to_port() with port=\"" + a_port_str + "\"")
168 bool nonzero{false};
169 int port;
170
171 // Only non empty strings. I have to check this to avoid stoi() exception
172 // below.
173 if (a_port_str.empty())
174 // An empty port string is defined to be the unknown port 0.
175 return 0;
176
177 // Now we check if the string are all digit characters
178 for (char ch : a_port_str) {
179 if (!std::isdigit(static_cast<unsigned char>(ch))) {
180 goto exit_fail;
181 } else if (ch != '0') {
182 nonzero = true;
183 }
184 }
185
186 // Only strings with max. 5 char may be valid (uint16_t has max. 65535).
187 if (a_port_str.length() > 5) {
188 if (nonzero)
189 goto exit_overrun; // value valid but more than 5 char.
190 else
191 goto exit_fail; // string is all zero with more than 5 char.
192 }
193
194 // Valid positive number but is it within the port range (uint16_t)?
195 // stoi() may throw std::invalid_argument if no conversion could be
196 // performed or std::out_of_range. But with the prechecked number string
197 // this should never be thrown.
198 port = std::stoi(a_port_str);
199 if (port > 65535)
200 goto exit_overrun;
201
202 // Type cast from int is no problem because the port value is checked to be
203 // 0..65535 so it always fit into in_port_t(uint16_t).
204 return static_cast<in_port_t>(port);
205
206
207// Both following exceptions are derived from std::logic_error() so that can be
208// used to catch any of them.
209exit_overrun:
210 throw std::out_of_range(UPnPsdk_LOGEXCEPT +
211 "MSG1127: Valid number string \"" + a_port_str +
212 "\" is out of port range 0..65535.");
213exit_fail:
214 throw std::invalid_argument(
215 UPnPsdk_LOGEXCEPT + "MSG1128: Failed to get port number for string \"" +
216 a_port_str + "\".");
217}

257// Assignment operator= to set socket address from string,
258// -------------------------------------------------------
259// For port conversion:
260// Don't use '::htons' (with colons) instead of 'htons', MacOS don't like it.
261// 'sin6_port' is also 'sin_port' due to union.
262void SSockaddr::operator=(const std::string& a_addr_str) {
263 // Valid input examles: "[2001:db8::1]", "[2001:db8::1]:50001",
264 // "192.168.1.1", "192.168.1.1:50001".
265 TRACE2(this, " Executing SSockaddr::operator=()")
266
267 // An empty address string clears the address storage.
268 if (a_addr_str.empty()) {
269 ::memset(&m_sa_union, 0, sizeof(m_sa_union));
270 return;
271 }
272
273 if (a_addr_str.front() == '[') { // IPv6 address
274 if (a_addr_str.back() == ']') { // IPv6 address without port
275 this->handle_ipv6(a_addr_str);
276
277 } else { // IPv6 with port
278 // Split address and port
279 size_t pos = a_addr_str.find("]:");
280 this->handle_ipv6(a_addr_str.substr(0, pos + 1));
281 sin6.sin6_port = htons(to_port(a_addr_str.substr(pos + 2)));
282 }
283 } else { // IPv4 address or port
284 size_t pos = a_addr_str.find_first_of(':');
285 if (pos == 0 && a_addr_str.size() > 1 && ss.ss_family != AF_UNSPEC) {
286 // Special case: only port with leading ':' is valid to change port
287 // of existing socket address.
288 sin6.sin6_port = htons(to_port(a_addr_str.substr(pos + 1)));
289
290 } else if (pos != std::string::npos) { // ':' found means ipv4 with port
291 this->handle_ipv4(a_addr_str.substr(0, pos));
292 sin6.sin6_port = htons(to_port(a_addr_str.substr(pos + 1)));
293
294 } else { // IPv4 without port or port only
295 if (a_addr_str.find_first_of(".") != std::string::npos) {
296 this->handle_ipv4(a_addr_str); // IPv4 address without port
297
298 } else { // port only
299 sin6.sin6_port = htons(to_port(a_addr_str));
300 }
301 }
302 }
