17// Free function to check for a netaddress without port
18// ----------------------------------------------------
19// I use the system function ::getaddrinfo() to check if the node string is
20// acceptable. Using ::getaddrinfo() is needed to cover all special address
21// features like scope id for link local addresses and so on.
22sa_family_t is_netaddr(const std::string& a_node,
23 const int a_addr_family) noexcept {
24 // clang-format off
25 TRACE("Executing is_netaddr(\"" + a_node + "\", " +
26 (a_addr_family == AF_INET6 ? "AF_INET6" :
27 (a_addr_family == AF_INET ? "AF_INET" :
28 (a_addr_family == AF_UNSPEC ? "AF_UNSPEC" :
29 std::to_string(a_addr_family)))) + ")")
30 // clang-format on
31
32 // The shortest numeric netaddress is "[::]".
33 if (a_node.size() < 4) // noexcept
34 return AF_UNSPEC;
35
36 // Provide resources for ::getaddrinfo()
37 // AI_NUMERICHOST ensures that only numeric addresses accepted.
38 std::string node;
39 ::addrinfo hints{};
40 hints.ai_flags = AI_NUMERICHOST;
41 hints.ai_family = a_addr_family;
42 ::addrinfo* res{nullptr};
43
44 // Check for ipv6 addresses and remove surounding brackets for
45 // ::getaddrinfo().
46 // front() and back() have undefined behavior with an empty string. Here
47 // its size() is at least 4. Substr() throws exception out of range if pos
48 // > size(). All this means we have noexcept here.
49 if (a_node.front() == '[' && a_node.back() == ']' &&
50 (a_addr_family == AF_UNSPEC || a_addr_family == AF_INET6)) {
51 node = a_node.substr(1, a_node.length() - 2);
52 hints.ai_family = AF_INET6;
53
54 } else if (a_node.find_first_of(":") != a_node.npos) {
55 // Ipv6 addresses are already checked and here are only ipv4 addresses
56 // and URL names possible. Both are not valid if they contain a colon.
57 // find_first_of() does not throw an exception.
58 return AF_UNSPEC;
59
60 } else {
61 node = a_node;
62 }
63
64 // Call ::getaddrinfo() to check the remaining node string.
65 int errcode = ::getaddrinfo(node.c_str(), nullptr, &hints, &res);
66 if (errcode != 0) {
67 TRACE("syscall ::getaddrinfo() returned \"" +
68 std::string(::gai_strerror(errcode)) + "\"")
69 freeaddrinfo(res);
70 return AF_UNSPEC;
71 }
72 TRACE("syscall ::getaddrinfo()")
73
74 int af_family = res->ai_family;
75 freeaddrinfo(res);
76 // Guard different types on different platforms (win32); need to cast to
77 // af_family (unsigned short).
78 if (af_family < 0 || af_family > 65535)
79 return AF_UNSPEC;
80 else
81 return static_cast<sa_family_t>(af_family);
82}

85// Free function to check if a string is a valid port number
86// ---------------------------------------------------------
87bool is_port(const std::string& a_port_str) noexcept {
88 TRACE("Executing is_port() with port=\"" + a_port_str + "\"")
89
90 // Only non empty strings
91 // and strings with max. 5 characters are valid (uint16_t has max.
92 // 65535)
93 if (a_port_str.empty() || a_port_str.length() > 5)
94 return false;
95
96 // Now we check if the string are all digit characters
97 for (char ch : a_port_str) {
98 if (!::isdigit(static_cast<unsigned char>(ch)))
99 return false;
100 }
101 // Valid positive number but is it within the port range (uint16_t)?
102 // stoi may throw std::invalid_argument if no conversion could be
103 // performed or std::out_of_range. But with the prechecked number string
104 // this should never be thrown.
105 return (std::stoi(a_port_str) <= 65535) ? true : false;
106}

119// Assignment operator to set a netaddress
120// ---------------------------------------
121void Netaddr::operator=(const std::string& a_addr_str) noexcept {
122 TRACE2(this, " Executing Netaddr=(), assign netaddress")
123
124 m_netaddrp.clear(); // noexcept
125
126 try {
127 if (is_netaddr(a_addr_str) != AF_UNSPEC) { // noexcept
128 // Here we have a valid netaddress without port.
129 m_netaddrp = a_addr_str + ":0"; // maybe throw std::length_error?
130 return;
131 }
132 // Following a valid netaddress must have a port.
133
134 // Look for a port part or an ambiguous netaddress.
135 size_t pos = a_addr_str.find_last_of(':'); // noexcept
136 if (pos == std::string::npos || a_addr_str == "[::1") {
137 // No port detected. The address string cannot be a netaddress.
138 return;
139 }
140
141 // Split address string into possible netaddress and port and check
142 // them. substr() throws exception std::out_of_range if pos > size().
143 // pos is 0-based, size() is 1-based so with ":" is size()==1, pos==0,
144 // pos+1==size() does not throw.
145 std::string netaddr = a_addr_str.substr(0, pos);
146 if (is_netaddr(netaddr) == AF_UNSPEC) {
147 // Unspecified netaddress
148 return;
149 }
150 std::string port = a_addr_str.substr(pos + 1);
151 if (!is_port(port)) { // noexcept
152 // Unspecified port
153 port = "0";
154 }
155
156 // Set netaddress with port.
157 if (!(netaddr.empty() && port == "0")) { // noexcept
158 m_netaddrp = netaddr + ":" + port; // maybe throw std::length_error?
159 }
160 return;
161
162 } catch (const std::length_error&) {
163 // If a resulting string is too long then I just return with an
164 // unspecified netaddress set.
165 return;
166 }
167}
