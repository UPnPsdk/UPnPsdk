46uint8_t netmask_to_bitmask(const ::sockaddr_storage* a_netmask) {
47 TRACE("Executing netmask_to_bitmask()")
48 if (a_netmask == nullptr)
49 throw std::runtime_error(UPnPsdk_LOGEXCEPT(
50 "MSG1059") "No network socket address information given.\n");
51
52 uint8_t bitmask{};
53 uint8_t nullbits{};
54 switch (a_netmask->ss_family) {
55 case AF_INET6: {
56 // Count set bits.
57 for (size_t i{}; i < sizeof(in6_addr); i++) {
58 uint8_t s6addr = reinterpret_cast<const ::sockaddr_in6*>(a_netmask)
59 ->sin6_addr.s6_addr[i];
60 if (s6addr == 255) {
61 bitmask += 8;
62 } else {
63 while (s6addr) {
64 bitmask++;
65 s6addr <<= 1;
66 }
67 break; // for() loop
68 }
69 }
70 // Check if all remaining bits are zero.
71 for (int i{sizeof(in6_addr) - 1}; i >= 0; i--) {
72 uint8_t s6addr = reinterpret_cast<const ::sockaddr_in6*>(a_netmask)
73 ->sin6_addr.s6_addr[i];
74 if (s6addr == 0) {
75 nullbits += 8;
76 } else {
77 while ((s6addr & 1) == 0) {
78 nullbits++;
79 s6addr >>= 1;
80 }
81 break; // for() loop
82 }
83 }
84 // Check valid netmask.
85 if (bitmask + nullbits != 128) {
86 char ip_str[INET6_ADDRSTRLEN]{};
87 ::inet_ntop(
88 AF_INET6,
89 &reinterpret_cast<const ::sockaddr_in6*>(a_netmask)->sin6_addr,
90 ip_str, sizeof(ip_str));
91 throw std::runtime_error(
92 UPnPsdk_LOGEXCEPT(
93 "MSG1067") "Invalid ip-address prefix bitmask \"[" +
94 std::string(ip_str) + "]\".\n");
95 }
96 } break; // switch()
97
98 case AF_INET: {
99 in_addr_t saddr = ntohl(
100 reinterpret_cast<const ::sockaddr_in*>(a_netmask)->sin_addr.s_addr);
101 in_addr_t saddr0 = saddr;
102 // Count set bits.
103 while (saddr) {
104 bitmask++;
105 saddr <<= 1;
106 }
107 // Check if all remaining bits are zero.
108 for (int i{}; (saddr0 & 1) == 0 && i < 32; i++) {
109 nullbits++;
110 saddr0 >>= 1;
111 }
112 // Check valid netmask.
113 if (bitmask + nullbits != 32) {
114 char ip_str[INET_ADDRSTRLEN]{};
115 ::inet_ntop(
116 AF_INET,
117 &reinterpret_cast<const ::sockaddr_in*>(a_netmask)->sin_addr,
118 ip_str, sizeof(ip_str));
119 throw std::runtime_error(
120 UPnPsdk_LOGEXCEPT("MSG1069") "Invalid ip-address netmask \"" +
121 std::string(ip_str) + "\".\n");
122 }
123 } break;
124
125 case AF_UNSPEC:
126 return 0;
127
128 default:
129 throw std::runtime_error(
130 UPnPsdk_LOGEXCEPT("MSG1028") "Unsupported address family(" +
131 std::to_string(a_netmask->ss_family) + "), only AF_INET6(" +
132 std::to_string(AF_INET6) + "), AF_INET(" + std::to_string(AF_INET) +
133 "), or AF_UNSPEC(" + std::to_string(AF_UNSPEC) + ") are valid.\n");
134 } // switch()
135
136 return bitmask;
137}

140void bitmask_to_netmask(const ::sockaddr_storage* a_saddr,
141 const unsigned int a_prefixlength,
142 SSockaddr& a_saddrObj) {
143 TRACE("Executing bitmask_to_netmask()")
144 if (a_saddr == nullptr)
145 throw std::runtime_error(UPnPsdk_LOGEXCEPT(
146 "MSG1070") "No associated socket address for the netmask given.\n");
147
148 switch (a_saddr->ss_family) {
149 case AF_INET6: {
150 // I have to manage 16 bytes from the binary IPv6 address to set its
151 // bits. 15 bytes are all ones or all zero bits. Only one byte may have
152 // partly ones and zero bits. First I calculate how many leading bytes
153 // have full one bits. Following is the partial set byte. The rest are
154 // bytes with full zero bits set.
155 in6_addr netmask6{};
156
157 // All prefix lengths > 128 will throw this exception.
158 if (a_prefixlength > 128)
159 throw std::runtime_error(
160 UPnPsdk_LOGEXCEPT(
161 "MSG1124") "Invalid IPv6 address prefix bitmask(" +
162 std::to_string(a_prefixlength) + ") exceeds 128.\n");
163
164 // Calculate number of leading bytes with full one bits.
165 unsigned int ones_bytes{a_prefixlength / 8};
166
167 // Fill leading bytes with all bit ones.
168 unsigned int i{};
169 for (; i < ones_bytes; i++)
170 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
171
172 // Handle the one partly bit-set byte.
173 if (i < 16 /*bytes*/) {
174 // Preset byte with all bit ones.
175 netmask6.s6_addr[i] = static_cast<uint8_t>(~0);
176 // Shift remaining zero bits from right into byte with using the
177 // reminder from the byte devision.
178 netmask6.s6_addr[i] <<= (8 - (a_prefixlength % 8));
179 i++;
180
181 // Fill remaining bytes with zero bits.
182 for (; i < 16 /*bytes*/; i++)
183 netmask6.s6_addr[i] = 0;
184 }
185
186 // Return the result.
187 a_saddrObj = "";
188 a_saddrObj.sin6.sin6_family = AF_INET6;
189 memcpy(&a_saddrObj.sin6.sin6_addr, &netmask6,
190 sizeof(a_saddrObj.sin6.sin6_addr));
191 } break;
192
193 case AF_INET: {
194 // All prefix lengths > 32 will throw this exception.
195 if (a_prefixlength > 32) {
196 throw std::runtime_error(
197 UPnPsdk_LOGEXCEPT(
198 "MSG1125") "Invalid IPv4 address prefix bitmask(" +
199 std::to_string(a_prefixlength) + ") exceeds 32.\n");
200 }
201 // Prepare socket address and shift zero bits from right into the mask.
202 ::sockaddr_in saddr{};
203 saddr.sin_family = AF_INET;
204 if (a_prefixlength) {
205 in_addr_t bitshift{~0u}; // all one bits
206 bitshift <<= (32 - a_prefixlength); // shift zero bits from right
207 saddr.sin_addr.s_addr = htonl(bitshift);
208 }
209 // Return the result.
210 a_saddrObj.sin = saddr;
211 } break;
212
213 case AF_UNSPEC: {
214 a_saddrObj = "";
215 } break;
216
217 default: {
218 throw std::runtime_error(
219 UPnPsdk_LOGEXCEPT("MSG1126") "Unsupported address family(" +
220 std::to_string(a_saddr->ss_family) + "), only AF_INET6(" +
221 std::to_string(AF_INET6) + "), AF_INET(" + std::to_string(AF_INET) +
222 "), or AF_UNSPEC(" + std::to_string(AF_UNSPEC) + ") are valid.\n");
223 }
224 } // switch
225}

262void CNetadapter::sockaddr(SSockaddr& a_saddr) const {
263 SSockaddr saddr;
264 m_na_platformPtr->sockaddr(saddr);
265
266 // If there is an IPv4 address then map it to IPv6.
267 if (saddr.ss.ss_family == AF_INET) {
268 // We will get something like "[::ffff:192.168.1.2]:49494".
269 SSockaddr saddr4to6;
270 saddr4to6.sin6.sin6_family = AF_INET6;
271 saddr4to6.sin6.sin6_port = saddr.sin.sin_port;
272 in6_addr& sa6{saddr4to6.sin6.sin6_addr};
273 sa6.s6_addr[10] = 0xff; // prefix id for ipv4 mapping.
274 sa6.s6_addr[11] = 0xff; // ./.
275 // Copy netorder binary ipv4 addr into ipv6 addr behind prefix.
276 ::memcpy(&sa6.s6_addr[12], &saddr.sin.sin_addr,
277 sizeof(saddr.sin.sin_addr));
278
279 a_saddr = saddr4to6;
280 } else {
281 a_saddr = saddr;
282 }
283}

306bool CNetadapter::find_first(std::string_view a_name_or_addr) {
307 TRACE2(this, " Executing CNetadapter::find_first()")
308
309 SSockaddr nad_saObj;
310 this->reset(); // noexcept
311
312 // By default look for a valid ip address without loopback, no matter what
313 // local network adapter.
314 // -----------------------------------------------------------------------
315 // The operating system presents one as best choise.
316 if (a_name_or_addr.empty()) {
317 if (this->index() == 0)
318 return false; // There isn't any adapter.
319
320 uint32_t index;
321 do {
322 this->sockaddr(nad_saObj);
323 index = this->index();
324 // To verify mocked expectations for Unit Test next cout is needed.
325 /* std::cout << "DEBUG: find_first() index=" << index
326 << ", netaddrp()=\"" << nad_saObj.netaddrp() << "\"\n"; */
327 if (index != m_index_loop &&
328 !IN6_IS_ADDR_V4MAPPED(&nad_saObj.sin6.sin6_addr)) {
329 m_find_flags = ADDRS::best;
330 return true;
331 }
332 } while (this->get_next());
333
334 // No usable address found, nothing more to do.
335 return false;
336 }
337
338 // Look for the ip address of a local network adapter.
339 // ---------------------------------------------------
340 // Last attempt to get a socket address from the input argument.
341 if (a_name_or_addr.front() == '[') {
342 SSockaddr input_saObj;
343 try { // Throws exception if invalid.
344 input_saObj = std::string_view(a_name_or_addr);
345 } catch (std::exception&) {
346 return false;
347 }
348 // Parse network adapter list for the given input ip address.
349 do {
350 this->sockaddr(nad_saObj);
351 // To verify mocked expectations for Unit Test next cout is needed.
352 /* std::cout << "DEBUG: find_first() a_name_or_addr=\"" //
353 << a_name_or_addr << "\", adapter_addr=\""
354 << nad_saObj.netaddrp() << "\"\n"; */
355 if (nad_saObj == input_saObj) {
356 // With a unique ip address given, there cannot be another.
357 m_find_flags = ADDRS::none;
358 return true;
359 }
360 } while (this->get_next());
361
362 // No usable address found, nothing more to do.
363 return false;
364 }
365
366 // Look for a local network adapter name that has at least one ip address.
367 // Also the loopback interface is accepted.
368 // ----------------------------------------------------------------------
369 do {
370 auto name = this->name();
371 // To verify mocked expectations for Unit Test next cout is needed.
372 /* std::cout << "DEBUG: find_first() a_name_or_addr=\"" //
373 << a_name_or_addr << "\", name=\"" << name << "\"\n"; */
374 if (name == a_name_or_addr) {
375 this->sockaddr(nad_saObj);
376 if (!IN6_IS_ADDR_V4MAPPED(&nad_saObj.sin6.sin6_addr)) {
377 m_index_find = this->index();
378 m_find_flags = ADDRS::index;
379 return true;
380 }
381 }
382 } while (this->get_next());
383
384 // No usable address found.
385 return false;
386}

416 TRACE2(this, " Executing CNetadapter::find_first(a_flags)")

418 if (this->index() == 0)
419 return false; // There isn't any adapter.
420
421 // Look for filtered addresses, no matter what local network adapter.
422 // ------------------------------------------------------------------
423 m_find_flags = a_flags;

442 return true;

445 IN6_IS_ADDR_V4MAPPED(&saObj.sin6.sin6_addr) &&
446 this->index() != m_index_loop) {
447 return true;
448 }
449 } while (this->get_next());

451 // No matching interface found, nothing more to do.
452 return false;

455bool CNetadapter::find_next() {
456 TRACE2(this, " Executing CNetadapter::find_next()")
457 if (m_find_flags == ADDRS::none)
458 return false;
459
460 SSockaddr saObj;
461 while (this->get_next()) {
462 this->sockaddr(saObj);
463 // To verify mocked expectations for Unit Test next cout is needed.
464 /* std::cout << "DEBUG: find_next() netaddrp()=\"" << saObj.netaddrp()
465 << "\"\n"; */
466 if ((m_find_flags & ADDRS::best) != ADDRS::none &&
467 !IN6_IS_ADDR_V4MAPPED(&saObj.sin6.sin6_addr) &&
468 this->index() != m_index_loop) {
469 return true;
470 }
471 if ((m_find_flags & ADDRS::index) != ADDRS::none &&
472 this->index() == m_index_find &&
473 !IN6_IS_ADDR_V4MAPPED(&saObj.sin6.sin6_addr)) {
474 return true;
475 }
476 if ((m_find_flags & ADDRS::lo) != ADDRS::none &&
477 IN6_IS_ADDR_LOOPBACK(&saObj.sin6.sin6_addr)) {
478 return true;
479 }
480 if ((m_find_flags & ADDRS::lla) != ADDRS::none &&
481 IN6_IS_ADDR_LINKLOCAL(&saObj.sin6.sin6_addr) &&
482 this->index() != m_index_loop) {
483 return true;
484 }
485 if ((m_find_flags & ADDRS::gua) != ADDRS::none &&
486 IN6_IS_ADDR_GLOBAL(&saObj.sin6.sin6_addr) &&
487 this->index() != m_index_loop) {
488 return true;
489 }
