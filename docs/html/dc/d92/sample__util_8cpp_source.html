76int SampleUtil_Initialize(print_string print_function) {
77 TRACE("Executing SampleUtil_Initialize()")
78 if (initialize_init) {
79 pthread_mutexattr_t attr;
80
81 pthread_mutexattr_init(&attr);
82 pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
83 pthread_mutex_init(&display_mutex, &attr);
84 pthread_mutexattr_destroy(&attr);
85 /* To shut up valgrind mutex warning. */
86 pthread_mutex_lock(&display_mutex);
87 gPrintFun = print_function;
88 pthread_mutex_unlock(&display_mutex);
89 /* Finished initializing. */
90 initialize_init = 0;
91 }
92
93 return UPNP_E_SUCCESS;
94}

200char* SampleUtil_GetFirstDocumentItem(IXML_Document* doc, const char* item) {
201 IXML_NodeList* nodeList = NULL;
202 IXML_Node* textNode = NULL;
203 IXML_Node* tmpNode = NULL;
204 const char* nodeValue = NULL;
205 char* ret = NULL;
206
207 nodeList = ixmlDocument_getElementsByTagName(doc, (char*)item);
208 if (nodeList) {
209 tmpNode = ixmlNodeList_item(nodeList, 0);
210 if (tmpNode) {
211 textNode = ixmlNode_getFirstChild(tmpNode);
212 if (!textNode) {
213 SampleUtil_Print("%s(%d): (BUG) "
214 "ixmlNode_getFirstChild("
215 "tmpNode) returned NULL\n",
216 __FILE__, __LINE__);
217 ret = strdup("");
218 goto epilogue;
219 }
220 nodeValue = ixmlNode_getNodeValue(textNode);
221 if (!nodeValue) {
222 SampleUtil_Print("%s(%d): ixmlNode_getNodeValue "
223 "returned NULL\n",
224 __FILE__, __LINE__);
225 ret = strdup("");
226 goto epilogue;
227 }
228 ret = strdup(nodeValue);
229 if (!ret) {
230 SampleUtil_Print("%s(%d): Error allocating "
231 "memory for XML Node value\n",
232 __FILE__, __LINE__);
233 ret = strdup("");
234 }
235 } else
236 SampleUtil_Print("%s(%d): ixmlNodeList_item(nodeList, "
237 "0) returned NULL\n",
238 __FILE__, __LINE__);
239 } // else
240 // SampleUtil_Print("%s(%d): Error finding %s in XML Node\n", __FILE__,
241 // __LINE__, item);
242
243epilogue:
244 if (nodeList)
245 ixmlNodeList_free(nodeList);
246
247 return ret;
248}

250char* SampleUtil_GetFirstElementItem(IXML_Element* element, const char* item) {
251 IXML_NodeList* nodeList = NULL;
252 IXML_Node* textNode = NULL;
253 IXML_Node* tmpNode = NULL;
254 char* ret = NULL;
255
256 nodeList = ixmlElement_getElementsByTagName(element, (char*)item);
257 if (nodeList == NULL) {
258 SampleUtil_Print("%s(%d): Error finding %s in XML Node\n", __FILE__,
259 __LINE__, item);
260 return NULL;
261 }
262 tmpNode = ixmlNodeList_item(nodeList, 0);
263 if (!tmpNode) {
264 SampleUtil_Print("%s(%d): Error finding %s value in XML Node\n",
265 __FILE__, __LINE__, item);
266 ixmlNodeList_free(nodeList);
267 return NULL;
268 }
269 textNode = ixmlNode_getFirstChild(tmpNode);
270 ret = strdup(ixmlNode_getNodeValue(textNode));
271 if (!ret) {
272 SampleUtil_Print("%s(%d): Error allocating memory for %s in XML Node\n",
273 __FILE__, __LINE__, item);
274 ixmlNodeList_free(nodeList);
275 return NULL;
276 }
277 ixmlNodeList_free(nodeList);
278
279 return ret;
280}

571int SampleUtil_FindAndParseService(IXML_Document* DescDoc, const char* location,
572 const char* serviceType, char** serviceId,
573 char** eventURL, char** controlURL) {
574 unsigned int i;
575 unsigned long length;
576 int found = 0;
577 int ret;
578#ifdef OLD_FIND_SERVICE_CODE
579#else /* OLD_FIND_SERVICE_CODE */
580 unsigned int sindex = 0;
581#endif /* OLD_FIND_SERVICE_CODE */
582 char* tempServiceType = NULL;
583 char* baseURL = NULL;
584 const char* base = NULL;
585 char* relcontrolURL = NULL;
586 char* releventURL = NULL;
587 IXML_NodeList* serviceList = NULL;
588 IXML_Element* service = NULL;
589
590 baseURL = SampleUtil_GetFirstDocumentItem(DescDoc, "URLBase");
591 if (baseURL)
592 base = baseURL;
593 else
594 base = location;
595#ifdef OLD_FIND_SERVICE_CODE
596 serviceList = SampleUtil_GetFirstServiceList(DescDoc);
597#else /* OLD_FIND_SERVICE_CODE */
598 for (sindex = 0;
599 (serviceList = SampleUtil_GetNthServiceList(DescDoc, sindex)) != NULL;
600 sindex++) {
601 tempServiceType = NULL;
602 relcontrolURL = NULL;
603 releventURL = NULL;
604 service = NULL;
605#endif /* OLD_FIND_SERVICE_CODE */
606 length = ixmlNodeList_length(serviceList);
607 for (i = 0; i < length; i++) {
608 service = (IXML_Element*)ixmlNodeList_item(serviceList, i);
609 tempServiceType = SampleUtil_GetFirstElementItem((IXML_Element*)service,
610 "serviceType");
611 if (tempServiceType && strcmp(tempServiceType, serviceType) == 0) {
612 SampleUtil_Print("Found service: %s\n", serviceType);
613 *serviceId = SampleUtil_GetFirstElementItem(service, "serviceId");
614 SampleUtil_Print("serviceId: %s\n", *serviceId);
615 relcontrolURL =
616 SampleUtil_GetFirstElementItem(service, "controlURL");
617 releventURL =
618 SampleUtil_GetFirstElementItem(service, "eventSubURL");
619 ret = UpnpResolveURL2(base, relcontrolURL, controlURL);
620 if (ret != UPNP_E_SUCCESS)
621 SampleUtil_Print("Error generating controlURL "
622 "from %s + %s\n",
623 base, relcontrolURL);
624 ret = UpnpResolveURL2(base, releventURL, eventURL);
625 if (ret != UPNP_E_SUCCESS)
626 SampleUtil_Print("Error generating eventURL "
627 "from %s + %s\n",
628 base, releventURL);
629 free(relcontrolURL);
630 free(releventURL);
631 relcontrolURL = NULL;
632 releventURL = NULL;
633 found = 1;
634 break;
635 }
636 free(tempServiceType);
637 tempServiceType = NULL;
638 }
639 free(tempServiceType);
640 tempServiceType = NULL;
641 if (serviceList)
642 ixmlNodeList_free(serviceList);
643 serviceList = NULL;
644#ifdef OLD_FIND_SERVICE_CODE
645#else /* OLD_FIND_SERVICE_CODE */
646 }
647#endif /* OLD_FIND_SERVICE_CODE */
648 free(baseURL);
649
650 return found;
651}

653int SampleUtil_Print(const char* fmt, ...) {
655#define MAX_BUF (8 * 1024)
657 va_list ap;
658 static char buf[MAX_BUF];
659 int rc;
660
661 /* Protect both the display and the static buffer with the mutex */
662 pthread_mutex_lock(&display_mutex);
663
664 va_start(ap, fmt);
665 rc = vsnprintf(buf, MAX_BUF, fmt, ap);
666 va_end(ap);
667 if (gPrintFun)
668 gPrintFun("%s", buf);
669
670 pthread_mutex_unlock(&display_mutex);
671
672 return rc;
673}
